57 /// Return true if there is a path from start operation to dest operation,
58 /// otherwise return false. The operations have to be in the same region.
59 bool TransferOptimization::isReachable(Operation *start, Operation *dest) {
60  assert(start->getParentRegion() == dest->getParentRegion() &&
61  "This function only works for ops i the same region");
62  // Simple case where the start op dominate the destination.
63  if (dominators.dominates(start, dest))
64  return true;
65  Block *startBlock = start->getBlock();
66  Block *destBlock = dest->getBlock();
67  SmallVector<Block *, 32> worklist(startBlock->succ_begin(),
68  startBlock->succ_end());
69  SmallPtrSet<Block *, 32> visited;
70  while (!worklist.empty()) {
71  Block *bb = worklist.pop_back_val();
72  if (!visited.insert(bb).second)
73  continue;
74  if (dominators.dominates(bb, destBlock))
75  return true;
76  worklist.append(bb->succ_begin(), bb->succ_end());
77  }
78  return false;
79 }

81 /// For transfer_write to overwrite fully another transfer_write must:
82 /// 1. Access the same memref with the same indices and vector type.
83 /// 2. Post-dominate the other transfer_write operation.
84 /// If several candidates are available, one must be post-dominated by all the
85 /// others since they are all post-dominating the same transfer_write. We only
86 /// consider the transfer_write post-dominated by all the other candidates as
87 /// this will be the first transfer_write executed after the potentially dead
88 /// transfer_write.
89 /// If we found such an overwriting transfer_write we know that the original
90 /// transfer_write is dead if all reads that can be reached from the potentially
91 /// dead transfer_write are dominated by the overwriting transfer_write.
92 void TransferOptimization::deadStoreOp(vector::TransferWriteOp write) {
93  LLVM_DEBUG(DBGS() << "Candidate for dead store: " << *write.getOperation()
94  << "\n");
95  llvm::SmallVector<Operation *, 8> reads;
96  Operation *firstOverwriteCandidate = nullptr;
97  for (auto *user : write.source().getUsers()) {
98  if (user == write.getOperation())
99  continue;
100  if (auto nextWrite = dyn_cast<vector::TransferWriteOp>(user)) {
101  // Check candidate that can override the store.
102  if (checkSameValueWAW(nextWrite, write) &&
103  postDominators.postDominates(nextWrite, write)) {
104  if (firstOverwriteCandidate == nullptr ||
105  postDominators.postDominates(firstOverwriteCandidate, nextWrite))
106  firstOverwriteCandidate = nextWrite;
107  else
108  assert(
109  postDominators.postDominates(nextWrite, firstOverwriteCandidate));
110  }
111  } else {
112  if (auto read = dyn_cast<vector::TransferReadOp>(user)) {
113  // Don't need to consider disjoint reads.
114  if (isDisjointTransferSet(
115  cast<VectorTransferOpInterface>(write.getOperation()),
116  cast<VectorTransferOpInterface>(read.getOperation())))
117  continue;
118  }
119  reads.push_back(user);
120  }
121  }
122  if (firstOverwriteCandidate == nullptr)
123  return;
124  Region *topRegion = firstOverwriteCandidate->getParentRegion();
125  Operation *writeAncestor = findAncestorOpInRegion(topRegion, write);
126  assert(writeAncestor &&
127  "write op should be recursively part of the top region");
128 
129  for (Operation *read : reads) {
130  Operation *readAncestor = findAncestorOpInRegion(topRegion, read);
131  // TODO: if the read and write have the same ancestor we could recurse in
132  // the region to know if the read is reachable with more precision.
133  if (readAncestor == nullptr || !isReachable(writeAncestor, readAncestor))
134  continue;
135  if (!dominators.dominates(firstOverwriteCandidate, read)) {
136  LLVM_DEBUG(DBGS() << "Store may not be dead due to op: " << *read
137  << "\n");
138  return;
139  }
140  }
141  LLVM_DEBUG(DBGS() << "Found dead store: " << *write.getOperation()
142  << " overwritten by: " << *firstOverwriteCandidate << "\n");
143  opToErase.push_back(write.getOperation());
144 }

146 /// A transfer_write candidate to storeToLoad forwarding must:
147 /// 1. Access the same memref with the same indices and vector type as the
148 /// transfer_read.
149 /// 2. Dominate the transfer_read operation.
150 /// If several candidates are available, one must be dominated by all the others
151 /// since they are all dominating the same transfer_read. We only consider the
152 /// transfer_write dominated by all the other candidates as this will be the
153 /// last transfer_write executed before the transfer_read.
154 /// If we found such a candidate we can do the forwarding if all the other
155 /// potentially aliasing ops that may reach the transfer_read are post-dominated
156 /// by the transfer_write.
157 void TransferOptimization::storeToLoadForwarding(vector::TransferReadOp read) {
158  if (read.hasOutOfBoundsDim())
159  return;
160  LLVM_DEBUG(DBGS() << "Candidate for Forwarding: " << *read.getOperation()
161  << "\n");
162  SmallVector<Operation *, 8> blockingWrites;
163  vector::TransferWriteOp lastwrite = nullptr;
164  for (Operation *user : read.source().getUsers()) {
165  if (isa<vector::TransferReadOp>(user))
166  continue;
167  if (auto write = dyn_cast<vector::TransferWriteOp>(user)) {
168  // If there is a write, but we can prove that it is disjoint we can ignore
169  // the write.
170  if (isDisjointTransferSet(
171  cast<VectorTransferOpInterface>(write.getOperation()),
172  cast<VectorTransferOpInterface>(read.getOperation())))
173  continue;
174  if (dominators.dominates(write, read) && checkSameValueRAW(write, read)) {
175  if (lastwrite == nullptr || dominators.dominates(lastwrite, write))
176  lastwrite = write;
177  else
178  assert(dominators.dominates(write, lastwrite));
179  continue;
180  }
181  }
182  blockingWrites.push_back(user);
183  }
184 
185  if (lastwrite == nullptr)
186  return;
187 
188  Region *topRegion = lastwrite->getParentRegion();
189  Operation *readAncestor = findAncestorOpInRegion(topRegion, read);
190  assert(readAncestor &&
191  "read op should be recursively part of the top region");
192 
193  for (Operation *write : blockingWrites) {
194  Operation *writeAncestor = findAncestorOpInRegion(topRegion, write);
195  // TODO: if the store and read have the same ancestor we could recurse in
196  // the region to know if the read is reachable with more precision.
197  if (writeAncestor == nullptr || !isReachable(writeAncestor, readAncestor))
198  continue;
199  if (!postDominators.postDominates(lastwrite, write)) {
200  LLVM_DEBUG(DBGS() << "Fail to do write to read forwarding due to op: "
201  << *write << "\n");
202  return;
203  }
204  }
205 
206  LLVM_DEBUG(DBGS() << "Forward value from " << *lastwrite.getOperation()
207  << " to: " << *read.getOperation() << "\n");
208  read.replaceAllUsesWith(lastwrite.vector());
209  opToErase.push_back(read.getOperation());
210 }

214 void mlir::vector::transferOpflowOpt(FuncOp func) {
215  TransferOptimization opt(func);
216  // Run store to load forwarding first since it can expose more dead store
217  // opportunity.
218  func.walk([&](vector::TransferReadOp read) {
219  if (read.getShapedType().isa<MemRefType>())
220  opt.storeToLoadForwarding(read);
221  });
222  opt.removeDeadOp();
223  func.walk([&](vector::TransferWriteOp write) {
224  if (write.getShapedType().isa<MemRefType>())
225  opt.deadStoreOp(write);
226  });
227  opt.removeDeadOp();
228 }
