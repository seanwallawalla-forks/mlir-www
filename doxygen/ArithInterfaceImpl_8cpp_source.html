23 struct ConstantOpInterface
24  : public BufferizableOpInterface::ExternalModel<ConstantOpInterface,
25  arith::ConstantOp> {
26  LogicalResult bufferize(Operation *op, OpBuilder &b,
27  BufferizationState &state) const {
28  auto constantOp = cast<arith::ConstantOp>(op);
29  assert(constantOp.getType().dyn_cast<RankedTensorType>() &&
30  "not a constant ranked tensor");
31  auto moduleOp = constantOp->getParentOfType<ModuleOp>();
32  if (!moduleOp)
33  return constantOp.emitError(
34  "cannot bufferize constants not within builtin.module op");
35 
36  GlobalCreator globalCreator(moduleOp);
37  auto globalMemref = globalCreator.getGlobalFor(constantOp);
38  Value memref = b.create<memref::GetGlobalOp>(
39  constantOp.getLoc(), globalMemref.type(), globalMemref.getName());
40  state.mapBuffer(constantOp, memref);
41 
42  return success();
43  }
