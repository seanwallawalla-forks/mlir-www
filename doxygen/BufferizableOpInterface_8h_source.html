182  /// llvm::EquivalenceClasses wants comparable elements. This comparator uses
183  /// uses pointer comparison on the defining op. This is a poor man's
184  /// comparison but it's not like UnionFind needs ordering anyway.
185  struct ValueComparator {
186  bool operator()(const Value &lhs, const Value &rhs) const {
187  return lhs.getImpl() < rhs.getImpl();
188  }
189  };

199  /// Auxiliary structure to store all the values a given value may alias with.
200  /// Alias information is "may be" conservative: In the presence of branches, a
201  /// value may alias with one of multiple other values. The concrete aliasing
202  /// value may not even be known at compile time. All such values are
203  /// considered to be aliases.
204  llvm::EquivalenceClasses<Value, ValueComparator> aliasInfo;
205 
206  /// Auxiliary structure to store all the equivalent buffer classes. Equivalent
207  /// buffer information is "must be" conservative: Only if two values are
208  /// guaranteed to be equivalent at runtime, they said to be equivalent. It is
209  /// possible that, in the presence of branches, it cannot be determined
210  /// statically if two values are equivalent. In that case, the values are
211  /// considered to be not equivalent.
212  llvm::EquivalenceClasses<Value, ValueComparator> equivalentInfo;
213 };

240 /// Starting from `value`, follow the use-def chain in reverse, always selecting
241 /// the aliasing OpOperands. Find and return Values for which `condition`
242 /// evaluates to true. OpOperands of such matching Values are not traversed any

244 ///

257 /// |
258 /// 1
259 ///

392 /// AllocationHoistingBarrierOnly is an external implementation of
393 /// BufferizableOpInterface for ops that are (not yet) bufferizable, but are
394 /// known to be allocation hoisting barriers. All interface methods (except for
395 /// `isAllocationHoistingBarrier`) are implemented conservatively.
396 template <typename OpTy>
397 struct AllocationHoistingBarrierOnly
398  : public BufferizableOpInterface::ExternalModel<
399  AllocationHoistingBarrierOnly<OpTy>, OpTy> {
400  bool bufferizesToMemoryRead(Operation *op, OpOperand &opOperand) const {
401  return true;
402  }
403 
404  bool bufferizesToMemoryWrite(Operation *op, OpOperand &opOperand) const {
405  return false;
406  }
407 
408  SmallVector<OpOperand *> getAliasingOpOperand(Operation *op,
409  OpResult opResult) const {
410  return {};
411  }
412 
413  OpResult getAliasingOpResult(Operation *op, OpOperand &opOperand) const {
414  return OpResult();
415  }
416 
417  BufferRelation bufferRelation(Operation *op, OpResult opResult,
418  const BufferizationAliasInfo &aliasInfo) const {
419  return BufferRelation::None;
420  }
421 
422  bool isWritable(Operation *op, Value value, BufferizationState &state) const {
423  return false;
424  }
425 
426  LogicalResult bufferize(Operation *op, OpBuilder &b,
427  BufferizationState &state) const {
428  auto isaTensor = [](Type t) { return t.isa<TensorType>(); };
429  if (any_of(op->getOperandTypes(), isaTensor) ||
430  any_of(op->getResultTypes(), isaTensor))
431  if (!state.options.allowUnknownOps)
432  return op->emitError() << "unsupported op with tensors";
433 
434  for (Region &region : op->getRegions())
435  if (failed(comprehensive_bufferize::bufferize(&region, state)))
436  return failure();
437 
438  return success();
439  }
440 
441  bool isAllocationHoistingBarrier(Operation *op) const { return true; }
442 };
