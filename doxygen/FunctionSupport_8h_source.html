103 /// This trait provides APIs for Ops that behave like functions. In particular:
104 /// - Ops must be symbols, i.e. also have the `Symbol` trait;
105 /// - Ops have a single region with multiple blocks that corresponds to the body
106 /// of the function;
107 /// - An op with a single empty region corresponds to an external function;
108 /// - leading arguments of the first block of the region are treated as function
109 /// arguments;
110 /// - they can have argument attributes that are stored in a dictionary
111 /// attribute on the Op itself.
112 ///
113 /// This trait provides limited type support for the declared or defined
114 /// functions. The convenience function `getTypeAttrName()` returns the name of
115 /// an attribute that can be used to store the function type. In addition, this
116 /// trait provides `getType` and `setType` helpers to store a `FunctionType` in
117 /// the attribute named by `getTypeAttrName()`.
118 ///
119 /// In general, this trait assumes concrete ops use `FunctionType` under the
120 /// hood. If this is not the case, in order to use the function type support,
121 /// concrete ops must define the following methods, using the same name, to hide
122 /// the ones defined for `FunctionType`: `addBodyBlock`, `getType`,
123 /// `getTypeWithoutArgsAndResults` and `setType`.
124 ///
125 /// Besides the requirements above, concrete ops must interact with this trait
126 /// using the following functions:
127 /// - Concrete ops *must* define a member function `getNumFuncArguments()` that
128 /// returns the number of function arguments based exclusively on type (so
129 /// that it can be called on function declarations).
130 /// - Concrete ops *must* define a member function `getNumFuncResults()` that
131 /// returns the number of function results based exclusively on type (so that
132 /// it can be called on function declarations).
133 /// - To verify that the type respects op-specific invariants, concrete ops may
134 /// redefine the `verifyType()` hook that will be called after verifying the
135 /// presence of the `type` attribute and before any call to
136 /// `getNumFuncArguments`/`getNumFuncResults` from the verifier.
137 /// - To verify that the body respects op-specific invariants, concrete ops may
138 /// redefine the `verifyBody()` hook that will be called after verifying the
139 /// function type and the presence of the (potentially empty) body region.
140 template <typename ConcreteType>

564 /// Default verifier checks that if the entry block exists, it has the same
565 /// number of arguments as the function-like operation.
566 template <typename ConcreteType>
567 LogicalResult FunctionLike<ConcreteType>::verifyBody() {
568  auto funcOp = cast<ConcreteType>(this->getOperation());
569 
570  if (funcOp.isExternal())
571  return success();
572 
573  unsigned numArguments = funcOp.getNumArguments();
574  if (funcOp.front().getNumArguments() != numArguments)
575  return funcOp.emitOpError("entry block must have ")
576  << numArguments << " arguments to match function signature";
577 
578  return success();
579 }

581 template <typename ConcreteType>
582 LogicalResult FunctionLike<ConcreteType>::verifyTrait(Operation *op) {
583  auto funcOp = cast<ConcreteType>(op);
584  if (!funcOp.isTypeAttrValid())
585  return funcOp.emitOpError("requires a type attribute '")
586  << getTypeAttrName() << '\'';
587 
588  if (failed(funcOp.verifyType()))
589  return failure();
590 
591  if (ArrayAttr allArgAttrs = funcOp.getAllArgAttrs()) {
592  unsigned numArgs = funcOp.getNumArguments();
593  if (allArgAttrs.size() != numArgs) {
594  return funcOp.emitOpError()
595  << "expects argument attribute array `"
596  << function_like_impl::getArgDictAttrName()
597  << "` to have the same number of elements as the number of "
598  "function arguments, got "
599  << allArgAttrs.size() << ", but expected " << numArgs;
600  }
601  for (unsigned i = 0; i != numArgs; ++i) {
602  DictionaryAttr argAttrs =
603  allArgAttrs[i].dyn_cast_or_null<DictionaryAttr>();
604  if (!argAttrs) {
605  return funcOp.emitOpError() << "expects argument attribute dictionary "
606  "to be a DictionaryAttr, but got `"
607  << allArgAttrs[i] << "`";
608  }

610  // Verify that all of the argument attributes are dialect attributes, i.e.
611  // that they contain a dialect prefix in their name. Call the dialect, if
612  // registered, to verify the attributes themselves.
613  for (auto attr : argAttrs) {
614  if (!attr.getName().strref().contains('.'))
615  return funcOp.emitOpError(
616  "arguments may only have dialect attributes");
617  if (Dialect *dialect = attr.getNameDialect()) {
618  if (failed(dialect->verifyRegionArgAttribute(op, /*regionIndex=*/0,
619  /*argIndex=*/i, attr)))
620  return failure();
621  }
622  }
623  }
624  }
625  if (ArrayAttr allResultAttrs = funcOp.getAllResultAttrs()) {
626  unsigned numResults = funcOp.getNumResults();
627  if (allResultAttrs.size() != numResults) {
628  return funcOp.emitOpError()
629  << "expects result attribute array `"
630  << function_like_impl::getResultDictAttrName()
631  << "` to have the same number of elements as the number of "
632  "function results, got "
633  << allResultAttrs.size() << ", but expected " << numResults;
634  }
635  for (unsigned i = 0; i != numResults; ++i) {
636  DictionaryAttr resultAttrs =
637  allResultAttrs[i].dyn_cast_or_null<DictionaryAttr>();
638  if (!resultAttrs) {
639  return funcOp.emitOpError() << "expects result attribute dictionary "
640  "to be a DictionaryAttr, but got `"
641  << allResultAttrs[i] << "`";
642  }
643 
644  // Verify that all of the result attributes are dialect attributes, i.e.
645  // that they contain a dialect prefix in their name. Call the dialect, if
646  // registered, to verify the attributes themselves.
647  for (auto attr : resultAttrs) {
648  if (!attr.getName().strref().contains('.'))
649  return funcOp.emitOpError("results may only have dialect attributes");
650  if (Dialect *dialect = attr.getNameDialect()) {
651  if (failed(dialect->verifyRegionResultAttribute(op, /*regionIndex=*/0,
652  /*resultIndex=*/i,
653  attr)))
654  return failure();
655  }
656  }
657  }
658  }
659 
660  // Check that the op has exactly one region for the body.
661  if (op->getNumRegions() != 1)
662  return funcOp.emitOpError("expects one region");
663 
664  return funcOp.verifyBody();
665 }
