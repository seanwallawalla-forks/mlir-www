25 memref::GlobalOp GlobalCreator::getGlobalFor(arith::ConstantOp constantOp) {
26  auto type = constantOp.getType().cast<RankedTensorType>();
27 
28  bufferization::BufferizeTypeConverter typeConverter;
29 
30  // If we already have a global for this constant value, no need to do
31  // anything else.
32  auto it = globals.find(constantOp.getValue());
33  if (it != globals.end())
34  return cast<memref::GlobalOp>(it->second);
35 
36  // Create a builder without an insertion point. We will insert using the
37  // symbol table to guarantee unique names.
38  OpBuilder globalBuilder(moduleOp.getContext());
39  SymbolTable symbolTable(moduleOp);
40 
41  // Create a pretty name.
42  SmallString<64> buf;
43  llvm::raw_svector_ostream os(buf);
44  interleave(type.getShape(), os, "x");
45  os << "x" << type.getElementType();
46 
47  // Add an optional alignment to the global memref.
48  IntegerAttr memrefAlignment =
49  alignment > 0 ? IntegerAttr::get(globalBuilder.getI64Type(), alignment)
50  : IntegerAttr();
51 
52  auto global = globalBuilder.create<memref::GlobalOp>(
53  constantOp.getLoc(), (Twine("__constant_") + os.str()).str(),
54  /*sym_visibility=*/globalBuilder.getStringAttr("private"),
55  /*type=*/typeConverter.convertType(type).cast<MemRefType>(),
56  /*initial_value=*/constantOp.getValue().cast<ElementsAttr>(),
57  /*constant=*/true,
58  /*alignment=*/memrefAlignment);
59  symbolTable.insert(global);
60  // The symbol table inserts at the end of the module, but globals are a bit
61  // nicer if they are at the beginning.
62  global->moveBefore(&moduleOp.front());
63  globals[constantOp.getValue()] = global;
64  return global;
65 }
