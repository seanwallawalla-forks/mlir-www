60  LogicalResult bufferize(Operation *op, OpBuilder &b,
61  BufferizationState &state) const {
62  auto castOp = cast<tensor::CastOp>(op);
63 
64  Value resultBuffer = state.getResultBuffer(castOp->getResult(0));
65  if (!resultBuffer)
66  return failure();
67  Type sourceType = resultBuffer.getType();
68  auto rankedMemRefType = sourceType.dyn_cast<MemRefType>();
69  auto unrankedMemRefType = sourceType.dyn_cast<UnrankedMemRefType>();
70  assert(rankedMemRefType || unrankedMemRefType);
71  Attribute memorySpace = rankedMemRefType
72  ? rankedMemRefType.getMemorySpace()
73  : unrankedMemRefType.getMemorySpace();
74  TensorType tensorType = castOp.getResult().getType().cast<TensorType>();
75  MemRefLayoutAttrInterface layout =
76  rankedMemRefType && tensorType.isa<RankedTensorType>()
77  ? rankedMemRefType.getLayout()
78  : MemRefLayoutAttrInterface();
79  Type memRefType = getContiguousOrUnrankedMemRefType(
80  castOp.getResult().getType(), layout, memorySpace);
81  Value res =
82  b.create<memref::CastOp>(castOp.getLoc(), memRefType, resultBuffer);
83  state.mapBuffer(castOp.getResult(), res);
84  return success();
85  }

137  LogicalResult bufferize(Operation *op, OpBuilder &b,
138  BufferizationState &state) const {
139  auto extractSliceOp = cast<tensor::ExtractSliceOp>(op);
140  Location loc = extractSliceOp.getLoc();
141  Value srcMemref = state.lookupBuffer(extractSliceOp.source());
142  auto srcMemrefType = srcMemref.getType().cast<MemRefType>();
143  auto dstTensorType =
144  extractSliceOp.result().getType().cast<RankedTensorType>();
145 
146  // If not inplaceable, alloc.
147  bool inplace = state.aliasInfo.isInPlace(extractSliceOp->getResult(0));
148  Value alloc;
149  if (!inplace)
150  alloc = state.createAllocDeallocFn(b, loc, extractSliceOp.result());
151 
152  // Bufferize to subview.
153  auto subviewMemRefType =
154  memref::SubViewOp::inferRankReducedResultType(
155  dstTensorType.getRank(), srcMemrefType,
156  extractSliceOp.getMixedOffsets(), extractSliceOp.getMixedSizes(),
157  extractSliceOp.getMixedStrides())
158  .cast<MemRefType>();
159  Value subView = b.create<memref::SubViewOp>(
160  loc, subviewMemRefType, srcMemref, extractSliceOp.getMixedOffsets(),
161  extractSliceOp.getMixedSizes(), extractSliceOp.getMixedStrides());
162  // Insert new alias.
163  state.aliasInfo.insertNewBufferAlias(subView, srcMemref);
164 
165  /// If not inplaceable, copy.
166  if (!inplace) {
167  // Do not copy if the copied data is never read.
168  if (isValueRead(extractSliceOp.result()))
169  state.options.allocationFns->memCpyFn(b, extractSliceOp.getLoc(),
170  subView, alloc);
171  subView = alloc;
172  }
173 
174  state.mapBuffer(extractSliceOp.result(), subView);
175  return success();
176  }

244 /// Return true if the (ExtractSliceOp, InsertSliceOp) pair match (i.e.
245 /// equivalent operand / result and same offset/sizes/strides specification).
246 ///
247 /// This is one particular type of relationship between ops on tensors that
248 /// reduce to an equivalence on buffers. This should be generalized and
249 /// exposed as interfaces on the proper types.
250 static bool
251 areEquivalentExtractSliceOps(const BufferizationAliasInfo &aliasInfo,
252  ExtractSliceOp st, InsertSliceOp sti) {
253  if (!st || !sti)
254  return false;
255  if (!aliasInfo.areEquivalentBufferizedValues(st.source(), sti.dest()))
256  return false;
257  if (!sameOffsetsSizesAndStrides(st, sti, isEqualConstantIntOrValue))
258  return false;
259  return true;
260 }

262 /// Return true if the source of a `insertSliceOp` bufferizes to an
263 /// equivalent ExtractSliceOp that bufferizes inplace.
264 static bool isSourceEquivalentToAMatchingInplaceExtractSliceOp(
265  const BufferizationAliasInfo &aliasInfo, InsertSliceOp insertSliceOp) {
266  bool foundOp = false;
267  aliasInfo.applyOnEquivalenceClass(insertSliceOp.source(), [&](Value value) {
268  auto extractSliceOp = value.getDefiningOp<ExtractSliceOp>();
269  if (extractSliceOp &&
270  areEquivalentExtractSliceOps(aliasInfo, extractSliceOp,
271  insertSliceOp) &&
272  aliasInfo.isInPlace(extractSliceOp->getResult(0))) {
273  foundOp = true;
274  }
275  });
276  return foundOp;
277 }

279 /// Return true if `value` is originating from an ExtractSliceOp that matches
280 /// the given InsertSliceOp.
281 static bool hasMatchingExtractSliceOp(const BufferizationAliasInfo &aliasInfo,
282  Value value, InsertSliceOp insertOp) {
283  auto condition = [&](Value val) {
284  if (auto extractOp = val.getDefiningOp<ExtractSliceOp>())
285  if (areEquivalentExtractSliceOps(aliasInfo, extractOp, insertOp))
286  return true;
287  return false;
288  };
289 
290  return llvm::all_of(findValueInReverseUseDefChain(value, condition),
291  condition);

316  bool isNotConflicting(Operation *op, OpOperand *uRead,
317  OpOperand *uConflictingWrite,
318  const BufferizationAliasInfo &aliasInfo) const {
319  Operation *readingOp = uRead->getOwner();
320  Operation *conflictingWritingOp = uConflictingWrite->getOwner();
321 
322  // Special rules for matching ExtractSliceOp/InsertSliceOp pairs. If
323  // uRead is an InsertSliceOp...
324  if (auto insertSliceOp = dyn_cast<InsertSliceOp>(readingOp)) {
325  // As an example, consider the following IR.
326  //
327  // %0 = tensor.extract_slice %t[%a, %b][%c, %d][1, 1] {inplace = [true] }
328  // %1 = linalg.fill %cst, %0 {inplace= [true] }
329  // %2 = tensor.insert_slice %1 into %t[%a, %b][%c, %d][1, 1]
330  // {inplace= [true] }
331 
332  // TODO: Use insertSliceOp.getDestOpOperand etc. when available.
333  if (uRead == &insertSliceOp->getOpOperand(1) /*dest*/ &&
334  hasMatchingExtractSliceOp(aliasInfo, uConflictingWrite->get(),
335  insertSliceOp))
336  // Case 1: The main insight is that InsertSliceOp reads only part of
337  // the destination tensor. The overwritten area is not read. If
338  // uConflictingWrite writes into exactly the memory location that is
339  // being read by uRead, this is not a conflict.
340  //
341  // In the above example:
342  // uRead = OpOperand 1 (%t) of tensor.insert_slice
343  // uConflictingWrite = OpOperand 1 (%0) of linalg.fill
344  //
345  // The read of %t does not conflict with the write of the FillOp
346  // (same aliases!) because the area that the FillOp operates on is
347  // exactly the one that is *not* read via %t.
348  return true;
349 
350  if (uRead == &insertSliceOp->getOpOperand(0) /*source*/ &&
351  uConflictingWrite == &insertSliceOp->getOpOperand(1) /*dest*/ &&
352  hasMatchingExtractSliceOp(aliasInfo, uRead->get(), insertSliceOp))
353  // Case 2: The read of the source tensor and the write to the dest
354  // tensor via an InsertSliceOp is not a conflict if the read is
355  // reading exactly that part of an equivalent tensor that the
356  // InsertSliceOp is writing.
357  //
358  // In the above example:
359  // uRead = OpOperand 0 (%1) of tensor.insert_slice
360  // uConflictingWrite = OpOperand 1 (%t) of tensor.insert_slice
361  return true;
362  }
363 
364  // If uConflictingWrite is an InsertSliceOp...
365  if (auto insertSliceOp = dyn_cast<InsertSliceOp>(conflictingWritingOp))
366  // As an example, consider the following IR.
367  //
368  // %0 = tensor.extract_slice %t[%a, %b][%c, %d][1, 1] {inplace = [true] }
369  // %1 = linalg.fill %cst, %0 {inplace= [true] }
370  // %2 = tensor.insert_slice %1 into %t[%a, %b][%c, %d][1, 1]
371  // {inplace= [true] }
372  // %3 = vector.transfer_read %1, %cst
373  //
374  // In the above example:
375  // uRead = OpOperand 0 (%1) of vector.transfer_read
376  // uConflictingWrite = OpOperand 1 (%t) of tensor.insert_slice
377  // lastWrite = %1
378  //
379  // This is not a conflict because the InsertSliceOp overwrites the
380  // memory segment of %1 with the exact same data. (Effectively, there
381  // is no memory write here.)
382  if (uConflictingWrite == &insertSliceOp->getOpOperand(1) /*dest*/ &&
383  aliasInfo.areEquivalentBufferizedValues(uRead->get(),
384  insertSliceOp.source()) &&
385  hasMatchingExtractSliceOp(aliasInfo, insertSliceOp.source(),
386  insertSliceOp))
387  return true;
388 
389  return false;
390  }
391 
392  LogicalResult bufferize(Operation *op, OpBuilder &b,
393  BufferizationState &state) const {
394  // insert_slice ops arise from tiling and bufferizing them out-of-place is
395  // generally a deal breaker. When used with loops, this ends up cloning the
396  // whole tensor on every single iteration and is a symptom of a
397  // catastrophically bad scheduling decision.
398  // TODO: be very loud about it or even consider failing the pass.
399  auto insertSliceOp = cast<tensor::InsertSliceOp>(op);
400  Location loc = insertSliceOp.getLoc();
401  TensorBufferizationState &tensorState = getTensorBufferizationState(state);
402 
403  // When bufferizing out-of-place, `getResultBuffer` allocates.
404  Value dstMemref = state.getResultBuffer(insertSliceOp->getResult(0));
405  if (!dstMemref)
406  return failure();
407 
408  bool needCopy =
409  !tensorState.insertSliceOpsWithoutCopy.contains(insertSliceOp);
410  if (needCopy) {
411  // Take a subview of the dst.
412  auto dstMemrefType = dstMemref.getType().cast<MemRefType>();
413  auto subviewMemRefType =
414  memref::SubViewOp::inferRankReducedResultType(
415  insertSliceOp.getSourceType().getRank(), dstMemrefType,
416  insertSliceOp.getMixedOffsets(), insertSliceOp.getMixedSizes(),
417  insertSliceOp.getMixedStrides())
418  .cast<MemRefType>();
419  Value subView = b.create<memref::SubViewOp>(
420  loc, subviewMemRefType, dstMemref, insertSliceOp.getMixedOffsets(),
421  insertSliceOp.getMixedSizes(), insertSliceOp.getMixedStrides());
422  // Copy tensor.
423  Value srcMemref = state.lookupBuffer(insertSliceOp.source());
424  state.options.allocationFns->memCpyFn(b, insertSliceOp.getLoc(),
425  srcMemref, subView);
426  }
427 
428  state.mapBuffer(insertSliceOp.result(), dstMemref);
429  return success();
430  }

438 LogicalResult mlir::linalg::comprehensive_bufferize::tensor_ext::
439  InplaceInsertSliceOpAnalysis::run(FuncOp funcOp, BufferizationState &state,
440  SmallVector<Operation *> &newOps) {
441  auto &tensorState = getTensorBufferizationState(state);
442  funcOp.walk([&](InsertSliceOp insertSliceOp) {
443  // A copy of the source buffer is needed if either:
444  // - The producer of `source` is not inplace. This is the case where a
445  // slice is computed out of place into the inplace full tensor.
446  // - The result is not inplace. This is the case where the whole tensor is
447  // cloned and the clone needs to be updated.
448  if (isSourceEquivalentToAMatchingInplaceExtractSliceOp(state.aliasInfo,
449  insertSliceOp) &&
450  state.aliasInfo.isInPlace(insertSliceOp->getResult(0)))
451  tensorState.insertSliceOpsWithoutCopy.insert(insertSliceOp);
452  });
453  return success();
454 }
