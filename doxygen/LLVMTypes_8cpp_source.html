156 /// The positions of different values in the data layout entry.
157 enum class DLEntryPos { Size = 0, Abi = 1, Preferred = 2, Address = 3 };
158 } // namespace
159 
160 constexpr const static unsigned kDefaultPointerSizeBits = 64;
161 constexpr const static unsigned kDefaultPointerAlignment = 8;
162 constexpr const static unsigned kBitsInByte = 8;
163 
164 /// Returns the value that corresponds to named position `pos` from the
165 /// attribute `attr` assuming it's a dense integer elements attribute.
166 static unsigned extractPointerSpecValue(Attribute attr, DLEntryPos pos) {
167  return attr.cast<DenseIntElementsAttr>()
168  .getValues<unsigned>()[static_cast<unsigned>(pos)];
169 }
170 
171 /// Returns the part of the data layout entry that corresponds to `pos` for the
172 /// given `type` by interpreting the list of entries `params`. For the pointer
173 /// type in the default address space, returns the default value if the entries
174 /// do not provide a custom one, for other address spaces returns None.
175 static Optional<unsigned>
176 getPointerDataLayoutEntry(DataLayoutEntryListRef params, LLVMPointerType type,
177  DLEntryPos pos) {
178  // First, look for the entry for the pointer in the current address space.
179  Attribute currentEntry;
180  for (DataLayoutEntryInterface entry : params) {
181  if (!entry.isTypeEntry())
182  continue;
183  if (entry.getKey().get<Type>().cast<LLVMPointerType>().getAddressSpace() ==
184  type.getAddressSpace()) {
185  currentEntry = entry.getValue();
186  break;
187  }
188  }
189  if (currentEntry) {
190  return extractPointerSpecValue(currentEntry, pos) /
191  (pos == DLEntryPos::Size ? 1 : kBitsInByte);
192  }
193 
194  // If not found, and this is the pointer to the default memory space, assume
195  // 64-bit pointers.
196  if (type.getAddressSpace() == 0) {
197  return pos == DLEntryPos::Size ? kDefaultPointerSizeBits
198  : kDefaultPointerAlignment;
199  }
200 
201  return llvm::None;
202 }

204 unsigned
205 LLVMPointerType::getTypeSizeInBits(const DataLayout &dataLayout,
206  DataLayoutEntryListRef params) const {
207  if (Optional<unsigned> size =
208  getPointerDataLayoutEntry(params, *this, DLEntryPos::Size))
209  return *size;
210 
211  // For other memory spaces, use the size of the pointer to the default memory
212  // space.
213  return dataLayout.getTypeSizeInBits(get(getElementType()));

235 bool LLVMPointerType::areCompatible(DataLayoutEntryListRef oldLayout,
236  DataLayoutEntryListRef newLayout) const {
237  for (DataLayoutEntryInterface newEntry : newLayout) {
238  if (!newEntry.isTypeEntry())
239  continue;
240  unsigned size = kDefaultPointerSizeBits;
241  unsigned abi = kDefaultPointerAlignment;
242  auto newType = newEntry.getKey().get<Type>().cast<LLVMPointerType>();
243  auto it = llvm::find_if(oldLayout, [&](DataLayoutEntryInterface entry) {
244  if (auto type = entry.getKey().dyn_cast<Type>()) {
245  return type.cast<LLVMPointerType>().getAddressSpace() ==
246  newType.getAddressSpace();
247  }
248  return false;
249  });
250  if (it == oldLayout.end()) {
251  llvm::find_if(oldLayout, [&](DataLayoutEntryInterface entry) {
252  if (auto type = entry.getKey().dyn_cast<Type>()) {
253  return type.cast<LLVMPointerType>().getAddressSpace() == 0;
254  }
255  return false;
256  });
257  }
258  if (it != oldLayout.end()) {
259  size = extractPointerSpecValue(*it, DLEntryPos::Size);
260  abi = extractPointerSpecValue(*it, DLEntryPos::Abi);
261  }
262 
263  Attribute newSpec = newEntry.getValue().cast<DenseIntElementsAttr>();
264  unsigned newSize = extractPointerSpecValue(newSpec, DLEntryPos::Size);
265  unsigned newAbi = extractPointerSpecValue(newSpec, DLEntryPos::Abi);
266  if (size != newSize || abi < newAbi || abi % newAbi != 0)
267  return false;
268  }
269  return true;
270 }

272 LogicalResult LLVMPointerType::verifyEntries(DataLayoutEntryListRef entries,
273  Location loc) const {
274  for (DataLayoutEntryInterface entry : entries) {
275  if (!entry.isTypeEntry())
276  continue;
277  auto key = entry.getKey().get<Type>().cast<LLVMPointerType>();
278  auto values = entry.getValue().dyn_cast<DenseIntElementsAttr>();
279  if (!values || (values.size() != 3 && values.size() != 4)) {
280  return emitError(loc)
281  << "expected layout attribute for " << entry.getKey().get<Type>()
282  << " to be a dense integer elements attribute with 3 or 4 "
283  "elements";
284  }
285  if (!key.getElementType().isInteger(8)) {
286  return emitError(loc) << "unexpected layout attribute for pointer to "
287  << key.getElementType();
288  }
289  if (extractPointerSpecValue(values, DLEntryPos::Abi) >
290  extractPointerSpecValue(values, DLEntryPos::Preferred)) {
291  return emitError(loc) << "preferred alignment is expected to be at least "
292  "as large as ABI alignment";
293  }
294  }
295  return success();
296 }

487 bool mlir::LLVM::isCompatibleOuterType(Type type) {
488  // clang-format off
489  if (type.isa<
490  BFloat16Type,
491  Float16Type,
492  Float32Type,
493  Float64Type,
494  Float80Type,
495  Float128Type,
496  LLVMArrayType,
497  LLVMFunctionType,
498  LLVMLabelType,
499  LLVMMetadataType,
500  LLVMPPCFP128Type,
501  LLVMPointerType,
502  LLVMStructType,
503  LLVMTokenType,
504  LLVMFixedVectorType,
505  LLVMScalableVectorType,
506  LLVMVoidType,
507  LLVMX86MMXType
508  >()) {
509  // clang-format on
510  return true;
511  }
512 
513  // Only signless integers are compatible.
514  if (auto intType = type.dyn_cast<IntegerType>())
515  return intType.isSignless();
516 
517  // 1D vector types are compatible.
518  if (auto vecType = type.dyn_cast<VectorType>())
519  return vecType.getRank() == 1;
520 
521  return false;
522 }

524 static bool isCompatibleImpl(Type type, SetVector<Type> &callstack) {
525  if (callstack.contains(type))
526  return true;
527 
528  callstack.insert(type);
529  auto stackPopper = llvm::make_scope_exit([&] { callstack.pop_back(); });
530 
531  auto isCompatible = [&](Type type) {
532  return isCompatibleImpl(type, callstack);
533  };
534 
535  return llvm::TypeSwitch<Type, bool>(type)
536  .Case<LLVMStructType>([&](auto structType) {
537  return llvm::all_of(structType.getBody(), isCompatible);
538  })
539  .Case<LLVMFunctionType>([&](auto funcType) {
540  return isCompatible(funcType.getReturnType()) &&
541  llvm::all_of(funcType.getParams(), isCompatible);
542  })
543  .Case<IntegerType>([](auto intType) { return intType.isSignless(); })
544  .Case<VectorType>([&](auto vecType) {
545  return vecType.getRank() == 1 && isCompatible(vecType.getElementType());
546  })
547  // clang-format off
548  .Case<
549  LLVMPointerType,
550  LLVMFixedVectorType,
551  LLVMScalableVectorType,
552  LLVMArrayType
553  >([&](auto containerType) {
554  return isCompatible(containerType.getElementType());
555  })
556  .Case<
557  BFloat16Type,
558  Float16Type,
559  Float32Type,
560  Float64Type,
561  Float80Type,
562  Float128Type,
563  LLVMLabelType,
564  LLVMMetadataType,
565  LLVMPPCFP128Type,
566  LLVMTokenType,
567  LLVMVoidType,
568  LLVMX86MMXType
569  >([](Type) { return true; })
570  // clang-format on
571  .Default([](Type) { return false; });
572 }

584 bool mlir::LLVM::isCompatibleVectorType(Type type) {
585  if (type.isa<LLVMFixedVectorType, LLVMScalableVectorType>())
586  return true;
587

589  if (vecType.getRank() != 1)

593  return intType.isSignless();

595  Float80Type, Float128Type>();
596  }
597  return false;
