30 /// Computes blocks number of executions information for the given region.
31 static void computeRegionBlockNumberOfExecutions(
32  Region &region, DenseMap<Block *, BlockNumberOfExecutionsInfo> &blockInfo) {
33  Operation *parentOp = region.getParentOp();
34  int regionId = region.getRegionNumber();
35 
36  auto regionKindInterface = dyn_cast<RegionKindInterface>(parentOp);
37  bool isGraphRegion =
38  regionKindInterface &&
39  regionKindInterface.getRegionKind(regionId) == RegionKind::Graph;
40 
41  // CFG analysis does not make sense for Graph regions, set the number of
42  // executions for all blocks as unknown.
43  if (isGraphRegion) {
44  for (Block &block : region)
45  blockInfo.insert({&block, {&block, None, None}});
46  return;
47  }
48 
49  // Number of region invocations for all attached regions.
50  SmallVector<int64_t, 4> numRegionsInvocations;
51 
52  // Query RegionBranchOpInterface interface if it is available.
53  if (auto regionInterface = dyn_cast<RegionBranchOpInterface>(parentOp)) {
54  SmallVector<Attribute, 4> operands(parentOp->getNumOperands());
55  for (auto operandIt : llvm::enumerate(parentOp->getOperands()))
56  matchPattern(operandIt.value(), m_Constant(&operands[operandIt.index()]));
57 
58  regionInterface.getNumRegionInvocations(operands, numRegionsInvocations);
59  }
60 
61  // Number of region invocations *each time* parent operation is invoked.
62  Optional<int64_t> numRegionInvocations;
63 
64  if (!numRegionsInvocations.empty() &&
65  numRegionsInvocations[regionId] != kUnknownNumRegionInvocations) {
66  numRegionInvocations = numRegionsInvocations[regionId];
67  }
68 
69  // DFS traversal looking for loops in the CFG.
70  llvm::SmallSet<Block *, 4> loopStart;
71 
72  llvm::unique_function<void(Block *, llvm::SmallSet<Block *, 4> &)> dfs =
73  [&](Block *block, llvm::SmallSet<Block *, 4> &visited) {
74  // Found a loop in the CFG that starts at the `block`.
75  if (visited.contains(block)) {
76  loopStart.insert(block);
77  return;
78  }
79 
80  // Continue DFS traversal.
81  visited.insert(block);
82  for (Block *successor : block->getSuccessors())
83  dfs(successor, visited);
84  visited.erase(block);
85  };
86 
87  llvm::SmallSet<Block *, 4> visited;
88  dfs(&region.front(), visited);
89 
90  // Start from the entry block and follow only blocks with single succesor.
91  Block *block = &region.front();
92  while (block && !loopStart.contains(block)) {
93  // Block will be executed exactly once.
94  blockInfo.insert(
95  {block, BlockNumberOfExecutionsInfo(block, numRegionInvocations,
96  /*numberOfBlockExecutions=*/1)});
97 
98  // We reached the exit block or block with multiple successors.
99  if (block->getNumSuccessors() != 1)
100  break;
101 
102  // Continue traversal.
103  block = block->getSuccessor(0);
104  }
105 
106  // For all blocks that we did not visit set the executions number to unknown.
107  for (Block &block : region)
108  if (blockInfo.count(&block) == 0)
109  blockInfo.insert({&block, BlockNumberOfExecutionsInfo(
110  &block, numRegionInvocations,
111  /*numberOfBlockExecutions=*/None)});
112 }

114 /// Creates a new NumberOfExecutions analysis that computes how many times a
115 /// block within a region is executed for all associated regions.
116 NumberOfExecutions::NumberOfExecutions(Operation *op) : operation(op) {
117  operation->walk<WalkOrder::PreOrder>([&](Region *region) {
118  computeRegionBlockNumberOfExecutions(*region, blockNumbersOfExecution);
119  });

122 Optional<int64_t>
123 NumberOfExecutions::getNumberOfExecutions(Operation *op,
124  Region *perEntryOfThisRegion) const {
125  // Assuming that all operations complete in a finite amount of time (do not
126  // abort and do not go into the infinite loop), the number of operation
127  // executions is equal to the number of block executions that contains the
128  // operation.
129  return getNumberOfExecutions(op->getBlock(), perEntryOfThisRegion);
130 }

132 Optional<int64_t>
133 NumberOfExecutions::getNumberOfExecutions(Block *block,
134  Region *perEntryOfThisRegion) const {
135  // Return None if the given `block` does not lie inside the
136  // `perEntryOfThisRegion` region.
137  if (!perEntryOfThisRegion->findAncestorBlockInRegion(*block))
138  return None;
139 
140  // Find the block information for the given `block.
141  auto blockIt = blockNumbersOfExecution.find(block);
142  if (blockIt == blockNumbersOfExecution.end())
143  return None;
144  const auto &blockInfo = blockIt->getSecond();
145 
146  // Override the number of region invocations with `1` if the
147  // `perEntryOfThisRegion` region owns the block.
148  auto getNumberOfExecutions = [&](const BlockNumberOfExecutionsInfo &info) {
149  if (info.getBlock()->getParent() == perEntryOfThisRegion)
150  return info.getNumberOfExecutions(/*numberOfRegionInvocations=*/1);
151  return info.getNumberOfExecutions();
152  };
153 
154  // Immediately return None if we do not know the block number of executions.
155  auto blockExecutions = getNumberOfExecutions(blockInfo);
156  if (!blockExecutions.hasValue())
157  return None;
158 
159  // Follow parent operations until we reach the operations that owns the
160  // `perEntryOfThisRegion`.
161  int64_t numberOfExecutions = *blockExecutions;
162  Operation *parentOp = block->getParentOp();
163 
164  while (parentOp != perEntryOfThisRegion->getParentOp()) {
165  // Find how many times will be executed the block that owns the parent
166  // operation.
167  Block *parentBlock = parentOp->getBlock();
168 
169  auto parentBlockIt = blockNumbersOfExecution.find(parentBlock);
170  if (parentBlockIt == blockNumbersOfExecution.end())
171  return None;
172  const auto &parentBlockInfo = parentBlockIt->getSecond();
173  auto parentBlockExecutions = getNumberOfExecutions(parentBlockInfo);
174 
175  // We stumbled upon an operation with unknown number of executions.
176  if (!parentBlockExecutions.hasValue())
177  return None;
178 
179  // Number of block executions is a product of all parent blocks executions.
180  numberOfExecutions *= *parentBlockExecutions;
181  parentOp = parentOp->getParentOp();
182 
183  assert(parentOp != nullptr);
184  }
185 
186  return numberOfExecutions;
187 }

189 void NumberOfExecutions::printBlockExecutions(
190  raw_ostream &os, Region *perEntryOfThisRegion) const {
191  unsigned blockId = 0;
192 
193  operation->walk<WalkOrder::PreOrder>([&](Block *block) {
194  llvm::errs() << "Block: " << blockId++ << "\n";
195  llvm::errs() << "Number of executions: ";
196  if (auto n = getNumberOfExecutions(block, perEntryOfThisRegion))
197  llvm::errs() << *n << "\n";
198  else
199  llvm::errs() << "<unknown>\n";
200  });

203 void NumberOfExecutions::printOperationExecutions(
204  raw_ostream &os, Region *perEntryOfThisRegion) const {
205  operation->walk<WalkOrder::PreOrder>([&](Block *block) {
206  block->walk([&](Operation *operation) {
207  // Skip the operation that was used to build the analysis.
208  if (operation == this->operation)
209  return;
210 
211  llvm::errs() << "Operation: " << operation->getName() << "\n";
212  llvm::errs() << "Number of executions: ";
213  if (auto n = getNumberOfExecutions(operation, perEntryOfThisRegion))
214  llvm::errs() << *n << "\n";
215  else
216  llvm::errs() << "<unknown>\n";
217  });
218  });
219 }

231 Optional<int64_t> BlockNumberOfExecutionsInfo::getNumberOfExecutions() const {
232  if (numberOfRegionInvocations && numberOfBlockExecutions)
233  return *numberOfRegionInvocations * *numberOfBlockExecutions;
234  return None;
