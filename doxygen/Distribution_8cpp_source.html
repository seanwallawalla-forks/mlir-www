29 struct DistributeTiledLoopPattern
30  : public OpRewritePattern<linalg::TiledLoopOp> {
31  DistributeTiledLoopPattern(MLIRContext *context,
32  LinalgLoopDistributionOptions options,
33  LinalgTransformationFilter marker)
34  : OpRewritePattern<linalg::TiledLoopOp>(context), options(options),
35  marker(marker) {}
36  LogicalResult matchAndRewrite(linalg::TiledLoopOp op,
37  PatternRewriter &rewriter) const override {
38  if (failed(marker.checkAndNotify(rewriter, op)))
39  return failure();
40  if (!op.distribution_types().hasValue())
41  return failure();
42 
43  Location loc = op.getLoc();
44  SmallVector<Value, 2> newLowerBounds = op.lowerBound();
45  SmallVector<Value, 2> newUpperBounds = op.upperBound();
46  SmallVector<Value, 2> newSteps = op.step();
47 
48  // Update bounds and steps.
49  auto distributionTypes = op.distribution_types().getValue();
50  for (int i = 0, e = op.getNumLoops(); i < e; ++i) {
51  StringRef type = distributionTypes[i].cast<StringAttr>().getValue();
52  auto procInfoCallback = options.procInfoMap.find(type);
53  if (procInfoCallback == options.procInfoMap.end())
54  continue;
55 
56  if (!isParallelIterator(op.iterator_types()[i])) {
57  op.emitOpError("only support for parallel loops is implemented");
58  return failure();
59  }
60  ProcInfo info = procInfoCallback->second(rewriter, loc);
61  updateBoundsForCyclicDistribution(rewriter, loc, info.procId, info.nprocs,
62  newLowerBounds[i], newUpperBounds[i],
63  newSteps[i]);
64  }
65  rewriter.updateRootInPlace(op, [&] {
66  op.setLowerBounds(newLowerBounds);
67  op.setUpperBounds(newUpperBounds);
68  op.setSteps(newSteps);
69  });
70  marker.replaceLinalgTransformationFilter(rewriter, op);
71  return success();
72  }
73 
74 private:
75  LinalgLoopDistributionOptions options;
76  LinalgTransformationFilter marker;
77 };
