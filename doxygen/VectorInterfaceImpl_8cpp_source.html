78  LogicalResult bufferize(Operation *op, OpBuilder &b,
79  BufferizationState &state) const {
80  auto writeOp = cast<vector::TransferWriteOp>(op);
81  assert(writeOp.getShapedType().isa<TensorType>() &&
82  "only tensor types expected");
83 
84  // Create a new transfer_write on buffer that doesn't have a return value.
85  // Leave the previous transfer_write to dead code as it still has uses at
86  // this point.
87  Value resultBuffer = state.getResultBuffer(op->getResult(0));
88  if (!resultBuffer)
89  return failure();
90  b.create<vector::TransferWriteOp>(
91  writeOp.getLoc(), writeOp.vector(), resultBuffer, writeOp.indices(),
92  writeOp.permutation_mapAttr(), writeOp.in_boundsAttr());
93  state.mapBuffer(op->getResult(0), resultBuffer);
94 
95  return success();
96  }
