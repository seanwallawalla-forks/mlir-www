86 /// Converts tensor.extract into loading using access chains from SPIR-V local
87 /// variables.
88 class TensorExtractPattern final
89  : public OpConversionPattern<tensor::ExtractOp> {
90 public:
91  TensorExtractPattern(TypeConverter &typeConverter, MLIRContext *context,
92  int64_t threshold, PatternBenefit benefit = 1)
93  : OpConversionPattern(typeConverter, context, benefit),
94  byteCountThreshold(threshold) {}
95 
96  LogicalResult
97  matchAndRewrite(tensor::ExtractOp extractOp, OpAdaptor adaptor,
98  ConversionPatternRewriter &rewriter) const override {
99  TensorType tensorType = extractOp.tensor().getType().cast<TensorType>();
100 
101  if (!tensorType.hasStaticShape())
102  return rewriter.notifyMatchFailure(extractOp, "non-static tensor");
103 
104  if (tensorType.getNumElements() * tensorType.getElementTypeBitWidth() >
105  byteCountThreshold * 8)
106  return rewriter.notifyMatchFailure(extractOp,
107  "exceeding byte count threshold");
108 
109  Location loc = extractOp.getLoc();
110 
111  int64_t rank = tensorType.getRank();
112  SmallVector<int64_t, 4> strides(rank, 1);
113  for (int i = rank - 2; i >= 0; --i) {
114  strides[i] = strides[i + 1] * tensorType.getDimSize(i + 1);
115  }
116 
117  Type varType = spirv::PointerType::get(adaptor.tensor().getType(),
118  spirv::StorageClass::Function);
119 
120  spirv::VariableOp varOp;
121  if (adaptor.tensor().getDefiningOp<spirv::ConstantOp>()) {
122  varOp = rewriter.create<spirv::VariableOp>(
123  loc, varType, spirv::StorageClass::Function,
124  /*initializer=*/adaptor.tensor());
125  } else {
126  // Need to store the value to the local variable. It's questionable
127  // whether we want to support such case though.
128  return failure();
129  }
130 
131  auto &typeConverter = *getTypeConverter<SPIRVTypeConverter>();
132  auto indexType = typeConverter.getIndexType();
133 
134  Value index = spirv::linearizeIndex(adaptor.indices(), strides,
135  /*offset=*/0, indexType, loc, rewriter);
136  auto acOp = rewriter.create<spirv::AccessChainOp>(loc, varOp, index);
137 
138  rewriter.replaceOpWithNewOp<spirv::LoadOp>(extractOp, acOp);
139 
140  return success();
141  }
142 
143 private:
144  int64_t byteCountThreshold;
145 };

153 LogicalResult
154 ReturnOpPattern::matchAndRewrite(ReturnOp returnOp, OpAdaptor adaptor,
155  ConversionPatternRewriter &rewriter) const {
156  if (returnOp.getNumOperands() > 1)
157  return failure();
158 
159  if (returnOp.getNumOperands() == 1) {
160  rewriter.replaceOpWithNewOp<spirv::ReturnValueOp>(returnOp,
161  adaptor.getOperands()[0]);
162  } else {
163  rewriter.replaceOpWithNewOp<spirv::ReturnOp>(returnOp);
164  }
165  return success();
166 }

185 LogicalResult
186 SplatPattern::matchAndRewrite(SplatOp op, OpAdaptor adaptor,
187  ConversionPatternRewriter &rewriter) const {
188  auto dstVecType = op.getType().dyn_cast<VectorType>();
189  if (!dstVecType || !spirv::CompositeType::isValid(dstVecType))
190  return failure();
191  SmallVector<Value, 4> source(dstVecType.getNumElements(), adaptor.getInput());
192  rewriter.replaceOpWithNewOp<spirv::CompositeConstructOp>(op, dstVecType,
193  source);
194  return success();
195 }
