102 /// This is a common class used for patterns of the form
103 /// ```
104 /// someop(memrefcast(%src)) -> someop(%src)
105 /// ```
106 /// It folds the source of the memref.cast into the root operation directly.
107 static LogicalResult foldMemRefCast(Operation *op) {
108  bool folded = false;
109  for (OpOperand &operand : op->getOpOperands()) {
110  auto castOp = operand.get().getDefiningOp<memref::CastOp>();
111  if (castOp && memref::CastOp::canFoldIntoConsumerOp(castOp)) {
112  operand.set(castOp.getOperand());
113  folded = true;
114  }
115  }
116  return success(folded);
117 }

119 /// This is a specialization of `foldMemRefCast` used for patterns of the form
120 /// ```
121 /// tiled_loop(memrefcast(%src)) -> tiled_loop(%src)
122 /// ```
123 /// It folds the source of the memref.cast into the root operation directly.
124 static LogicalResult foldMemRefCastInTiledLoopOp(TiledLoopOp op) {
125  bool folded = false;
126  Location loc = op->getLoc();
127 
128  Block *body = op.getBody();
129  OpBuilder b = OpBuilder::atBlockBegin(body);
130 
131  // Update `input` and `output` operands and block arguments if necessary.
132  // Operands list: [lbs, ubs, steps, inputs, outputs].
133  // Block args list: [ivs, inputs, outputs].
134  for (size_t operandIndex = op.getNumControlOperands(),
135  bbArgIndex = op.getNumLoops(), e = op.getNumOperands();
136  operandIndex < e; ++operandIndex, ++bbArgIndex) {
137  OpOperand &operand = op->getOpOperand(operandIndex);
138 
139  auto castOp = operand.get().getDefiningOp<memref::CastOp>();
140  if (castOp && memref::CastOp::canFoldIntoConsumerOp(castOp)) {
141  operand.set(castOp.getOperand());
142  BlockArgument newBbArg =
143  body->insertArgument(bbArgIndex, castOp.getOperand().getType());
144  BlockArgument oldBbArg = body->getArgument(newBbArg.getArgNumber() + 1);
145 
146  // Insert memref.cast back to the original type.
147  oldBbArg.replaceAllUsesWith(
148  b.create<memref::CastOp>(loc, oldBbArg.getType(), newBbArg));
149  body->eraseArgument(oldBbArg.getArgNumber());
150 
151  folded = true;
152  }
153  }
154  return success(folded);
155 }

159 // TODO: Move this to a utility library.
160 // The public methods on this class are referenced directly from generated code
161 // and bind by name to math functions in the DSL as:
162 // `applyfn__{fnName}`
163 // Examples:
164 // `applyfn__add`
165 // `applyfn__mul`
166 // The naming convention is intentional in order to match snake-cased DSL names.
167 // See mlir-linalg-ods-yaml-gen.cpp for the code that mates to this class.
168 //
169 // Implementations of the math functions must be polymorphic over numeric types,
170 // internally performing necessary casts. If the function application makes no
171 // sense, then the only recourse is to assert and return nullptr. This can be
172 // extended later if it becomes possible to fail construction of the region. The
173 // invariant should be enforced at a higher level.
174 //
175 // TODO: These helpers are currently type polymorphic over the class of integer
176 // and floating point types, but they will not internally cast within bit
177 // widths of a class (mixed precision such as i8->i32) or across classes
178 // (i.e. mixed float and integer). Many such combinations are ambiguous or need
179 // to be handled with care and work is being considered to extend the op
180 // language to make such cases explicit. In the mean-time, violating this will
181 // fail verification, which is deemed acceptable.

191  // Generates operations to cast the given operand to a specified type.
192  // If the cast cannot be performed, a warning will be issued and the
193  // operand returned as-is (which will presumably yield a verification
194  // issue downstream).
195  Value cast(Type toType, Value operand, bool isUnsignedCast) {
196  OpBuilder builder = getBuilder();
197  auto loc = operand.getLoc();
198 
199  if (operand.getType() == toType)
200  return operand;
201  if (auto toIntType = toType.dyn_cast<IntegerType>()) {
202  // If operand is floating point, cast directly to the int type.
203  if (operand.getType().isa<FloatType>()) {
204  if (isUnsignedCast)
205  return builder.create<arith::FPToUIOp>(loc, toType, operand);
206  return builder.create<arith::FPToSIOp>(loc, toType, operand);
207  }
208  // Cast index operands directly to the int type.
209  if (operand.getType().isIndex())
210  return builder.create<arith::IndexCastOp>(loc, toType, operand);
211  if (auto fromIntType = operand.getType().dyn_cast<IntegerType>()) {
212  // Either extend or truncate.
213  if (toIntType.getWidth() > fromIntType.getWidth()) {
214  if (isUnsignedCast)
215  return builder.create<arith::ExtUIOp>(loc, toType, operand);
216  return builder.create<arith::ExtSIOp>(loc, toType, operand);
217  }
218  if (toIntType.getWidth() < fromIntType.getWidth())
219  return builder.create<arith::TruncIOp>(loc, toType, operand);
220  }
221  } else if (auto toFloatType = toType.dyn_cast<FloatType>()) {
222  // If operand is integer, cast directly to the float type.
223  // Note that it is unclear how to cast from BF16<->FP16.
224  if (operand.getType().isa<IntegerType>()) {
225  if (isUnsignedCast)
226  return builder.create<arith::UIToFPOp>(loc, toFloatType, operand);
227  return builder.create<arith::SIToFPOp>(loc, toFloatType, operand);
228  }
229  if (auto fromFloatType = operand.getType().dyn_cast<FloatType>()) {
230  if (toFloatType.getWidth() > fromFloatType.getWidth())
231  return builder.create<arith::ExtFOp>(loc, toFloatType, operand);
232  if (toFloatType.getWidth() < fromFloatType.getWidth())
233  return builder.create<arith::TruncFOp>(loc, toFloatType, operand);
234  }
235  }
236 
237  emitWarning(operand.getLoc()) << "could not cast operand of type "
238  << operand.getType() << " to " << toType;
239  return operand;
240  }

402 static LogicalResult verify(CopyOp op) {
403  OpOperand *output = op.getOutputOperand(0);
404  OpOperand *input = op.getInputOperand(0);
405  if (getElementTypeOrSelf(input->get()) != getElementTypeOrSelf(output->get()))
406  return op.emitOpError("expects views of the same type");
407  if (op.getRank(input) != op.getRank(output))
408  return op.emitOpError("expects views of the same rank");
409  auto rank = op.getNumParallelLoops();
410  auto inputPermutationMap = op.inputPermutation();
411  if (inputPermutationMap) {
412  if (inputPermutationMap->getNumInputs() != rank)
413  return op.emitOpError("expects optional input_permutation map of rank ")
414  << rank;
415  if (!inputPermutationMap->isPermutation())
416  return op.emitOpError(
417  "expects optional input_permutation map to be a permutation");
418  }
419  auto outputPermutationMap = op.outputPermutation();
420  if (outputPermutationMap) {
421  if (outputPermutationMap->getNumInputs() != rank)
422  return op.emitOpError("expects optional output_permutation map of rank ")
423  << rank;
424  if (!outputPermutationMap->isPermutation())
425  return op.emitOpError(
426  "expects optional output_permutation map to be a permutation");
427  }
428  if (rank == 0 && inputPermutationMap)
429  return op.emitOpError("expected no input permutation when rank == 0");
430  if (rank == 0 && outputPermutationMap)
431  return op.emitOpError("expected no output permutation when rank == 0");
432  return success();
433 }

695 // Deduplicate redundant args of a linalg generic op.
696 // An arg is redundant if it has the same Value and indexing map as another.
697 struct DeduplicateGenericOpInputs : public OpRewritePattern<GenericOp> {
698  using OpRewritePattern<GenericOp>::OpRewritePattern;
699 
700  LogicalResult matchAndRewrite(GenericOp genericOp,
701  PatternRewriter &rewriter) const override {
702  // Associate each input to an equivalent "canonical" input that has the same
703  // Value and indexing map.
704  //
705  // In the non-duplicate case, input `i` will have canonical input `i`. But
706  // in the case of duplicated inputs, the canonical input could be some other
707  // input `< i`. That is, a later input will have some earlier input as its
708  // canonical input.
709  llvm::SmallDenseMap<std::pair<Value, AffineMap>, unsigned> canonicalInput;
710  // For later remapping tasks like deduplicating payload block arguments,
711  // having a simple "inputIndex -> canonicalInputIndex" integer mapping is
712  // convenient.
713  SmallVector<unsigned> canonicalInputIndices;
714  for (OpOperand *opOperand : genericOp.getInputOperands()) {
715  AffineMap indexingMap = genericOp.getTiedIndexingMap(opOperand);
716  // STL-like maps have a convenient behavior for our use case here. In the
717  // case of duplicate keys, the insertion is rejected, and the returned
718  // iterator gives access to the value already in the map.
719  auto pair = canonicalInput.insert(
720  {{opOperand->get(), indexingMap}, opOperand->getOperandNumber()});
721  canonicalInputIndices.push_back(pair.first->second);
722  }
723 
724  // If there are no duplicate args, then bail out.
725  if (canonicalInput.size() == genericOp.getNumInputs())
726  return failure();
727 
728  // The operands for the newly canonicalized op.
729  SmallVector<Value> newInputOperands;
730  for (OpOperand *opOperand : genericOp.getInputOperands())
731  if (canonicalInputIndices[opOperand->getOperandNumber()] ==
732  opOperand->getOperandNumber())
733  newInputOperands.push_back(opOperand->get());
734 
735  // Repair the indexing maps by filtering out the ones that have been
736  // eliminated.
737  SmallVector<AffineMap> newIndexingMaps;
738  for (OpOperand *opOperand : genericOp.getInputOperands())
739  if (canonicalInputIndices[opOperand->getOperandNumber()] ==
740  opOperand->getOperandNumber())
741  newIndexingMaps.push_back(genericOp.getTiedIndexingMap(opOperand));
742  for (OpOperand *opOperand : genericOp.getOutputOperands())
743  newIndexingMaps.push_back(genericOp.getTiedIndexingMap(opOperand));
744 
745  // Clone the old op with new operands.
746  SmallVector<Value> outputOperands = genericOp.getOutputOperands();
747  auto newOp = rewriter.create<GenericOp>(
748  genericOp.getLoc(), genericOp->getResultTypes(), newInputOperands,
749  outputOperands, rewriter.getAffineMapArrayAttr(newIndexingMaps),
750  genericOp.iterator_types(), genericOp.docAttr(),
751  genericOp.library_callAttr());
752 
753  // Copy over unknown attributes. They might be load bearing for some flow.
754  ArrayRef<StringRef> odsAttrs = genericOp.getAttributeNames();
755  for (NamedAttribute kv : genericOp->getAttrs()) {
756  if (!llvm::is_contained(odsAttrs, kv.getName().getValue())) {
757  newOp->setAttr(kv.getName(), kv.getValue());
758  }
759  }
760 
761  rewriter.inlineRegionBefore(genericOp.region(), newOp.region(),
762  newOp.region().begin());
763 
764  // Repair the payload entry block by RAUW'ing redundant arguments and
765  // erasing them.
766  Block &payload = newOp.region().front();
767  SmallVector<OpOperand *> inputOperands = genericOp.getInputOperands();
768  for (OpOperand *opOperand : llvm::reverse(inputOperands)) {
769  // Iterate in reverse, so that we erase later args first, preventing the
770  // argument list from shifting unexpectedly and invalidating all our
771  // indices.
772  unsigned operandNumber = opOperand->getOperandNumber();
773  if (canonicalInputIndices[operandNumber] == operandNumber)
774  continue;
775  payload.getArgument(operandNumber)
776  .replaceAllUsesWith(
777  payload.getArgument(canonicalInputIndices[operandNumber]));
778  payload.eraseArgument(operandNumber);
779  }
780 
781  rewriter.replaceOp(genericOp, newOp->getResults());
782  return success();
783  }
784 };

786 /// Remove generic operations (on tensors) that are just copying
787 /// the values from inputs to the results. Requirements are
788 /// 1) All iterator types are parallel
789 /// 2) The body contains just a yield operation with the yielded values being
790 /// the arguments corresponding to the operands.
791 struct EraseIdentityGenericOp : public OpRewritePattern<GenericOp> {
792  using OpRewritePattern<GenericOp>::OpRewritePattern;
793 
794  LogicalResult matchAndRewrite(GenericOp genericOp,
795  PatternRewriter &rewriter) const override {
796  if (!genericOp.hasTensorSemantics())
797  return failure();
798  // Check all indexing maps are identity.
799  if (llvm::any_of(genericOp.getIndexingMaps(),
800  [](AffineMap map) { return !map.isIdentity(); }))
801  return failure();
802 
803  // Check that the body of the linalg operation is just a linalg.yield
804  // operation.
805  Block &body = genericOp.region().front();
806  if (!llvm::hasSingleElement(body))
807  return failure();
808  auto yieldOp = dyn_cast<linalg::YieldOp>(body.getTerminator());
809  if (!yieldOp)
810  return failure();
811 
812  // Get the argument number of the returned values. That is the operand
813  // number to use for replacing uses of this operation.
814  SmallVector<Value> returnedArgs;
815  for (Value yieldVal : yieldOp.values()) {
816  auto yieldArg = yieldVal.dyn_cast<BlockArgument>();
817  if (!yieldArg || yieldArg.getOwner() != &body)
818  return failure();
819  unsigned argumentNumber = yieldArg.getArgNumber();
820  returnedArgs.push_back(genericOp->getOperand(argumentNumber));
821  }
822  if (returnedArgs.size() != genericOp->getNumResults())
823  return failure();
824  rewriter.replaceOp(genericOp, returnedArgs);
825  return success();
826  }
827 };

882 /// Change the type of the result of a `linalg.init_tensor` by making the result
883 /// type statically sized along dimension that in the original operation where
884 /// defined as dynamic, but the size was defined using a `constant` op. For
885 /// example
886 ///
887 /// %c5 = arith.constant 5: index
888 /// %0 = linalg.init_tensor [%arg0, %c5] : tensor<?x?xf32>
889 ///
890 /// to
891 ///
892 /// %0 = linalg.init_tensor [%arg0, 5] : tensor<?x5xf32>
893 struct ReplaceStaticShapeDims : OpRewritePattern<InitTensorOp> {
894  using OpRewritePattern<InitTensorOp>::OpRewritePattern;
895 
896  LogicalResult matchAndRewrite(InitTensorOp op,
897  PatternRewriter &rewriter) const override {
898  SmallVector<Value, 4> dynamicSizes;
899  SmallVector<int64_t, 4> staticSizes;
900  for (unsigned i = 0, e = op.getType().getRank(); i != e; ++i) {
901  // If the size is already static, nothing to do.
902  if (!op.isDynamicSize(i)) {
903  staticSizes.push_back(op.getStaticSize(i));
904  continue;
905  }
906 
907  // If the size is dynamic but defined using a `constant` op, get the
908  // constant value to find the static size to use.
909  unsigned operandNum = op.getIndexOfDynamicSize(i);
910  Value sizeOperand = op.getOperand(operandNum);
911  if (auto constantIndexOp =
912  sizeOperand.getDefiningOp<arith::ConstantIndexOp>()) {
913  staticSizes.push_back(constantIndexOp.value());
914  continue;
915  }
916 
917  // Fallback case. Keep the size dynamic.
918  dynamicSizes.push_back(sizeOperand);
919  staticSizes.push_back(ShapedType::kDynamicSize);
920  }
921  RankedTensorType newType =
922  RankedTensorType::get(staticSizes, op.getType().getElementType());
923  if (newType == op.getType())
924  return failure();
925  auto newOp =
926  rewriter.create<InitTensorOp>(op.getLoc(), newType, dynamicSizes,
927  rewriter.getI64ArrayAttr(staticSizes));
928  rewriter.replaceOpWithNewOp<tensor::CastOp>(op, op.getType(), newOp);
929  return success();
930  }
931 };

1186 LogicalResult PadTensorOp::reifyResultShapes(
1187  OpBuilder &b, ReifiedRankedShapedTypeDims &reifiedReturnShapes) {
1188  Location loc = getLoc();

1190  auto highPad = getMixedHighPad();

1192  for (auto dim : llvm::seq<int64_t>(0, getSourceType().getRank())) {
1193  // Shape along each dimension is source dim + low pad + high pad.
1194  SmallVector<Value> mapOperands;
1195  mapOperands.push_back(b.createOrFold<tensor::DimOp>(loc, source(), dim));
1196  AffineExpr expr = b.getAffineDimExpr(0);
1197  unsigned numSymbols = 0;
1198  auto addOpFoldResult = [&](OpFoldResult valueOrAttr) {
1199  if (Value v = valueOrAttr.dyn_cast<Value>()) {
1200  expr = expr + b.getAffineSymbolExpr(numSymbols++);
1201  mapOperands.push_back(v);
1202  return;
1203  }
1204  int64_t staticValue =
1205  valueOrAttr.get<Attribute>().cast<IntegerAttr>().getInt();
1206  expr = expr + staticValue;
1207  };
1208  addOpFoldResult(lowPad[dim]);
1209  addOpFoldResult(highPad[dim]);
1210  shapes.push_back(applyMapToValues(
1211  b, loc, AffineMap::get(1, numSymbols, expr), mapOperands)[0]);
1212  }
1213  reifiedReturnShapes.emplace_back(std::move(shapes));
1214  return success();
1215 }

1249 Operation *PadTensorOp::getTiledImplementation(OpBuilder &b, ValueRange dest,
1250  ArrayRef<OpFoldResult> offsets,
1251  ArrayRef<OpFoldResult> sizes) {
1252  // Only constant padding value supported.
1253  Value padValue = getConstantPaddingValue();
1254  if (!padValue)
1255  return nullptr;
1256 
1257  // Helper variables and functions for various arithmetic operations. These are
1258  // used extensively for computing new offset/length and padding values.
1259  Location loc = getLoc();
1260  AffineExpr dim0, dim1;
1261  bindDims(b.getContext(), dim0, dim1);
1262  // Add two integers.
1263  auto addMap = AffineMap::get(2, 0, {dim0 + dim1});
1264  auto add = [&](Value v1, Value v2) {
1265  return b.createOrFold<AffineApplyOp>(loc, addMap, ValueRange{v1, v2});
1266  };
1267  // Subtract two integers.
1268  auto subMap = AffineMap::get(2, 0, {dim0 - dim1});
1269  auto sub = [&](Value v1, Value v2) {
1270  return b.createOrFold<AffineApplyOp>(loc, subMap, ValueRange{v1, v2});
1271  };
1272  // Take the minimum of two integers.
1273  auto idMap = AffineMap::getMultiDimIdentityMap(2, b.getContext());
1274  auto min = [&](Value v1, Value v2) {
1275  return b.createOrFold<AffineMinOp>(loc, idMap, ValueRange{v1, v2});
1276  };
1277  // Take the maximum of two integers.
1278  auto max = [&](Value v1, Value v2) {
1279  return b.createOrFold<AffineMaxOp>(loc, idMap, ValueRange{v1, v2});
1280  };
1281  // Zero index-typed integer.
1282  auto zero = b.create<arith::ConstantIndexOp>(loc, 0);
1283 
1284  // Helper function for filling static/dynamic low/high padding indices vectors
1285  // of PadTensorOp.
1286  auto appendIndex = [&](Value val, SmallVector<Value> &dynIndices,
1287  SmallVector<int64_t> &staticIndices) {
1288  if (auto constInt = getConstantIntValue(val)) {
1289  staticIndices.push_back(*constInt);
1290  } else {
1291  staticIndices.push_back(ShapedType::kDynamicSize);
1292  dynIndices.push_back(val);
1293  }
1294  };
1295 
1296  // Compute new offsets, lengths, low padding, high padding.
1297  SmallVector<OpFoldResult> newOffsets, newLengths, newStrides;
1298  SmallVector<Value> newLows, newHighs;
1299  SmallVector<int64_t> staticNewLows, staticNewHighs;
1300  // Set to true if the original data source is not read at all.
1301  bool hasZeroLen = false;
1302  // Same as hasZeroLen, but for dynamic dimension sizes. This condition
1303  // is true if the original data source turns out to be unused at runtime.
1304  Value dynHasZeroLenCond;
1305 
1306  int64_t rank = getSourceType().getRank();
1307  for (unsigned dim = 0; dim < rank; ++dim) {
1308  auto low = getValueOrCreateConstantIndexOp(b, loc, getMixedLowPad()[dim]);
1309  bool hasLowPad = getConstantIntValue(low) != static_cast<int64_t>(0);
1310  auto high = getValueOrCreateConstantIndexOp(b, loc, getMixedHighPad()[dim]);
1311  bool hasHighPad = getConstantIntValue(high) != static_cast<int64_t>(0);
1312  auto offset = getValueOrCreateConstantIndexOp(b, loc, offsets[dim]);
1313  auto length = getValueOrCreateConstantIndexOp(b, loc, sizes[dim]);
1314  auto srcSize = b.createOrFold<tensor::DimOp>(loc, source(), dim);
1315 
1316  // The new amount of low padding is `low - offset`. Except for the case
1317  // where none of the low padding is read. In that case, the new amount of
1318  // low padding is zero.
1319  //
1320  // Optimization: If low = 0, then newLow = 0.
1321  Value newLow = hasLowPad ? max(zero, sub(low, offset)) : zero;
1322  appendIndex(newLow, newLows, staticNewLows);
1323 
1324  // Start reading the data from position `offset - low`. Since the original
1325  // read may have started in the low padding zone, this value could be
1326  // negative. Therefore, start reading from:
1327  //
1328  // max(offset - low, 0)
1329  //
1330  // The original read could also have started in the high padding zone.
1331  // In that case, set the offset to the end of source tensor. The new
1332  // ExtractSliceOp length will be zero in that case. (Effectively reading no
1333  // data from the source.)
1334  //
1335  // Optimization: If low = 0, then the formula can be simplified.
1336  Value newOffset = hasLowPad ? min(max(sub(offset, low), zero), srcSize)
1337  : min(offset, srcSize);
1338  newOffsets.push_back(getAsOpFoldResult(newOffset));
1339 
1340  // The original ExtractSliceOp was reading until position `offset + length`.
1341  // Therefore, the corresponding position within the source tensor is:
1342  //
1343  // offset + length - low
1344  //
1345  // In case the original ExtractSliceOp stopped reading within the low
1346  // padding zone, this value can be negative. In that case, the end position
1347  // of the read should be zero. (Similar to newOffset.)
1348  //
1349  // The original read could also have stopped in the high padding zone.
1350  // In that case, set the end positition of the read should be the end of the
1351  // source tensor. (Similar to newOffset.)
1352  //
1353  // endLoc = min(max(offset - low + length, 0), srcSize)
1354  //
1355  // The new ExtractSliceOp length is `endLoc - newOffset`.
1356  //
1357  // Optimization: If low = 0, then the formula can be simplified.
1358  Value endLoc = hasLowPad
1359  ? min(max(add(sub(offset, low), length), zero), srcSize)
1360  : min(add(offset, length), srcSize);
1361  Value newLength = sub(endLoc, newOffset);
1362  newLengths.push_back(getAsOpFoldResult(newLength));
1363 
1364  // Check if newLength is zero. In that case, no SubTensorOp should be
1365  // executed.
1366  if (auto newLengthInt = getConstantIntValue(newLength)) {
1367  hasZeroLen |= *newLengthInt == 0;
1368  } else {
1369  Value check = b.create<arith::CmpIOp>(loc, arith::CmpIPredicate::eq,
1370  newLength, zero);
1371  dynHasZeroLenCond =
1372  dynHasZeroLenCond
1373  ? b.create<arith::OrIOp>(loc, check, dynHasZeroLenCond)
1374  : check;
1375  }
1376 
1377  // The amount of high padding is simply the number of elements remaining,
1378  // so that the result has the same length as the original ExtractSliceOp.
1379  // As an optimization, if the original high padding is zero, then the new
1380  // high padding must also be zero.
1381  Value newHigh = hasHighPad ? sub(sub(length, newLength), newLow) : zero;
1382  appendIndex(newHigh, newHighs, staticNewHighs);
1383 
1384  // Only unit stride supported.
1385  newStrides.push_back(b.getIndexAttr(1));
1386  }
1387 
1388  // The shape of the result can be obtained from the sizes passed in.
1389  SmallVector<Value> dynDims;
1390  SmallVector<int64_t> shape;
1391  dispatchIndexOpFoldResults(sizes, dynDims, shape, ShapedType::kDynamicSize);
1392  RankedTensorType resultType =
1393  RankedTensorType::get(shape, getResultType().getElementType());
1394 
1395  // Insert cast to ensure that types match. (May be folded away.)
1396  auto castResult = [&](Value val) -> Operation * {
1397  auto castOp = b.create<tensor::CastOp>(loc, resultType, val);
1398  return castOp;
1399  };
1400 
1401  // In cases where the original data source is unused: Emit a GenerateOp and
1402  // do not generate a SliceOp. (The result shape of the SliceOp would
1403  // have a dimension of size 0, the semantics of which is unclear.)
1404  auto createGenerateOp = [&]() {
1405  // Create GenerateOp.
1406  auto generateOp = b.create<tensor::GenerateOp>(
1407  loc, resultType, dynDims,
1408  [&](OpBuilder &builder, Location gLoc, ValueRange indices) {
1409  builder.create<tensor::YieldOp>(gLoc, padValue);
1410  });
1411  return castResult(generateOp);
1412  };
1413 
1414  // Emit a SliceOp and a PadTensorOp. Should not be used in cases where
1415  // the result shape of the new SliceOp has a zero dimension.
1416  auto createPadTensorOfSubTensor = [&]() {
1417  // Create pad_tensor(subtensor(x)).
1418  auto newSliceOp = b.create<tensor::ExtractSliceOp>(
1419  loc, source(), newOffsets, newLengths, newStrides);
1420  auto newPadTensorOp = b.create<PadTensorOp>(
1421  loc, newSliceOp, staticNewLows, staticNewHighs, newLows, newHighs);
1422 
1423  // Copy region to new PadTensorOp.
1424  BlockAndValueMapping bvm;
1425  region().cloneInto(&newPadTensorOp.getRegion(), bvm);
1426 
1427  // Cast result and return.
1428  return castResult(newPadTensorOp);
1429  };
1430 
1431  // Rewrite subtensor(pad_tensor(x)) into a GenerateOp it is statically known
1432  // that the original data source x is not used.
1433  if (hasZeroLen) {
1434  return createGenerateOp();
1435  }
1436 
1437  // If there are dynamic dimensions: Generate an scf.if check to avoid creating
1438  // SliceOps with result dimensions of size 0 at runtime.
1439  if (dynHasZeroLenCond) {
1440  auto result = b.create<scf::IfOp>(
1441  loc, resultType, dynHasZeroLenCond,
1442  /*thenBuilder=*/
1443  [&](OpBuilder &b, Location loc) {
1444  b.create<scf::YieldOp>(loc, createGenerateOp()->getResult(0));
1445  },
1446  /*elseBuilder=*/
1447  [&](OpBuilder &b, Location loc) {
1448  b.create<scf::YieldOp>(loc,
1449  createPadTensorOfSubTensor()->getResult(0));
1450  });
1451  return result;
1452  }
1453  return createPadTensorOfSubTensor();
1454 }

2342 /// A simple, conservative analysis to determine if the loop is shape
2343 /// conserving. I.e., the type of the arg-th yielded value is the same as the
2344 /// type of the corresponding basic block argument of the loop.
2345 /// Note: This function handles only simple cases. Expand as needed.
2346 static bool isShapePreserving(TiledLoopOp loopOp, int64_t arg) {
2347  auto yieldOp = cast<YieldOp>(loopOp.getLoopBody().front().getTerminator());
2348  if (yieldOp.values().empty())
2349  // Tiled loop either has no outputs or is a "memref-based version". In
2350  // either case, the loop is shape conserving.
2351  return true;
2352  assert(arg < static_cast<int64_t>(yieldOp.values().size()) &&
2353  "arg is out of bounds");
2354  Value value = yieldOp.values()[arg];
2355  while (value) {
2356  if (value == loopOp.getRegionOutputArgs()[arg])
2357  return true;
2358  OpResult opResult = value.dyn_cast<OpResult>();
2359  if (!opResult)
2360  return false;
2361 
2362  using tensor::InsertSliceOp;
2363  value = llvm::TypeSwitch<Operation *, Value>(opResult.getOwner())
2364  .template Case<InsertSliceOp>(
2365  [&](InsertSliceOp op) { return op.dest(); })
2366  .template Case<TiledLoopOp>([&](TiledLoopOp loopOp) {
2367  return isShapePreserving(loopOp, opResult.getResultNumber())
2368  ? loopOp.outputs()[opResult.getResultNumber()]
2369  : Value();
2370  })
2371  .Default([&](auto op) { return Value(); });
2372  }
2373  return false;
2374 }

2561  OpBuilder::atBlockEnd(newTiledLoop.getBody(), rewriter.getListener());

2969  op.clone(rewriter, op->getLoc(), newResultTypes, newOperands);
