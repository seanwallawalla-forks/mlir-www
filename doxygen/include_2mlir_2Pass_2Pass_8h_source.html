327 /// Pass to transform an operation of a specific type.
328 ///
329 /// Operation passes must not:
330 /// - modify any other operations within the parent region, as other threads
331 /// may be manipulating them concurrently.
332 /// - modify any state within the parent operation, this includes adding
333 /// additional operations.
334 ///
335 /// Derived function passes are expected to provide the following:
336 /// - A 'void runOnOperation()' method.
337 /// - A 'StringRef getName() const' method.
338 /// - A 'std::unique_ptr<Pass> clonePass() const' method.
339 template <typename OpT = void> class OperationPass : public Pass {
340 protected:
341  OperationPass(TypeID passID) : Pass(passID, OpT::getOperationName()) {}
342  OperationPass(const OperationPass &) = default;
343 
344  /// Support isa/dyn_cast functionality.
345  static bool classof(const Pass *pass) {
346  return pass->getOpName() == OpT::getOperationName();
347  }
348 
349  /// Return the current operation being transformed.
350  OpT getOperation() { return cast<OpT>(Pass::getOperation()); }
351 
352  /// Query an analysis for the current operation of the specific derived
353  /// operation type.
354  template <typename AnalysisT>
355  AnalysisT &getAnalysis() {
356  return Pass::getAnalysis<AnalysisT, OpT>();
357  }
358 };

360 /// Pass to transform an operation.
361 ///
362 /// Operation passes must not:
363 /// - modify any other operations within the parent region, as other threads
364 /// may be manipulating them concurrently.
365 /// - modify any state within the parent operation, this includes adding
366 /// additional operations.
367 ///
368 /// Derived function passes are expected to provide the following:
369 /// - A 'void runOnOperation()' method.
370 /// - A 'StringRef getName() const' method.
371 /// - A 'std::unique_ptr<Pass> clonePass() const' method.
372 template <> class OperationPass<void> : public Pass {
373 protected:
374  OperationPass(TypeID passID) : Pass(passID) {}
375  OperationPass(const OperationPass &) = default;
376 };
377 
378 /// A model for providing function pass specific utilities.
379 ///
380 /// Derived function passes are expected to provide the following:
381 /// - A 'void runOnFunction()' method.
382 /// - A 'StringRef getName() const' method.
383 /// - A 'std::unique_ptr<Pass> clonePass() const' method.
384 class FunctionPass : public OperationPass<FuncOp> {
385 public:
386  using OperationPass<FuncOp>::OperationPass;
387 
388  /// The polymorphic API that runs the pass over the currently held function.
389  virtual void runOnFunction() = 0;
390 
391  /// The polymorphic API that runs the pass over the currently held operation.
392  void runOnOperation() final {
393  if (!getFunction().isExternal())
394  runOnFunction();
395  }
396 
397  /// Return the current function being transformed.
398  FuncOp getFunction() { return this->getOperation(); }
399 };
400 
401 /// This class provides a CRTP wrapper around a base pass class to define
402 /// several necessary utility methods. This should only be used for passes that
403 /// are not suitably represented using the declarative pass specification(i.e.
404 /// tablegen backend).
405 template <typename PassT, typename BaseT> class PassWrapper : public BaseT {
406 public:
407  /// Support isa/dyn_cast functionality for the derived pass class.
408  static bool classof(const Pass *pass) {
409  return pass->getTypeID() == TypeID::get<PassT>();
410  }
411 
412 protected:
413  PassWrapper() : BaseT(TypeID::get<PassT>()) {}
414  PassWrapper(const PassWrapper &) = default;
415 
416  /// Returns the derived pass name.
417  StringRef getName() const override { return llvm::getTypeName<PassT>(); }
418 
419  /// A clone method to create a copy of this pass.
420  std::unique_ptr<Pass> clonePass() const override {
421  return std::make_unique<PassT>(*static_cast<const PassT *>(this));
422  }
423 };
