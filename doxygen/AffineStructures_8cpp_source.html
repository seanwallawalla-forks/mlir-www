43 public:

45  // div's) to existing (dimensional and symbolic) ones. These are always

71 // Flattens the expressions in map. Returns failure if 'expr' was unable to be
72 // flattened (i.e., semi-affine expressions not handled yet).
73 static LogicalResult
74 getFlattenedAffineExprs(ArrayRef<AffineExpr> exprs, unsigned numDims,
75  unsigned numSymbols,
76  std::vector<SmallVector<int64_t, 8>> *flattenedExprs,
77  FlatAffineConstraints *localVarCst) {
78  if (exprs.empty()) {
79  localVarCst->reset(numDims, numSymbols);
80  return success();
81  }
82 
83  AffineExprFlattener flattener(numDims, numSymbols, exprs[0].getContext());
84  // Use the same flattener to simplify each expression successively. This way
85  // local identifiers / expressions are shared.
86  for (auto expr : exprs) {
87  if (!expr.isPureAffine())
88  return failure();
89 
90  flattener.walkPostOrder(expr);
91  }
92 
93  assert(flattener.operandExprStack.size() == exprs.size());
94  flattenedExprs->clear();
95  flattenedExprs->assign(flattener.operandExprStack.begin(),
96  flattener.operandExprStack.end());
97 
98  if (localVarCst)
99  localVarCst->clearAndCopyFrom(flattener.localVarCst);
100 
101  return success();
102 }

158 // Construct from an IntegerSet.
159 FlatAffineConstraints::FlatAffineConstraints(IntegerSet set)
160  : numIds(set.getNumDims() + set.getNumSymbols()), numDims(set.getNumDims()),
161  numSymbols(set.getNumSymbols()),
162  equalities(0, numIds + 1, set.getNumEqualities(), numIds + 1),
163  inequalities(0, numIds + 1, set.getNumInequalities(), numIds + 1) {
164  // Flatten expressions and add them to the constraint system.
165  std::vector<SmallVector<int64_t, 8>> flatExprs;
166  FlatAffineConstraints localVarCst;
167  if (failed(getFlattenedAffineExprs(set, &flatExprs, &localVarCst))) {
168  assert(false && "flattening unimplemented for semi-affine integer sets");
169  return;
170  }
171  assert(flatExprs.size() == set.getNumConstraints());
172  appendLocalId(/*num=*/localVarCst.getNumLocalIds());
173 
174  for (unsigned i = 0, e = flatExprs.size(); i < e; ++i) {
175  const auto &flatExpr = flatExprs[i];
176  assert(flatExpr.size() == getNumCols());
177  if (set.getEqFlags()[i]) {
178  addEquality(flatExpr);
179  } else {
180  addInequality(flatExpr);
181  }
182  }
183  // Add the other constraints involving local id's from flattening.
184  append(localVarCst);
185 }

193 // Construct a hyperrectangular constraint set from ValueRanges that represent
194 // induction variables, lower and upper bounds. `ivs`, `lbs` and `ubs` are
195 // expected to match one to one. The order of variables and constraints is:
196 //
197 // ivs | lbs | ubs | eq/ineq
198 // ----+-----+-----+---------
199 // 1 -1 0 >= 0
200 // ----+-----+-----+---------
201 // -1 0 1 >= 0
202 //
203 // All dimensions as set as DimId.
204 FlatAffineValueConstraints
205 FlatAffineValueConstraints::getHyperrectangular(ValueRange ivs, ValueRange lbs,
206  ValueRange ubs) {
207  FlatAffineValueConstraints res;
208  unsigned nIvs = ivs.size();
209  assert(nIvs == lbs.size() && "expected as many lower bounds as ivs");
210  assert(nIvs == ubs.size() && "expected as many upper bounds as ivs");
211 
212  if (nIvs == 0)
213  return res;
214 
215  res.appendDimId(ivs);
216  unsigned lbsStart = res.appendDimId(lbs);
217  unsigned ubsStart = res.appendDimId(ubs);
218 
219  MLIRContext *ctx = ivs.front().getContext();
220  for (int ivIdx = 0, e = nIvs; ivIdx < e; ++ivIdx) {
221  // iv - lb >= 0
222  AffineMap lb = AffineMap::get(/*dimCount=*/3 * nIvs, /*symbolCount=*/0,
223  getAffineDimExpr(lbsStart + ivIdx, ctx));
224  if (failed(res.addBound(BoundType::LB, ivIdx, lb)))
225  llvm_unreachable("Unexpected FlatAffineValueConstraints creation error");
226  // -iv + ub >= 0
227  AffineMap ub = AffineMap::get(/*dimCount=*/3 * nIvs, /*symbolCount=*/0,
228  getAffineDimExpr(ubsStart + ivIdx, ctx));
229  if (failed(res.addBound(BoundType::UB, ivIdx, ub)))
230  llvm_unreachable("Unexpected FlatAffineValueConstraints creation error");
231  }
232  return res;
233 }

491 /// Merge and align the identifiers of A and B starting at 'offset', so that
492 /// both constraint systems get the union of the contained identifiers that is
493 /// dimension-wise and symbol-wise unique; both constraint systems are updated
494 /// so that they have the union of all identifiers, with A's original
495 /// identifiers appearing first followed by any of B's identifiers that didn't
496 /// appear in A. Local identifiers in B that have the same division
497 /// representation as local identifiers in A are merged into one.
498 // E.g.: Input: A has ((%i, %j) [%M, %N]) and B has (%k, %j) [%P, %N, %M])
499 // Output: both A, B have (%i, %j, %k) [%M, %N, %P]
500 static void mergeAndAlignIds(unsigned offset, FlatAffineValueConstraints *a,
501  FlatAffineValueConstraints *b) {
502  assert(offset <= a->getNumDimIds() && offset <= b->getNumDimIds());
503  // A merge/align isn't meaningful if a cst's ids aren't distinct.
504  assert(areIdsUnique(*a) && "A's values aren't unique");
505  assert(areIdsUnique(*b) && "B's values aren't unique");
506 
507  assert(std::all_of(a->getMaybeValues().begin() + offset,
508  a->getMaybeValues().begin() + a->getNumDimAndSymbolIds(),
509  [](Optional<Value> id) { return id.hasValue(); }));
510 
511  assert(std::all_of(b->getMaybeValues().begin() + offset,
512  b->getMaybeValues().begin() + b->getNumDimAndSymbolIds(),
513  [](Optional<Value> id) { return id.hasValue(); }));
514 
515  SmallVector<Value, 4> aDimValues;
516  a->getValues(offset, a->getNumDimIds(), &aDimValues);
517 
518  {
519  // Merge dims from A into B.
520  unsigned d = offset;
521  for (auto aDimValue : aDimValues) {
522  unsigned loc;
523  if (b->findId(aDimValue, &loc)) {
524  assert(loc >= offset && "A's dim appears in B's aligned range");
525  assert(loc < b->getNumDimIds() &&
526  "A's dim appears in B's non-dim position");
527  b->swapId(d, loc);
528  } else {
529  b->insertDimId(d, aDimValue);
530  }
531  d++;
532  }
533  // Dimensions that are in B, but not in A, are added at the end.
534  for (unsigned t = a->getNumDimIds(), e = b->getNumDimIds(); t < e; t++) {
535  a->appendDimId(b->getValue(t));
536  }
537  assert(a->getNumDimIds() == b->getNumDimIds() &&
538  "expected same number of dims");
539  }
540 
541  // Merge and align symbols of A and B
542  a->mergeSymbolIds(*b);
543  // Merge and align local ids of A and B
544  a->mergeLocalIds(*b);
545 
546  assert(areIdsAligned(*a, *b) && "IDs expected to be aligned");
547 }

561 // Similar to `composeMap` except that no Values need be associated with the
562 // constraint system nor are they looked at -- the dimensions and symbols of
563 // `other` are expected to correspond 1:1 to `this` system.
564 LogicalResult FlatAffineConstraints::composeMatchingMap(AffineMap other) {
565  assert(other.getNumDims() == getNumDimIds() && "dim mismatch");
566  assert(other.getNumSymbols() == getNumSymbolIds() && "symbol mismatch");
567 
568  std::vector<SmallVector<int64_t, 8>> flatExprs;
569  if (failed(flattenAlignedMapAndMergeLocals(other, &flatExprs)))
570  return failure();
571  assert(flatExprs.size() == other.getNumResults());
572 
573  // Add dimensions corresponding to the map's results.
574  insertDimId(/*pos=*/0, /*num=*/other.getNumResults());
575 
576  // We add one equality for each result connecting the result dim of the map to
577  // the other identifiers.
578  // E.g.: if the expression is 16*i0 + i1, and this is the r^th
579  // iteration/result of the value map, we are adding the equality:
580  // d_r - 16*i0 - i1 = 0. Similarly, when flattening (i0 + 1, i0 + 8*i2), we
581  // add two equalities: d_0 - i0 - 1 == 0, d1 - i0 - 8*i2 == 0.
582  for (unsigned r = 0, e = flatExprs.size(); r < e; r++) {
583  const auto &flatExpr = flatExprs[r];
584  assert(flatExpr.size() >= other.getNumInputs() + 1);
585 
586  SmallVector<int64_t, 8> eqToAdd(getNumCols(), 0);
587  // Set the coefficient for this result to one.
588  eqToAdd[r] = 1;
589 
590  // Dims and symbols.
591  for (unsigned i = 0, f = other.getNumInputs(); i < f; i++) {
592  // Negate `eq[r]` since the newly added dimension will be set to this one.
593  eqToAdd[e + i] = -flatExpr[i];
594  }
595  // Local columns of `eq` are at the beginning.
596  unsigned j = getNumDimIds() + getNumSymbolIds();
597  unsigned end = flatExpr.size() - 1;
598  for (unsigned i = other.getNumInputs(); i < end; i++, j++) {
599  eqToAdd[j] = -flatExpr[i];
600  }
601 
602  // Constant term.
603  eqToAdd[getNumCols() - 1] = -flatExpr[flatExpr.size() - 1];
604 
605  // Add the equality connecting the result of the map to this constraint set.
606  addEquality(eqToAdd);
607  }
608 
609  return success();

623 /// of symbols that are unique. Symbols in `this` and `other` should be
624 /// unique. Symbols with Value as `None` are considered to be inequal to all

627  FlatAffineValueConstraints &other) {

697 LogicalResult
698 FlatAffineValueConstraints::addAffineForOpDomain(AffineForOp forOp) {
699  unsigned pos;
700  // Pre-condition for this method.
701  if (!findId(forOp.getInductionVar(), &pos)) {
702  assert(false && "Value not found");
703  return failure();
704  }
705 
706  int64_t step = forOp.getStep();
707  if (step != 1) {
708  if (!forOp.hasConstantLowerBound())
709  LLVM_DEBUG(forOp.emitWarning("domain conservatively approximated"));
710  else {
711  // Add constraints for the stride.
712  // (iv - lb) % step = 0 can be written as:
713  // (iv - lb) - step * q = 0 where q = (iv - lb) / step.
714  // Add local variable 'q' and add the above equality.
715  // The first constraint is q = (iv - lb) floordiv step
716  SmallVector<int64_t, 8> dividend(getNumCols(), 0);
717  int64_t lb = forOp.getConstantLowerBound();
718  dividend[pos] = 1;
719  dividend.back() -= lb;
720  addLocalFloorDiv(dividend, step);
721  // Second constraint: (iv - lb) - step * q = 0.
722  SmallVector<int64_t, 8> eq(getNumCols(), 0);
723  eq[pos] = 1;
724  eq.back() -= lb;
725  // For the local var just added above.
726  eq[getNumCols() - 2] = -step;
727  addEquality(eq);
728  }
729  }
730 
731  if (forOp.hasConstantLowerBound()) {
732  addBound(BoundType::LB, pos, forOp.getConstantLowerBound());
733  } else {
734  // Non-constant lower bound case.
735  if (failed(addBound(BoundType::LB, pos, forOp.getLowerBoundMap(),
736  forOp.getLowerBoundOperands())))
737  return failure();
738  }
739 
740  if (forOp.hasConstantUpperBound()) {
741  addBound(BoundType::UB, pos, forOp.getConstantUpperBound() - 1);
742  return success();
743  }
744  // Non-constant upper bound case.
745  return addBound(BoundType::UB, pos, forOp.getUpperBoundMap(),
746  forOp.getUpperBoundOperands());
747 }

833 // Normalizes the coefficient values across all columns in `rowIdx` by their
834 // GCD in equality or inequality constraints as specified by `isEq`.
835 template <bool isEq>
836 static void normalizeConstraintByGCD(FlatAffineConstraints *constraints,
837  unsigned rowIdx) {
838  auto at = [&](unsigned colIdx) -> int64_t {
839  return isEq ? constraints->atEq(rowIdx, colIdx)
840  : constraints->atIneq(rowIdx, colIdx);
841  };
842  uint64_t gcd = std::abs(at(0));
843  for (unsigned j = 1, e = constraints->getNumCols(); j < e; ++j) {
844  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(at(j)));
845  }
846  if (gcd > 0 && gcd != 1) {
847  for (unsigned j = 0, e = constraints->getNumCols(); j < e; ++j) {
848  int64_t v = at(j) / static_cast<int64_t>(gcd);
849  isEq ? constraints->atEq(rowIdx, j) = v
850  : constraints->atIneq(rowIdx, j) = v;
851  }
852  }

882 bool FlatAffineConstraints::hasInvalidConstraint() const {
883  assert(hasConsistentState());
884  auto check = [&](bool isEq) -> bool {
885  unsigned numCols = getNumCols();
886  unsigned numRows = isEq ? getNumEqualities() : getNumInequalities();
887  for (unsigned i = 0, e = numRows; i < e; ++i) {
888  unsigned j;
889  for (j = 0; j < numCols - 1; ++j) {
890  int64_t v = isEq ? atEq(i, j) : atIneq(i, j);
891  // Skip rows with non-zero variable coefficients.
892  if (v != 0)
893  break;
894  }
895  if (j < numCols - 1) {
896  continue;
897  }
898  // Check validity of constant term at 'numCols - 1' w.r.t 'isEq'.
899  // Example invalid constraints include: '1 == 0' or '-1 >= 0'
900  int64_t v = isEq ? atEq(i, numCols - 1) : atIneq(i, numCols - 1);
901  if ((isEq && v != 0) || (!isEq && v < 0)) {
902  return true;
903  }
904  }
905  return false;
906  };
907  if (check(/*isEq=*/true))
908  return true;
909  return check(/*isEq=*/false);
910 }

912 /// Eliminate identifier from constraint at `rowIdx` based on coefficient at
913 /// pivotRow, pivotCol. Columns in range [elimColStart, pivotCol) will not be
914 /// updated as they have already been eliminated.
915 static void eliminateFromConstraint(FlatAffineConstraints *constraints,
916  unsigned rowIdx, unsigned pivotRow,
917  unsigned pivotCol, unsigned elimColStart,
918  bool isEq) {
919  // Skip if equality 'rowIdx' if same as 'pivotRow'.
920  if (isEq && rowIdx == pivotRow)
921  return;
922  auto at = [&](unsigned i, unsigned j) -> int64_t {
923  return isEq ? constraints->atEq(i, j) : constraints->atIneq(i, j);
924  };
925  int64_t leadCoeff = at(rowIdx, pivotCol);
926  // Skip if leading coefficient at 'rowIdx' is already zero.
927  if (leadCoeff == 0)
928  return;
929  int64_t pivotCoeff = constraints->atEq(pivotRow, pivotCol);
930  int64_t sign = (leadCoeff * pivotCoeff > 0) ? -1 : 1;
931  int64_t lcm = mlir::lcm(pivotCoeff, leadCoeff);
932  int64_t pivotMultiplier = sign * (lcm / std::abs(pivotCoeff));
933  int64_t rowMultiplier = lcm / std::abs(leadCoeff);
934 
935  unsigned numCols = constraints->getNumCols();
936  for (unsigned j = 0; j < numCols; ++j) {
937  // Skip updating column 'j' if it was just eliminated.
938  if (j >= elimColStart && j < pivotCol)
939  continue;
940  int64_t v = pivotMultiplier * constraints->atEq(pivotRow, j) +
941  rowMultiplier * at(rowIdx, j);
942  isEq ? constraints->atEq(rowIdx, j) = v
943  : constraints->atIneq(rowIdx, j) = v;
944  }
945 }

988 /// Returns the position of the identifier that has the minimum <number of lower
989 /// bounds> times <number of upper bounds> from the specified range of
990 /// identifiers [start, end). It is often best to eliminate in the increasing
991 /// order of these counts when doing Fourier-Motzkin elimination since FM adds
992 /// that many new constraints.
993 static unsigned getBestIdToEliminate(const FlatAffineConstraints &cst,
994  unsigned start, unsigned end) {
995  assert(start < cst.getNumIds() && end < cst.getNumIds() + 1);
996 
997  auto getProductOfNumLowerUpperBounds = [&](unsigned pos) {
998  unsigned numLb = 0;
999  unsigned numUb = 0;
1000  for (unsigned r = 0, e = cst.getNumInequalities(); r < e; r++) {
1001  if (cst.atIneq(r, pos) > 0) {
1002  ++numLb;
1003  } else if (cst.atIneq(r, pos) < 0) {
1004  ++numUb;
1005  }
1006  }
1007  return numLb * numUb;
1008  };
1009

1011  unsigned min = getProductOfNumLowerUpperBounds(start);
1012  for (unsigned c = start + 1; c < end; c++) {
1013  unsigned numLbUbProduct = getProductOfNumLowerUpperBounds(c);
1014  if (numLbUbProduct < min) {
1015  min = numLbUbProduct;
1016  minLoc = c;
1017  }

1022 // Checks for emptiness of the set by eliminating identifiers successively and
1023 // using the GCD test (on all equality constraints) and checking for trivially
1024 // invalid constraints. Returns 'true' if the constraint system is found to be
1025 // empty; false otherwise.
1026 bool FlatAffineConstraints::isEmpty() const {
1027  if (isEmptyByGCDTest() || hasInvalidConstraint())
1028  return true;
1029 
1030  FlatAffineConstraints tmpCst(*this);
1031 
1032  // First, eliminate as many local variables as possible using equalities.
1033  tmpCst.removeRedundantLocalVars();
1034  if (tmpCst.isEmptyByGCDTest() || tmpCst.hasInvalidConstraint())
1035  return true;
1036 
1037  // Eliminate as many identifiers as possible using Gaussian elimination.
1038  unsigned currentPos = 0;
1039  while (currentPos < tmpCst.getNumIds()) {
1040  tmpCst.gaussianEliminateIds(currentPos, tmpCst.getNumIds());
1041  ++currentPos;
1042  // We check emptiness through trivial checks after eliminating each ID to
1043  // detect emptiness early. Since the checks isEmptyByGCDTest() and
1044  // hasInvalidConstraint() are linear time and single sweep on the constraint
1045  // buffer, this appears reasonable - but can optimize in the future.
1046  if (tmpCst.hasInvalidConstraint() || tmpCst.isEmptyByGCDTest())
1047  return true;
1048  }
1049 
1050  // Eliminate the remaining using FM.
1051  for (unsigned i = 0, e = tmpCst.getNumIds(); i < e; i++) {
1052  tmpCst.fourierMotzkinEliminate(
1053  getBestIdToEliminate(tmpCst, 0, tmpCst.getNumIds()));
1054  // Check for a constraint explosion. This rarely happens in practice, but
1055  // this check exists as a safeguard against improperly constructed
1056  // constraint systems or artificially created arbitrarily complex systems
1057  // that aren't the intended use case for FlatAffineConstraints. This is
1058  // needed since FM has a worst case exponential complexity in theory.
1059  if (tmpCst.getNumConstraints() >= kExplosionFactor * getNumIds()) {
1060  LLVM_DEBUG(llvm::dbgs() << "FM constraint explosion detected\n");
1061  return false;
1062  }
1063 
1064  // FM wouldn't have modified the equalities in any way. So no need to again
1065  // run GCD test. Check for trivial invalid constraints.
1066  if (tmpCst.hasInvalidConstraint())
1067  return true;
1068  }
1069  return false;
1070 }

1072 // Runs the GCD test on all equality constraints. Returns 'true' if this test
1073 // fails on any equality. Returns 'false' otherwise.
1074 // This test can be used to disprove the existence of a solution. If it returns
1075 // true, no integer solution to the equality constraints can exist.
1076 //
1077 // GCD test definition:
1078 //
1079 // The equality constraint:
1080 //
1081 // c_1*x_1 + c_2*x_2 + ... + c_n*x_n = c_0
1082 //
1083 // has an integer solution iff:
1084 //
1085 // GCD of c_1, c_2, ..., c_n divides c_0.
1086 //
1087 bool FlatAffineConstraints::isEmptyByGCDTest() const {
1088  assert(hasConsistentState());
1089  unsigned numCols = getNumCols();
1090  for (unsigned i = 0, e = getNumEqualities(); i < e; ++i) {
1091  uint64_t gcd = std::abs(atEq(i, 0));
1092  for (unsigned j = 1; j < numCols - 1; ++j) {
1093  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(atEq(i, j)));
1094  }
1095  int64_t v = std::abs(atEq(i, numCols - 1));
1096  if (gcd > 0 && (v % gcd != 0)) {
1097  return true;
1098  }
1099  }
1100  return false;
1101 }

1103 // Returns a matrix where each row is a vector along which the polytope is
1104 // bounded. The span of the returned vectors is guaranteed to contain all
1105 // such vectors. The returned vectors are NOT guaranteed to be linearly
1106 // independent. This function should not be called on empty sets.
1107 //
1108 // It is sufficient to check the perpendiculars of the constraints, as the set
1109 // of perpendiculars which are bounded must span all bounded directions.
1110 Matrix FlatAffineConstraints::getBoundedDirections() const {
1111  // Note that it is necessary to add the equalities too (which the constructor
1112  // does) even though we don't need to check if they are bounded; whether an
1113  // inequality is bounded or not depends on what other constraints, including
1114  // equalities, are present.
1115  Simplex simplex(*this);
1116 
1117  assert(!simplex.isEmpty() && "It is not meaningful to ask whether a "
1118  "direction is bounded in an empty set.");
1119 
1120  SmallVector<unsigned, 8> boundedIneqs;
1121  // The constructor adds the inequalities to the simplex first, so this
1122  // processes all the inequalities.
1123  for (unsigned i = 0, e = getNumInequalities(); i < e; ++i) {
1124  if (simplex.isBoundedAlongConstraint(i))
1125  boundedIneqs.push_back(i);
1126  }
1127 
1128  // The direction vector is given by the coefficients and does not include the
1129  // constant term, so the matrix has one fewer column.
1130  unsigned dirsNumCols = getNumCols() - 1;
1131  Matrix dirs(boundedIneqs.size() + getNumEqualities(), dirsNumCols);
1132 
1133  // Copy the bounded inequalities.
1134  unsigned row = 0;
1135  for (unsigned i : boundedIneqs) {
1136  for (unsigned col = 0; col < dirsNumCols; ++col)
1137  dirs(row, col) = atIneq(i, col);
1138  ++row;
1139  }
1140

1182 /// Let this set be S. If S is bounded then we directly call into the GBR
1183 /// sampling algorithm. Otherwise, there are some unbounded directions, i.e.,
1184 /// vectors v such that S extends to infinity along v or -v. In this case we
1185 /// use an algorithm described in the integer set library (isl) manual and used
1186 /// by the isl_set_sample function in that library. The algorithm is:
1187 ///
1188 /// 1) Apply a unimodular transform T to S to obtain S*T, such that all
1189 /// dimensions in which S*T is bounded lie in the linear span of a prefix of the
1190 /// dimensions.
1191 ///
1192 /// 2) Construct a set B by removing all constraints that involve
1193 /// the unbounded dimensions and then deleting the unbounded dimensions. Note
1194 /// that B is a Bounded set.
1195 ///
1196 /// 3) Try to obtain a sample from B using the GBR sampling
1197 /// algorithm. If no sample is found, return that S is empty.
1198 ///
1199 /// 4) Otherwise, substitute the obtained sample into S*T to obtain a set
1200 /// C. C is a full-dimensional Cone and always contains a sample.
1201 ///
1202 /// 5) Obtain an integer sample from C.
1203 ///
1204 /// 6) Return T*v, where v is the concatenation of the samples from B and C.
1205 ///
1206 /// The following is a sketch of a proof that
1207 /// a) If the algorithm returns empty, then S is empty.
1208 /// b) If the algorithm returns a sample, it is a valid sample in S.
1209 ///
1210 /// The algorithm returns empty only if B is empty, in which case S*T is
1211 /// certainly empty since B was obtained by removing constraints and then
1212 /// deleting unconstrained dimensions from S*T. Since T is unimodular, a vector
1213 /// v is in S*T iff T*v is in S. So in this case, since
1214 /// S*T is empty, S is empty too.
1215 ///
1216 /// Otherwise, the algorithm substitutes the sample from B into S*T. All the
1217 /// constraints of S*T that did not involve unbounded dimensions are satisfied
1218 /// by this substitution. All dimensions in the linear span of the dimensions
1219 /// outside the prefix are unbounded in S*T (step 1). Substituting values for
1220 /// the bounded dimensions cannot make these dimensions bounded, and these are
1221 /// the only remaining dimensions in C, so C is unbounded along every vector (in
1222 /// the positive or negative direction, or both). C is hence a full-dimensional
1223 /// cone and therefore always contains an integer point.
1224 ///
1225 /// Concatenating the samples from B and C gives a sample v in S*T, so the
1226 /// returned sample T*v is a sample in S.
1227 Optional<SmallVector<int64_t, 8>>
1228 FlatAffineConstraints::findIntegerSample() const {
1229  // First, try the GCD test heuristic.
1230  if (isEmptyByGCDTest())
1231  return {};
1232 
1233  Simplex simplex(*this);
1234  if (simplex.isEmpty())
1235  return {};
1236 
1237  // For a bounded set, we directly call into the GBR sampling algorithm.
1238  if (!simplex.isUnbounded())
1239  return simplex.findIntegerSample();
1240 
1241  // The set is unbounded. We cannot directly use the GBR algorithm.
1242  //
1243  // m is a matrix containing, in each row, a vector in which S is
1244  // bounded, such that the linear span of all these dimensions contains all
1245  // bounded dimensions in S.
1246  Matrix m = getBoundedDirections();
1247  // In column echelon form, each row of m occupies only the first rank(m)
1248  // columns and has zeros on the other columns. The transform T that brings S
1249  // to column echelon form is unimodular as well, so this is a suitable
1250  // transform to use in step 1 of the algorithm.
1251  std::pair<unsigned, LinearTransform> result =
1252  LinearTransform::makeTransformToColumnEchelon(std::move(m));
1253  const LinearTransform &transform = result.second;
1254  // 1) Apply T to S to obtain S*T.
1255  FlatAffineConstraints transformedSet = transform.applyTo(*this);
1256 
1257  // 2) Remove the unbounded dimensions and constraints involving them to
1258  // obtain a bounded set.
1259  FlatAffineConstraints boundedSet = transformedSet;
1260  unsigned numBoundedDims = result.first;
1261  unsigned numUnboundedDims = getNumIds() - numBoundedDims;
1262  removeConstraintsInvolvingSuffixDims(boundedSet, numUnboundedDims);
1263  boundedSet.removeIdRange(numBoundedDims, boundedSet.getNumIds());
1264 
1265  // 3) Try to obtain a sample from the bounded set.
1266  Optional<SmallVector<int64_t, 8>> boundedSample =
1267  Simplex(boundedSet).findIntegerSample();
1268  if (!boundedSample)
1269  return {};
1270  assert(boundedSet.containsPoint(*boundedSample) &&
1271  "Simplex returned an invalid sample!");
1272 
1273  // 4) Substitute the values of the bounded dimensions into S*T to obtain a
1274  // full-dimensional cone, which necessarily contains an integer sample.
1275  transformedSet.setAndEliminate(0, *boundedSample);
1276  FlatAffineConstraints &cone = transformedSet;
1277 
1278  // 5) Obtain an integer sample from the cone.
1279  //
1280  // We shrink the cone such that for any rational point in the shrunken cone,
1281  // rounding up each of the point's coordinates produces a point that still
1282  // lies in the original cone.
1283  //
1284  // Rounding up a point x adds a number e_i in [0, 1) to each coordinate x_i.
1285  // For each inequality sum_i a_i x_i + c >= 0 in the original cone, the
1286  // shrunken cone will have the inequality tightened by some amount s, such
1287  // that if x satisfies the shrunken cone's tightened inequality, then x + e
1288  // satisfies the original inequality, i.e.,
1289  //
1290  // sum_i a_i x_i + c + s >= 0 implies sum_i a_i (x_i + e_i) + c >= 0
1291  //
1292  // for any e_i values in [0, 1). In fact, we will handle the slightly more
1293  // general case where e_i can be in [0, 1]. For example, consider the
1294  // inequality 2x_1 - 3x_2 - 7x_3 - 6 >= 0, and let x = (3, 0, 0). How low
1295  // could the LHS go if we added a number in [0, 1] to each coordinate? The LHS
1296  // is minimized when we add 1 to the x_i with negative coefficient a_i and
1297  // keep the other x_i the same. In the example, we would get x = (3, 1, 1),
1298  // changing the value of the LHS by -3 + -7 = -10.
1299  //
1300  // In general, the value of the LHS can change by at most the sum of the
1301  // negative a_i, so we accomodate this by shifting the inequality by this
1302  // amount for the shrunken cone.
1303  for (unsigned i = 0, e = cone.getNumInequalities(); i < e; ++i) {
1304  for (unsigned j = 0; j < cone.numIds; ++j) {
1305  int64_t coeff = cone.atIneq(i, j);
1306  if (coeff < 0)
1307  cone.atIneq(i, cone.numIds) += coeff;
1308  }
1309  }
1310 
1311  // Obtain an integer sample in the cone by rounding up a rational point from
1312  // the shrunken cone. Shrinking the cone amounts to shifting its apex
1313  // "inwards" without changing its "shape"; the shrunken cone is still a
1314  // full-dimensional cone and is hence non-empty.
1315  Simplex shrunkenConeSimplex(cone);
1316  assert(!shrunkenConeSimplex.isEmpty() && "Shrunken cone cannot be empty!");
1317  SmallVector<Fraction, 8> shrunkenConeSample =
1318  shrunkenConeSimplex.getRationalSample();
1319 
1320  SmallVector<int64_t, 8> coneSample(llvm::map_range(shrunkenConeSample, ceil));
1321 
1322  // 6) Return transform * concat(boundedSample, coneSample).
1323  SmallVector<int64_t, 8> &sample = boundedSample.getValue();
1324  sample.append(coneSample.begin(), coneSample.end());
1325  return transform.preMultiplyColumn(sample);
1326 }

1355 /// Check if the pos^th identifier can be represented as a division using upper
1356 /// bound inequality at position `ubIneq` and lower bound inequality at position
1357 /// `lbIneq`.
1358 ///
1359 /// Let `id` be the pos^th identifier, then `id` is equivalent to
1360 /// `expr floordiv divisor` if there are constraints of the form:
1361 /// 0 <= expr - divisor * id <= divisor - 1
1362 /// Rearranging, we have:
1363 /// divisor * id - expr + (divisor - 1) >= 0 <-- Lower bound for 'id'
1364 /// -divisor * id + expr >= 0 <-- Upper bound for 'id'
1365 ///
1366 /// For example:
1367 /// 32*k >= 16*i + j - 31 <-- Lower bound for 'k'
1368 /// 32*k <= 16*i + j <-- Upper bound for 'k'
1369 /// expr = 16*i + j, divisor = 32
1370 /// k = ( 16*i + j ) floordiv 32
1371 ///
1372 /// 4q >= i + j - 2 <-- Lower bound for 'q'
1373 /// 4q <= i + j + 1 <-- Upper bound for 'q'
1374 /// expr = i + j + 1, divisor = 4
1375 /// q = (i + j + 1) floordiv 4
1376 ///
1377 /// If successful, `expr` is set to dividend of the division and `divisor` is
1378 /// set to the denominator of the division.
1379 static LogicalResult getDivRepr(const FlatAffineConstraints &cst, unsigned pos,
1380  unsigned ubIneq, unsigned lbIneq,
1381  SmallVector<int64_t, 8> &expr,
1382  unsigned &divisor) {
1383 
1384  assert(pos <= cst.getNumIds() && "Invalid identifier position");
1385  assert(ubIneq <= cst.getNumInequalities() &&
1386  "Invalid upper bound inequality position");
1387  assert(lbIneq <= cst.getNumInequalities() &&
1388  "Invalid upper bound inequality position");
1389 
1390  // Due to the form of the inequalities, sum of constants of the
1391  // inequalities is (divisor - 1).
1392  int64_t denominator = cst.atIneq(lbIneq, cst.getNumCols() - 1) +
1393  cst.atIneq(ubIneq, cst.getNumCols() - 1) + 1;
1394 
1395  // Divisor should be positive.
1396  if (denominator <= 0)
1397  return failure();
1398 
1399  // Check if coeff of variable is equal to divisor.
1400  if (denominator != cst.atIneq(lbIneq, pos))
1401  return failure();
1402 
1403  // Check if constraints are opposite of each other. Constant term
1404  // is not required to be opposite and is not checked.
1405  unsigned i = 0, e = 0;
1406  for (i = 0, e = cst.getNumIds(); i < e; ++i)
1407  if (cst.atIneq(ubIneq, i) != -cst.atIneq(lbIneq, i))
1408  break;
1409 
1410  if (i < e)
1411  return failure();
1412 
1413  // Set expr with dividend of the division.
1414  SmallVector<int64_t, 8> dividend(cst.getNumCols());
1415  for (i = 0, e = cst.getNumCols(); i < e; ++i)
1416  if (i != pos)
1417  dividend[i] = cst.atIneq(ubIneq, i);
1418  expr = dividend;
1419 
1420  // Set divisor.
1421  divisor = denominator;
1422 
1423  return success();
1424 }

1529 /// Tightens inequalities given that we are dealing with integer spaces. This is
1530 /// analogous to the GCD test but applied to inequalities. The constant term can
1531 /// be reduced to the preceding multiple of the GCD of the coefficients, i.e.,
1532 /// 64*i - 100 >= 0 => 64*i - 128 >= 0 (since 'i' is an integer). This is a
1533 /// fast method - linear in the number of coefficients.
1534 // Example on how this affects practical cases: consider the scenario:
1535 // 64*i >= 100, j = 64*i; without a tightening, elimination of i would yield
1536 // j >= 100 instead of the tighter (exact) j >= 128.

1540  uint64_t gcd = std::abs(atIneq(i, 0));
1541  for (unsigned j = 1; j < numCols - 1; ++j) {
1542  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(atIneq(i, j)));
1543  }
1544  if (gcd > 0 && gcd != 1) {
1545  int64_t gcdI = static_cast<int64_t>(gcd);
1546  // Tighten the constant term and normalize the constraint by the GCD.
1547  atIneq(i, numCols - 1) = mlir::floorDiv(atIneq(i, numCols - 1), gcdI);
1548  for (unsigned j = 0, e = numCols - 1; j < e; ++j)
1549  atIneq(i, j) /= gcdI;

1606 // Determine whether the identifier at 'pos' (say id_r) can be expressed as
1607 // modulo of another known identifier (say id_n) w.r.t a constant. For example,
1608 // if the following constraints hold true:
1609 // ```
1610 // 0 <= id_r <= divisor - 1
1611 // id_n - (divisor * q_expr) = id_r
1612 // ```
1613 // where `id_n` is a known identifier (called dividend), and `q_expr` is an
1614 // `AffineExpr` (called the quotient expression), `id_r` can be written as:
1615 //
1616 // `id_r = id_n mod divisor`.
1617 //
1618 // Additionally, in a special case of the above constaints where `q_expr` is an
1619 // identifier itself that is not yet known (say `id_q`), it can be written as a
1620 // floordiv in the following way:
1621 //
1622 // `id_q = id_n floordiv divisor`.
1623 //
1624 // Returns true if the above mod or floordiv are detected, updating 'memo' with
1625 // these new expressions. Returns false otherwise.
1626 static bool detectAsMod(const FlatAffineConstraints &cst, unsigned pos,
1627  int64_t lbConst, int64_t ubConst,
1628  SmallVectorImpl<AffineExpr> &memo,
1629  MLIRContext *context) {
1630  assert(pos < cst.getNumIds() && "invalid position");
1631 
1632  // Check if a divisor satisfying the condition `0 <= id_r <= divisor - 1` can
1633  // be determined.
1634  if (lbConst != 0 || ubConst < 1)
1635  return false;
1636  int64_t divisor = ubConst + 1;
1637 
1638  // Check for the aforementioned conditions in each equality.
1639  for (unsigned curEquality = 0, numEqualities = cst.getNumEqualities();
1640  curEquality < numEqualities; curEquality++) {
1641  int64_t coefficientAtPos = cst.atEq(curEquality, pos);
1642  // If current equality does not involve `id_r`, continue to the next
1643  // equality.
1644  if (coefficientAtPos == 0)
1645  continue;
1646 
1647  // Constant term should be 0 in this equality.
1648  if (cst.atEq(curEquality, cst.getNumCols() - 1) != 0)
1649  continue;
1650 
1651  // Traverse through the equality and construct the dividend expression
1652  // `dividendExpr`, to contain all the identifiers which are known and are
1653  // not divisible by `(coefficientAtPos * divisor)`. Hope here is that the
1654  // `dividendExpr` gets simplified into a single identifier `id_n` discussed
1655  // above.
1656  auto dividendExpr = getAffineConstantExpr(0, context);
1657 
1658  // Track the terms that go into quotient expression, later used to detect
1659  // additional floordiv.
1660  unsigned quotientCount = 0;
1661  int quotientPosition = -1;
1662  int quotientSign = 1;
1663 
1664  // Consider each term in the current equality.
1665  unsigned curId, e;
1666  for (curId = 0, e = cst.getNumDimAndSymbolIds(); curId < e; ++curId) {
1667  // Ignore id_r.
1668  if (curId == pos)
1669  continue;
1670  int64_t coefficientOfCurId = cst.atEq(curEquality, curId);
1671  // Ignore ids that do not contribute to the current equality.
1672  if (coefficientOfCurId == 0)
1673  continue;
1674  // Check if the current id goes into the quotient expression.
1675  if (coefficientOfCurId % (divisor * coefficientAtPos) == 0) {
1676  quotientCount++;
1677  quotientPosition = curId;
1678  quotientSign = (coefficientOfCurId * coefficientAtPos) > 0 ? 1 : -1;
1679  continue;
1680  }
1681  // Identifiers that are part of dividendExpr should be known.
1682  if (!memo[curId])
1683  break;
1684  // Append the current identifier to the dividend expression.
1685  dividendExpr = dividendExpr + memo[curId] * coefficientOfCurId;
1686  }
1687 
1688  // Can't construct expression as it depends on a yet uncomputed id.
1689  if (curId < e)
1690  continue;
1691 
1692  // Express `id_r` in terms of the other ids collected so far.
1693  if (coefficientAtPos > 0)
1694  dividendExpr = (-dividendExpr).floorDiv(coefficientAtPos);
1695  else
1696  dividendExpr = dividendExpr.floorDiv(-coefficientAtPos);
1697 
1698  // Simplify the expression.
1699  dividendExpr = simplifyAffineExpr(dividendExpr, cst.getNumDimIds(),
1700  cst.getNumSymbolIds());
1701  // Only if the final dividend expression is just a single id (which we call
1702  // `id_n`), we can proceed.
1703  // TODO: Handle AffineSymbolExpr as well. There is no reason to restrict it
1704  // to dims themselves.
1705  auto dimExpr = dividendExpr.dyn_cast<AffineDimExpr>();
1706  if (!dimExpr)
1707  continue;
1708 
1709  // Express `id_r` as `id_n % divisor` and store the expression in `memo`.
1710  if (quotientCount >= 1) {
1711  auto ub = cst.getConstantBound(FlatAffineConstraints::BoundType::UB,
1712  dimExpr.getPosition());
1713  // If `id_n` has an upperbound that is less than the divisor, mod can be
1714  // eliminated altogether.
1715  if (ub.hasValue() && ub.getValue() < divisor)
1716  memo[pos] = dimExpr;
1717  else
1718  memo[pos] = dimExpr % divisor;
1719  // If a unique quotient `id_q` was seen, it can be expressed as
1720  // `id_n floordiv divisor`.
1721  if (quotientCount == 1 && !memo[quotientPosition])
1722  memo[quotientPosition] = dimExpr.floorDiv(divisor) * quotientSign;
1723 
1724  return true;
1725  }
1726  }
1727  return false;
1728 }

3115 /// variable elimination. This technique is exact for rational spaces but
3116 /// conservative (in "rare" cases) for integer spaces. The operation corresponds
3117 /// to a projection operation yielding the (convex) set of integer points
3118 /// contained in the rational shadow of the set. An emptiness test that relies
3119 /// on this method will guarantee emptiness, i.e., it disproves the existence of
3120 /// a solution if it says it's empty.
3121 /// If a non-null isResultIntegerExact is passed, it is set to true if the
3122 /// result is also integer exact. If it's set to false, the obtained solution
3123 /// *may* not be exact, i.e., it may contain integer points that do not have an
3124 /// integer pre-image in the original set.
3125 ///
3126 /// Eg:
3127 /// j >= 0, j <= i + 1
3128 /// i >= 0, i <= N + 1

3130 /// j >= 0, 0 <= N + 1, j - 1 <= N + 1

3132 /// If darkShadow = true, this method computes the dark shadow on elimination;
3133 /// the dark shadow is a convex integer subset of the exact integer shadow. A

3136 /// should not be used for scanning sets or used by itself for dependence
3137 /// checking.

3816 
3817  // Convert `rel` from
3818  // [otherDomain] -> [otherRange]
3819  // to
3820  // [otherDomain] -> [otherRange thisRange]
3821  // and `this` from
3822  // [thisDomain] -> [thisRange]

3830  mergeSymbolIds(rel);

3834  // [otherDomain] -> [thisRange] by converting first otherRange range ids
3835  // to local ids.
3836  rel.convertDimToLocal(rel.getNumDomainDims(),
3837  rel.getNumDomainDims() + removeDims);

3839  // [otherDomain] -> [thisRange] by converting last thisDomain domain ids
3840  // to local ids.
3841  convertDimToLocal(getNumDomainDims() - removeDims, getNumDomainDims());

3844  auto relMaybeValues = rel.getMaybeDimValues();

3847  for (unsigned i = 0, e = rel.getNumDomainDims(); i < e; ++i)
3848  if (relMaybeValues[i].hasValue())

3850  // Add and match range of `this` to range of `rel`.

3854  rel.setValue(rangeIdx, thisMaybeValues[rangeIdx].getValue());

3858  rel.append(*this);
3859  rel.removeRedundantLocalVars();
