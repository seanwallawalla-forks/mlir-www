81 BufferizationAliasInfo::BufferizationAliasInfo(Operation *rootOp) {
82  rootOp->walk([&](Operation *op) {
83  for (Value v : op->getResults())
84  if (v.getType().isa<TensorType>())
85  createAliasInfoEntry(v);
86  for (Region &r : op->getRegions())
87  for (Block &b : r.getBlocks())
88  for (auto bbArg : b.getArguments())
89  if (bbArg.getType().isa<TensorType>())
90  createAliasInfoEntry(bbArg);
91  });
92 
93  // Set up alias sets for OpResults that must bufferize in-place. This should
94  // be done before making any other bufferization decisions.
95  rootOp->walk([&](BufferizableOpInterface bufferizableOp) {
96  for (OpResult opResult : bufferizableOp->getOpResults()) {
97  if (opResult.getType().isa<TensorType>())
98  if (bufferizableOp.mustBufferizeInPlace(opResult)) {
99  SmallVector<OpOperand *> operands =
100  bufferizableOp.getAliasingOpOperand(opResult);
101  assert(!operands.empty() &&
102  "expected that OpResult has aliasing OpOperand");
103  for (OpOperand *operand : operands)
104  aliasInfo.unionSets(operand->get(), opResult);
105  markInPlace(opResult);
106  }
107  }
108  });
109 }

177 BufferizationAliasInfo::getAliases(Value v) const {
178  DenseSet<Value> res;

259 /// Return true if the given value is read by an op that bufferizes to a memory
260 /// read. Also takes into account ops that create an alias but do not read by
261 /// themselves (e.g., ExtractSliceOp).
262 bool mlir::linalg::comprehensive_bufferize::isValueRead(Value value) {
263  SmallVector<OpOperand *> workingSet;
264  for (OpOperand &use : value.getUses())
265  workingSet.push_back(&use);
266 
267  while (!workingSet.empty()) {
268  OpOperand *uMaybeReading = workingSet.pop_back_val();
269  // Skip over all ops that neither read nor write (but create an alias).
270  if (bufferizesToAliasOnly(*uMaybeReading))
271  for (OpOperand &use : getAliasingOpResult(*uMaybeReading).getUses())
272  workingSet.push_back(&use);
273  if (bufferizesToMemoryRead(*uMaybeReading))
274  return true;
275  }
276 
277  return false;
278 }

280 // Starting from `value`, follow the use-def chain in reverse, always selecting
281 // the aliasing OpOperands. Find and return Values for which `condition`
282 // evaluates to true. OpOperands of such matching Values are not traversed any
283 // further.
284 llvm::SetVector<Value>
285 mlir::linalg::comprehensive_bufferize::findValueInReverseUseDefChain(
286  Value value, std::function<bool(Value)> condition) {
287  llvm::SetVector<Value> result, workingSet;
288  workingSet.insert(value);
289 
290  while (!workingSet.empty()) {
291  Value value = workingSet.pop_back_val();
292  if (condition(value) || value.isa<BlockArgument>()) {
293  result.insert(value);
294  continue;
295  }
296 
297  OpResult opResult = value.cast<OpResult>();
298  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(opResult);
299  if (opOperands.empty()) {
300  result.insert(value);
301  continue;
302  }
303 
304  for (OpOperand *o : opOperands)
305  workingSet.insert(o->get());
306  }
307 
308  return result;
309 }

333 /// Return the result buffer (memref) for a given OpResult (tensor). Allocate
334 /// a new buffer and copy over data from the existing buffer if out-of-place
335 /// bufferization is necessary.
336 Value mlir::linalg::comprehensive_bufferize::BufferizationState::
337  getResultBuffer(OpResult result) {
338  OpBuilder::InsertionGuard guard(builder);
339  Operation *op = result.getOwner();
340  SmallVector<OpOperand *> aliasingOperands = getAliasingOpOperand(result);
341  assert(!aliasingOperands.empty() && "could not get aliasing OpOperand");
342  OpOperand *opOperand = aliasingOperands.front();
343  Value operand = opOperand->get();
344  Value operandBuffer = lookupBuffer(operand);
345  // Make sure that all OpOperands are the same buffer. If this is not the case,
346  // we would have to materialize a memref value.
347  // TODO: Should be looking for checking for "equivalent buffers" instead of
348  // operator== here, but equivalent buffers for scf.if yield values are not
349  // set up yet.
350  if (aliasingOperands.size() > 1 &&
351  !llvm::all_of(aliasingOperands, [&](OpOperand *o) {
352  return lookupBuffer(o->get()) == operandBuffer;
353  })) {
354  op->emitError("result buffer is ambiguous");
355  return Value();
356  }
357 
358  // If bufferizing out-of-place, allocate a new buffer.
359  if (!aliasInfo.isInPlace(result)) {
360  // Ops with multiple aliasing operands can currently not bufferize
361  // out-of-place.
362  assert(
363  aliasingOperands.size() == 1 &&
364  "ops with multiple aliasing OpOperands cannot bufferize out-of-place");
365  Location loc = op->getLoc();
366  // Move insertion point right after `operandBuffer`. That is where the
367  // allocation should be inserted (in the absence of allocation hoisting).
368  setInsertionPointAfter(builder, operandBuffer);
369  // Allocate the result buffer.
370  Value resultBuffer = createAllocDeallocFn(builder, loc, operandBuffer);
371  bool skipCopy = false;
372  // Do not copy if the last preceding write of `operand` is an op that does
373  // not write (skipping ops that merely create aliases). E.g., InitTensorOp.
374  // Note: If `findLastPrecedingWrite` reaches the end of the reverse SSA
375  // use-def chain, it returns that value, regardless of whether it is a
376  // memory write or not.
377  Value lastWrite = findLastPrecedingWrite(operand);
378  if (auto bufferizableOp =
379  lastWrite.getDefiningOp<BufferizableOpInterface>())
380  if (!bufferizableOp.isMemoryWrite(lastWrite.cast<OpResult>()))
381  skipCopy = true;
382  // Do not copy if the copied data is never read.
383  if (!isValueRead(result))
384  skipCopy = true;
385  // Do not copy if this op does not read the data, but writes it.
386  if (bufferizesToMemoryWrite(*opOperand) &&
387  !bufferizesToMemoryRead(*opOperand))
388  skipCopy = true;
389  if (!skipCopy) {
390  // The copy happens right before the op that is bufferized.
391  builder.setInsertionPoint(op);
392  options.allocationFns->memCpyFn(builder, loc, operandBuffer,
393  resultBuffer);
394  }
395  return resultBuffer;
396  }
397 
398  // Bufferizing in-place. No need to allocate a new buffer.
399  return operandBuffer;
400 }

420 LogicalResult
421 mlir::linalg::comprehensive_bufferize::bufferize(Operation *op,
422  BufferizationState &state) {
423  OpBuilder &b = state.builder;
424 
425  // Check if op has tensor results or operands.
426  auto isaTensor = [](Type t) { return t.isa<TensorType>(); };
427  bool hasTensorResult = any_of(op->getResultTypes(), isaTensor);
428  bool hasTensorOperand = any_of(op->getOperandTypes(), isaTensor);
429 
430  // No tensor results or operands: Simply bufferize all nested ops.
431  if (!hasTensorResult && !hasTensorOperand) {
432  for (Region &region : op->getRegions())
433  if (failed(bufferize(&region, state)))
434  return failure();
435  return success();
436  }
437 
438  // Bufferize using `BufferizableOpInterface`. Interface implementations are
439  // responsible for bufferizing nested ops.
440  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op)) {
441  b.setInsertionPoint(op);
442  return bufferizableOp.bufferize(b, state);
443  }
444 
445  // `op` is an unbufferizable tensor op.
446  if (!state.options.allowUnknownOps)
447  return op->emitError() << "unsupported op with tensors";
448 
449  // Replace all OpOperands with "to-tensor casted" bufferized values.
450  for (OpOperand &operand : op->getOpOperands()) {
451  if (operand.get().getType().isa<TensorType>() &&
452  state.isMapped(operand.get())) {
453  b.setInsertionPoint(op);
454  Value toTensorOp = b.create<bufferization::ToTensorOp>(
455  op->getLoc(), state.lookupBuffer(operand.get()));
456  operand.set(toTensorOp);
457  }
458  }
459 
460  for (Region &region : op->getRegions())
461  if (failed(bufferize(&region, state)))
462  return failure();
463 
464  return success();
465 }

471 /// Move the insertion point of the given builder to the beginning of a
472 /// surrounding block as much as possible, while not crossing any allocation
473 /// hoisting barriers.
474 static void moveInsertionPointToAllocationHoistingBarrier(OpBuilder &b) {
475  Operation *op = b.getInsertionBlock()->getParentOp();
476  while (op) {
477  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op))
478  if (bufferizableOp.isAllocationHoistingBarrier())
479  break;
480  op = op->getParentOp();
481  }
482 
483  if (!op) {
484  // No allocation hoisting barrier found. Hoist to FuncOp.
485  op = b.getInsertionBlock()->getParentOp();
486  if (!isa<FuncOp>(op))
487  op = op->getParentOfType<FuncOp>();
488  assert(op && "could not find enclosing FuncOp");
489  }
490 
491  // TODO: Handle cases where allocation hoisting barrier has more than one
492  // region or block.
493  assert(op->getNumRegions() == 1 &&
494  "allocation hoisting barriers with >1 regions not supported");
495  assert(op->getRegion(0).getBlocks().size() == 1 &&
496  "allocation hoisting barriers with >1 blocks not supported");
497  b.setInsertionPointToStart(&(op->getRegion(0).front()));
498 }

500 /// Compute the type of the `memref` to use for allocating the buffer for
501 /// `shapedValue`. Also returns (by reference in `dynShape`), the value for the
502 /// dynamic dimensions in the returned `memref` type. The function may also set
503 /// the insertion point to an earlier location, where the allocation should
504 /// happen ("allocation hoisting").
505 static MemRefType getAllocationTypeAndShape(OpBuilder &b, Location loc,
506  Value shapedValue,
507  SmallVectorImpl<Value> &dynShape) {
508  MemRefType allocMemRefType =
509  getContiguousMemRefType(shapedValue.getType().cast<ShapedType>());
510 
511  // Compute the dynamic part of the shape.
512  bool reifiedShapes = false;
513  if (auto rankedOp = dyn_cast_or_null<ReifyRankedShapedTypeOpInterface>(
514  shapedValue.getDefiningOp())) {
515  ReifiedRankedShapedTypeDims resultDims;
516  if (succeeded(rankedOp.reifyResultShapes(b, resultDims))) {
517  reifiedShapes = true;
518  OpResult resultValue = shapedValue.dyn_cast<OpResult>();
519  auto &shape = resultDims[resultValue.getResultNumber()];
520  for (const auto &dim : enumerate(allocMemRefType.getShape()))
521  if (ShapedType::isDynamic(dim.value()))
522  dynShape.push_back(shape[dim.index()]);
523  }
524  }
525 
526  if (!reifiedShapes) {
527  for (const auto &dim : enumerate(allocMemRefType.getShape()))
528  if (ShapedType::isDynamic(dim.value())) {
529  assert((shapedValue.getType().isa<UnrankedMemRefType>() ||
530  shapedValue.getType().isa<MemRefType>()) &&
531  "expected MemRef type");
532  dynShape.push_back(
533  b.create<memref::DimOp>(loc, shapedValue, dim.index()));
534  }
535  }
536 
537  // If the buffer is statically shaped, try to hoist it to the first enclosing
538  // parallel region.
539  // TODO: also hoist in the dynamic case. For now this relies on subsequent
540  // calls to LICM and buffer hoisting which will most likely not succeed.
541  // TODO: when packing, allocate a static bounding box which will enable more
542  // hoisting.
543  if (dynShape.empty())
544  moveInsertionPointToAllocationHoistingBarrier(b);
545 
546  return allocMemRefType;
547 }

549 /// Create an Allocop/DeAllocOp pair, where the AllocOp is after
550 /// `shapedValue.getDefiningOp` (or at the top of the block in case of a
551 /// bbArg) and the DeallocOp is at the end of the block.
552 Value mlir::linalg::comprehensive_bufferize::BufferizationState::
553  createAllocDeallocFn(OpBuilder &b, Location loc, Value shapedValue) {
554  // Take a guard before anything else.
555  OpBuilder::InsertionGuard g(b);
556 
557  // 1. Create memory allocation.
558  assert(shapedValue.getType().isa<ShapedType>());
559  MemRefType memRefType = shapedValue.getType().dyn_cast<MemRefType>();
560  SmallVector<Value> dynShape;
561  // Note: getAllocationTypeAndShape also sets the insertion point.
562  MemRefType allocMemRefType =
563  getAllocationTypeAndShape(b, loc, shapedValue, dynShape);
564  Optional<Value> allocated =
565  options.allocationFns->allocationFn(b, loc, allocMemRefType, dynShape);
566  // TODO: For now just assert the value is returned. Eventually need to
567  // error-propagate.
568  assert(allocated && "allocation failed");
569  Value casted = allocated.getValue();
570  if (memRefType && memRefType != allocMemRefType) {
571  casted = b.create<memref::CastOp>(loc, memRefType, allocated.getValue());
572  }
573 
574  // 2. Create memory deallocation.
575  b.setInsertionPoint(allocated.getValue().getParentBlock()->getTerminator());
576  options.allocationFns->deallocationFn(b, loc, allocated.getValue());
577  return casted;
578 }
