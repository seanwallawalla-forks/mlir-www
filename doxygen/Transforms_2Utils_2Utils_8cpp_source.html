32 // Perform the replacement in `op`.
33 LogicalResult mlir::replaceAllMemRefUsesWith(Value oldMemRef, Value newMemRef,
34  Operation *op,
35  ArrayRef<Value> extraIndices,
36  AffineMap indexRemap,
37  ArrayRef<Value> extraOperands,
38  ArrayRef<Value> symbolOperands,
39  bool allowNonDereferencingOps) {
40  unsigned newMemRefRank = newMemRef.getType().cast<MemRefType>().getRank();
41  (void)newMemRefRank; // unused in opt mode
42  unsigned oldMemRefRank = oldMemRef.getType().cast<MemRefType>().getRank();
43  (void)oldMemRefRank; // unused in opt mode
44  if (indexRemap) {
45  assert(indexRemap.getNumSymbols() == symbolOperands.size() &&
46  "symbolic operand count mismatch");
47  assert(indexRemap.getNumInputs() ==
48  extraOperands.size() + oldMemRefRank + symbolOperands.size());
49  assert(indexRemap.getNumResults() + extraIndices.size() == newMemRefRank);
50  } else {
51  assert(oldMemRefRank + extraIndices.size() == newMemRefRank);
52  }
53 
54  // Assert same elemental type.
55  assert(oldMemRef.getType().cast<MemRefType>().getElementType() ==
56  newMemRef.getType().cast<MemRefType>().getElementType());
57 
58  SmallVector<unsigned, 2> usePositions;
59  for (const auto &opEntry : llvm::enumerate(op->getOperands())) {
60  if (opEntry.value() == oldMemRef)
61  usePositions.push_back(opEntry.index());
62  }
63 
64  // If memref doesn't appear, nothing to do.
65  if (usePositions.empty())
66  return success();
67 
68  if (usePositions.size() > 1) {
69  // TODO: extend it for this case when needed (rare).
70  assert(false && "multiple dereferencing uses in a single op not supported");
71  return failure();
72  }
73 
74  unsigned memRefOperandPos = usePositions.front();
75 
76  OpBuilder builder(op);
77  // The following checks if op is dereferencing memref and performs the access
78  // index rewrites.
79  auto affMapAccInterface = dyn_cast<AffineMapAccessInterface>(op);
80  if (!affMapAccInterface) {
81  if (!allowNonDereferencingOps) {
82  // Failure: memref used in a non-dereferencing context (potentially
83  // escapes); no replacement in these cases unless allowNonDereferencingOps
84  // is set.
85  return failure();
86  }
87  op->setOperand(memRefOperandPos, newMemRef);
88  return success();
89  }
90  // Perform index rewrites for the dereferencing op and then replace the op
91  NamedAttribute oldMapAttrPair =
92  affMapAccInterface.getAffineMapAttrForMemRef(oldMemRef);
93  AffineMap oldMap = oldMapAttrPair.getValue().cast<AffineMapAttr>().getValue();
94  unsigned oldMapNumInputs = oldMap.getNumInputs();
95  SmallVector<Value, 4> oldMapOperands(
96  op->operand_begin() + memRefOperandPos + 1,
97  op->operand_begin() + memRefOperandPos + 1 + oldMapNumInputs);
98 
99  // Apply 'oldMemRefOperands = oldMap(oldMapOperands)'.
100  SmallVector<Value, 4> oldMemRefOperands;
101  SmallVector<Value, 4> affineApplyOps;
102  oldMemRefOperands.reserve(oldMemRefRank);
103  if (oldMap != builder.getMultiDimIdentityMap(oldMap.getNumDims())) {
104  for (auto resultExpr : oldMap.getResults()) {
105  auto singleResMap = AffineMap::get(oldMap.getNumDims(),
106  oldMap.getNumSymbols(), resultExpr);
107  auto afOp = builder.create<AffineApplyOp>(op->getLoc(), singleResMap,
108  oldMapOperands);
109  oldMemRefOperands.push_back(afOp);
110  affineApplyOps.push_back(afOp);
111  }
112  } else {
113  oldMemRefOperands.assign(oldMapOperands.begin(), oldMapOperands.end());
114  }
115 
116  // Construct new indices as a remap of the old ones if a remapping has been
117  // provided. The indices of a memref come right after it, i.e.,
118  // at position memRefOperandPos + 1.
119  SmallVector<Value, 4> remapOperands;
120  remapOperands.reserve(extraOperands.size() + oldMemRefRank +
121  symbolOperands.size());
122  remapOperands.append(extraOperands.begin(), extraOperands.end());
123  remapOperands.append(oldMemRefOperands.begin(), oldMemRefOperands.end());
124  remapOperands.append(symbolOperands.begin(), symbolOperands.end());
125 
126  SmallVector<Value, 4> remapOutputs;
127  remapOutputs.reserve(oldMemRefRank);
128 
129  if (indexRemap &&
130  indexRemap != builder.getMultiDimIdentityMap(indexRemap.getNumDims())) {
131  // Remapped indices.
132  for (auto resultExpr : indexRemap.getResults()) {
133  auto singleResMap = AffineMap::get(
134  indexRemap.getNumDims(), indexRemap.getNumSymbols(), resultExpr);
135  auto afOp = builder.create<AffineApplyOp>(op->getLoc(), singleResMap,
136  remapOperands);
137  remapOutputs.push_back(afOp);
138  affineApplyOps.push_back(afOp);
139  }
140  } else {
141  // No remapping specified.
142  remapOutputs.assign(remapOperands.begin(), remapOperands.end());
143  }
144 
145  SmallVector<Value, 4> newMapOperands;
146  newMapOperands.reserve(newMemRefRank);
147 
148  // Prepend 'extraIndices' in 'newMapOperands'.
149  for (Value extraIndex : extraIndices) {
150  assert(extraIndex.getDefiningOp()->getNumResults() == 1 &&
151  "single result op's expected to generate these indices");
152  assert((isValidDim(extraIndex) || isValidSymbol(extraIndex)) &&
153  "invalid memory op index");
154  newMapOperands.push_back(extraIndex);
155  }
156 
157  // Append 'remapOutputs' to 'newMapOperands'.
158  newMapOperands.append(remapOutputs.begin(), remapOutputs.end());
159 
160  // Create new fully composed AffineMap for new op to be created.
161  assert(newMapOperands.size() == newMemRefRank);
162  auto newMap = builder.getMultiDimIdentityMap(newMemRefRank);
163  // TODO: Avoid creating/deleting temporary AffineApplyOps here.
164  fullyComposeAffineMapAndOperands(&newMap, &newMapOperands);
165  newMap = simplifyAffineMap(newMap);
166  canonicalizeMapAndOperands(&newMap, &newMapOperands);
167  // Remove any affine.apply's that became dead as a result of composition.
168  for (Value value : affineApplyOps)
169  if (value.use_empty())
170  value.getDefiningOp()->erase();
171 
172  OperationState state(op->getLoc(), op->getName());
173  // Construct the new operation using this memref.
174  state.operands.reserve(op->getNumOperands() + extraIndices.size());
175  // Insert the non-memref operands.
176  state.operands.append(op->operand_begin(),
177  op->operand_begin() + memRefOperandPos);
178  // Insert the new memref value.
179  state.operands.push_back(newMemRef);
180 
181  // Insert the new memref map operands.
182  state.operands.append(newMapOperands.begin(), newMapOperands.end());
183 
184  // Insert the remaining operands unmodified.
185  state.operands.append(op->operand_begin() + memRefOperandPos + 1 +
186  oldMapNumInputs,
187  op->operand_end());
188 
189  // Result types don't change. Both memref's are of the same elemental type.
190  state.types.reserve(op->getNumResults());
191  for (auto result : op->getResults())
192  state.types.push_back(result.getType());
193 
194  // Add attribute for 'newMap', other Attributes do not change.
195  auto newMapAttr = AffineMapAttr::get(newMap);
196  for (auto namedAttr : op->getAttrs()) {
197  if (namedAttr.getName() == oldMapAttrPair.getName())
198  state.attributes.push_back({namedAttr.getName(), newMapAttr});
199  else
200  state.attributes.push_back(namedAttr);
201  }
202 
203  // Create the new operation.
204  auto *repOp = builder.createOperation(state);
205  op->replaceAllUsesWith(repOp);
206  op->erase();
207 
208  return success();
209 }

211 LogicalResult mlir::replaceAllMemRefUsesWith(
212  Value oldMemRef, Value newMemRef, ArrayRef<Value> extraIndices,
213  AffineMap indexRemap, ArrayRef<Value> extraOperands,
214  ArrayRef<Value> symbolOperands, Operation *domOpFilter,
215  Operation *postDomOpFilter, bool allowNonDereferencingOps,
216  bool replaceInDeallocOp) {
217  unsigned newMemRefRank = newMemRef.getType().cast<MemRefType>().getRank();
218  (void)newMemRefRank; // unused in opt mode
219  unsigned oldMemRefRank = oldMemRef.getType().cast<MemRefType>().getRank();
220  (void)oldMemRefRank;
221  if (indexRemap) {
222  assert(indexRemap.getNumSymbols() == symbolOperands.size() &&
223  "symbol operand count mismatch");
224  assert(indexRemap.getNumInputs() ==
225  extraOperands.size() + oldMemRefRank + symbolOperands.size());
226  assert(indexRemap.getNumResults() + extraIndices.size() == newMemRefRank);
227  } else {
228  assert(oldMemRefRank + extraIndices.size() == newMemRefRank);
229  }
230 
231  // Assert same elemental type.
232  assert(oldMemRef.getType().cast<MemRefType>().getElementType() ==
233  newMemRef.getType().cast<MemRefType>().getElementType());
234 
235  std::unique_ptr<DominanceInfo> domInfo;
236  std::unique_ptr<PostDominanceInfo> postDomInfo;
237  if (domOpFilter)
238  domInfo =
239  std::make_unique<DominanceInfo>(domOpFilter->getParentOfType<FuncOp>());
240 
241  if (postDomOpFilter)
242  postDomInfo = std::make_unique<PostDominanceInfo>(
243  postDomOpFilter->getParentOfType<FuncOp>());
244 
245  // Walk all uses of old memref; collect ops to perform replacement. We use a
246  // DenseSet since an operation could potentially have multiple uses of a
247  // memref (although rare), and the replacement later is going to erase ops.
248  DenseSet<Operation *> opsToReplace;
249  for (auto *op : oldMemRef.getUsers()) {
250  // Skip this use if it's not dominated by domOpFilter.
251  if (domOpFilter && !domInfo->dominates(domOpFilter, op))
252  continue;
253 
254  // Skip this use if it's not post-dominated by postDomOpFilter.
255  if (postDomOpFilter && !postDomInfo->postDominates(postDomOpFilter, op))
256  continue;
257 
258  // Skip dealloc's - no replacement is necessary, and a memref replacement
259  // at other uses doesn't hurt these dealloc's.
260  if (isa<memref::DeallocOp>(op) && !replaceInDeallocOp)
261  continue;
262 
263  // Check if the memref was used in a non-dereferencing context. It is fine
264  // for the memref to be used in a non-dereferencing way outside of the
265  // region where this replacement is happening.
266  if (!isa<AffineMapAccessInterface>(*op)) {
267  if (!allowNonDereferencingOps) {
268  LLVM_DEBUG(llvm::dbgs()
269  << "Memref replacement failed: non-deferencing memref op: \n"
270  << *op << '\n');
271  return failure();
272  }
273  // Non-dereferencing ops with the MemRefsNormalizable trait are
274  // supported for replacement.
275  if (!op->hasTrait<OpTrait::MemRefsNormalizable>()) {
276  LLVM_DEBUG(llvm::dbgs() << "Memref replacement failed: use without a "
277  "memrefs normalizable trait: \n"
278  << *op << '\n');
279  return failure();
280  }
281  }
282 
283  // We'll first collect and then replace --- since replacement erases the op
284  // that has the use, and that op could be postDomFilter or domFilter itself!
285  opsToReplace.insert(op);
286  }
287 
288  for (auto *op : opsToReplace) {
289  if (failed(replaceAllMemRefUsesWith(
290  oldMemRef, newMemRef, op, extraIndices, indexRemap, extraOperands,
291  symbolOperands, allowNonDereferencingOps)))
292  llvm_unreachable("memref replacement guaranteed to succeed here");
293  }
294 
295  return success();
296 }

298 /// Given an operation, inserts one or more single result affine
299 /// apply operations, results of which are exclusively used by this operation
300 /// operation. The operands of these newly created affine apply ops are
301 /// guaranteed to be loop iterators or terminal symbols of a function.
302 ///
303 /// Before
304 ///
305 /// affine.for %i = 0 to #map(%N)
306 /// %idx = affine.apply (d0) -> (d0 mod 2) (%i)
307 /// "send"(%idx, %A, ...)
308 /// "compute"(%idx)
309 ///
310 /// After
311 ///
312 /// affine.for %i = 0 to #map(%N)
313 /// %idx = affine.apply (d0) -> (d0 mod 2) (%i)
314 /// "send"(%idx, %A, ...)
315 /// %idx_ = affine.apply (d0) -> (d0 mod 2) (%i)
316 /// "compute"(%idx_)
317 ///
318 /// This allows applying different transformations on send and compute (for eg.
319 /// different shifts/delays).
320 ///
321 /// Returns nullptr either if none of opInst's operands were the result of an
322 /// affine.apply and thus there was no affine computation slice to create, or if
323 /// all the affine.apply op's supplying operands to this opInst did not have any
324 /// uses besides this opInst; otherwise returns the list of affine.apply
325 /// operations created in output argument `sliceOps`.
326 void mlir::createAffineComputationSlice(
327  Operation *opInst, SmallVectorImpl<AffineApplyOp> *sliceOps) {
328  // Collect all operands that are results of affine apply ops.
329  SmallVector<Value, 4> subOperands;
330  subOperands.reserve(opInst->getNumOperands());
331  for (auto operand : opInst->getOperands())
332  if (isa_and_nonnull<AffineApplyOp>(operand.getDefiningOp()))
333  subOperands.push_back(operand);
334 
335  // Gather sequence of AffineApplyOps reachable from 'subOperands'.
336  SmallVector<Operation *, 4> affineApplyOps;
337  getReachableAffineApplyOps(subOperands, affineApplyOps);
338  // Skip transforming if there are no affine maps to compose.
339  if (affineApplyOps.empty())
340  return;
341 
342  // Check if all uses of the affine apply op's lie only in this op op, in
343  // which case there would be nothing to do.
344  bool localized = true;
345  for (auto *op : affineApplyOps) {
346  for (auto result : op->getResults()) {
347  for (auto *user : result.getUsers()) {
348  if (user != opInst) {
349  localized = false;
350  break;
351  }
352  }
353  }
354  }
355  if (localized)
356  return;
357 
358  OpBuilder builder(opInst);
359  SmallVector<Value, 4> composedOpOperands(subOperands);
360  auto composedMap = builder.getMultiDimIdentityMap(composedOpOperands.size());
361  fullyComposeAffineMapAndOperands(&composedMap, &composedOpOperands);
362 
363  // Create an affine.apply for each of the map results.
364  sliceOps->reserve(composedMap.getNumResults());
365  for (auto resultExpr : composedMap.getResults()) {
366  auto singleResMap = AffineMap::get(composedMap.getNumDims(),
367  composedMap.getNumSymbols(), resultExpr);
368  sliceOps->push_back(builder.create<AffineApplyOp>(
369  opInst->getLoc(), singleResMap, composedOpOperands));
370  }
371 
372  // Construct the new operands that include the results from the composed
373  // affine apply op above instead of existing ones (subOperands). So, they
374  // differ from opInst's operands only for those operands in 'subOperands', for
375  // which they will be replaced by the corresponding one from 'sliceOps'.
376  SmallVector<Value, 4> newOperands(opInst->getOperands());
377  for (unsigned i = 0, e = newOperands.size(); i < e; i++) {
378  // Replace the subOperands from among the new operands.
379  unsigned j, f;
380  for (j = 0, f = subOperands.size(); j < f; j++) {
381  if (newOperands[i] == subOperands[j])
382  break;
383  }
384  if (j < subOperands.size()) {
385  newOperands[i] = (*sliceOps)[j];
386  }
387  }
388  for (unsigned idx = 0, e = newOperands.size(); idx < e; idx++) {
389  opInst->setOperand(idx, newOperands[idx]);
390  }
391 }

393 /// Enum to set patterns of affine expr in tiled-layout map.
394 /// TileFloorDiv: <dim expr> div <tile size>
395 /// TileMod: <dim expr> mod <tile size>
396 /// TileNone: None of the above
397 /// Example:
398 /// #tiled_2d_128x256 = affine_map<(d0, d1)
399 /// -> (d0 div 128, d1 div 256, d0 mod 128, d1 mod 256)>
400 /// "d0 div 128" and "d1 div 256" ==> TileFloorDiv
401 /// "d0 mod 128" and "d1 mod 256" ==> TileMod
402 enum TileExprPattern { TileFloorDiv, TileMod, TileNone };
403 
404 /// Check if `map` is a tiled layout. In the tiled layout, specific k dimensions
405 /// being floordiv'ed by respective tile sizes appeare in a mod with the same
406 /// tile sizes, and no other expression involves those k dimensions. This
407 /// function stores a vector of tuples (`tileSizePos`) including AffineExpr for
408 /// tile size, positions of corresponding `floordiv` and `mod`. If it is not a
409 /// tiled layout, an empty vector is returned.
410 static LogicalResult getTileSizePos(
411  AffineMap map,
412  SmallVectorImpl<std::tuple<AffineExpr, unsigned, unsigned>> &tileSizePos) {
413  // Create `floordivExprs` which is a vector of tuples including LHS and RHS of
414  // `floordiv` and its position in `map` output.
415  // Example: #tiled_2d_128x256 = affine_map<(d0, d1)
416  // -> (d0 div 128, d1 div 256, d0 mod 128, d1 mod 256)>
417  // In this example, `floordivExprs` includes {d0, 128, 0} and {d1, 256, 1}.
418  SmallVector<std::tuple<AffineExpr, AffineExpr, unsigned>, 4> floordivExprs;
419  unsigned pos = 0;
420  for (AffineExpr expr : map.getResults()) {
421  if (expr.getKind() == AffineExprKind::FloorDiv) {
422  AffineBinaryOpExpr binaryExpr = expr.cast<AffineBinaryOpExpr>();
423  if (binaryExpr.getRHS().isa<AffineConstantExpr>())
424  floordivExprs.emplace_back(
425  std::make_tuple(binaryExpr.getLHS(), binaryExpr.getRHS(), pos));
426  }
427  pos++;
428  }
429  // Not tiled layout if `floordivExprs` is empty.
430  if (floordivExprs.empty()) {
431  tileSizePos = SmallVector<std::tuple<AffineExpr, unsigned, unsigned>>{};
432  return success();
433  }
434 
435  // Check if LHS of `floordiv` is used in LHS of `mod`. If not used, `map` is
436  // not tiled layout.
437  for (std::tuple<AffineExpr, AffineExpr, unsigned> fexpr : floordivExprs) {
438  AffineExpr floordivExprLHS = std::get<0>(fexpr);
439  AffineExpr floordivExprRHS = std::get<1>(fexpr);
440  unsigned floordivPos = std::get<2>(fexpr);
441 
442  // Walk affinexpr of `map` output except `fexpr`, and check if LHS and RHS
443  // of `fexpr` are used in LHS and RHS of `mod`. If LHS of `fexpr` is used
444  // other expr, the map is not tiled layout. Example of non tiled layout:
445  // affine_map<(d0, d1, d2) -> (d0, d1, d2 floordiv 256, d2 floordiv 256)>
446  // affine_map<(d0, d1, d2) -> (d0, d1, d2 floordiv 256, d2 mod 128)>
447  // affine_map<(d0, d1, d2) -> (d0, d1, d2 floordiv 256, d2 mod 256, d2 mod
448  // 256)>
449  bool found = false;
450  pos = 0;
451  for (AffineExpr expr : map.getResults()) {
452  bool notTiled = false;
453  if (pos != floordivPos) {
454  expr.walk([&](AffineExpr e) {
455  if (e == floordivExprLHS) {
456  if (expr.getKind() == AffineExprKind::Mod) {
457  AffineBinaryOpExpr binaryExpr = expr.cast<AffineBinaryOpExpr>();
458  // If LHS and RHS of `mod` are the same with those of floordiv.
459  if (floordivExprLHS == binaryExpr.getLHS() &&
460  floordivExprRHS == binaryExpr.getRHS()) {
461  // Save tile size (RHS of `mod`), and position of `floordiv` and
462  // `mod` if same expr with `mod` is not found yet.
463  if (!found) {
464  tileSizePos.emplace_back(
465  std::make_tuple(binaryExpr.getRHS(), floordivPos, pos));
466  found = true;
467  } else {
468  // Non tiled layout: Have multilpe `mod` with the same LHS.
469  // eg. affine_map<(d0, d1, d2) -> (d0, d1, d2 floordiv 256, d2
470  // mod 256, d2 mod 256)>
471  notTiled = true;
472  }
473  } else {
474  // Non tiled layout: RHS of `mod` is different from `floordiv`.
475  // eg. affine_map<(d0, d1, d2) -> (d0, d1, d2 floordiv 256, d2
476  // mod 128)>
477  notTiled = true;
478  }
479  } else {
480  // Non tiled layout: LHS is the same, but not `mod`.
481  // eg. affine_map<(d0, d1, d2) -> (d0, d1, d2 floordiv 256, d2
482  // floordiv 256)>
483  notTiled = true;
484  }
485  }
486  });
487  }
488  if (notTiled) {
489  tileSizePos = SmallVector<std::tuple<AffineExpr, unsigned, unsigned>>{};
490  return success();
491  }
492  pos++;
493  }
494  }
495  return success();
496 }

498 /// Check if `dim` dimension of memrefType with `layoutMap` becomes dynamic
499 /// after normalization. Dimensions that include dynamic dimensions in the map
500 /// output will become dynamic dimensions. Return true if `dim` is dynamic
501 /// dimension.
502 ///
503 /// Example:
504 /// #map0 = affine_map<(d0, d1) -> (d0, d1 floordiv 32, d1 mod 32)>
505 ///
506 /// If d1 is dynamic dimension, 2nd and 3rd dimension of map output are dynamic.
507 /// memref<4x?xf32, #map0> ==> memref<4x?x?xf32>
508 static bool
509 isNormalizedMemRefDynamicDim(unsigned dim, AffineMap layoutMap,
510  SmallVectorImpl<unsigned> &inMemrefTypeDynDims,
511  MLIRContext *context) {
512  bool isDynamicDim = false;
513  AffineExpr expr = layoutMap.getResults()[dim];
514  // Check if affine expr of the dimension includes dynamic dimension of input
515  // memrefType.
516  expr.walk([&inMemrefTypeDynDims, &isDynamicDim, &context](AffineExpr e) {
517  if (e.isa<AffineDimExpr>()) {
518  for (unsigned dm : inMemrefTypeDynDims) {
519  if (e == getAffineDimExpr(dm, context)) {
520  isDynamicDim = true;
521  }
522  }
523  }
524  });
525  return isDynamicDim;
526 }

528 /// Create affine expr to calculate dimension size for a tiled-layout map.
529 static AffineExpr createDimSizeExprForTiledLayout(AffineExpr oldMapOutput,
530  TileExprPattern pat) {
531  // Create map output for the patterns.
532  // "floordiv <tile size>" ==> "ceildiv <tile size>"
533  // "mod <tile size>" ==> "<tile size>"
534  AffineExpr newMapOutput;
535  AffineBinaryOpExpr binaryExpr = nullptr;
536  switch (pat) {
537  case TileExprPattern::TileMod:
538  binaryExpr = oldMapOutput.cast<AffineBinaryOpExpr>();
539  newMapOutput = binaryExpr.getRHS();
540  break;
541  case TileExprPattern::TileFloorDiv:
542  binaryExpr = oldMapOutput.cast<AffineBinaryOpExpr>();
543  newMapOutput = getAffineBinaryOpExpr(
544  AffineExprKind::CeilDiv, binaryExpr.getLHS(), binaryExpr.getRHS());
545  break;
546  default:
547  newMapOutput = oldMapOutput;
548  }
549  return newMapOutput;
550 }

552 /// Create new maps to calculate each dimension size of `newMemRefType`, and
553 /// create `newDynamicSizes` from them by using AffineApplyOp.
554 ///
555 /// Steps for normalizing dynamic memrefs for a tiled layout map
556 /// Example:
557 /// #map0 = affine_map<(d0, d1) -> (d0, d1 floordiv 32, d1 mod 32)>
558 /// %0 = dim %arg0, %c1 :memref<4x?xf32>
559 /// %1 = alloc(%0) : memref<4x?xf32, #map0>
560 ///
561 /// (Before this function)
562 /// 1. Check if `map`(#map0) is a tiled layout using `getTileSizePos()`. Only
563 /// single layout map is supported.
564 ///
565 /// 2. Create normalized memrefType using `isNormalizedMemRefDynamicDim()`. It
566 /// is memref<4x?x?xf32> in the above example.
567 ///
568 /// (In this function)
569 /// 3. Create new maps to calculate each dimension of the normalized memrefType
570 /// using `createDimSizeExprForTiledLayout()`. In the tiled layout, the
571 /// dimension size can be calculated by replacing "floordiv <tile size>" with
572 /// "ceildiv <tile size>" and "mod <tile size>" with "<tile size>".
573 /// - New map in the above example
574 /// #map0 = affine_map<(d0, d1) -> (d0)>
575 /// #map1 = affine_map<(d0, d1) -> (d1 ceildiv 32)>
576 /// #map2 = affine_map<(d0, d1) -> (32)>
577 ///
578 /// 4. Create AffineApplyOp to apply the new maps. The output of AffineApplyOp
579 /// is used in dynamicSizes of new AllocOp.
580 /// %0 = dim %arg0, %c1 : memref<4x?xf32>
581 /// %c4 = arith.constant 4 : index
582 /// %1 = affine.apply #map1(%c4, %0)
583 /// %2 = affine.apply #map2(%c4, %0)
584 static void createNewDynamicSizes(MemRefType oldMemRefType,
585  MemRefType newMemRefType, AffineMap map,
586  memref::AllocOp *allocOp, OpBuilder b,
587  SmallVectorImpl<Value> &newDynamicSizes) {
588  // Create new input for AffineApplyOp.
589  SmallVector<Value, 4> inAffineApply;
590  ArrayRef<int64_t> oldMemRefShape = oldMemRefType.getShape();
591  unsigned dynIdx = 0;
592  for (unsigned d = 0; d < oldMemRefType.getRank(); ++d) {
593  if (oldMemRefShape[d] < 0) {
594  // Use dynamicSizes of allocOp for dynamic dimension.
595  inAffineApply.emplace_back(allocOp->dynamicSizes()[dynIdx]);
596  dynIdx++;
597  } else {
598  // Create ConstantOp for static dimension.
599  Attribute constantAttr =
600  b.getIntegerAttr(b.getIndexType(), oldMemRefShape[d]);
601  inAffineApply.emplace_back(
602  b.create<arith::ConstantOp>(allocOp->getLoc(), constantAttr));
603  }
604  }
605 
606  // Create new map to calculate each dimension size of new memref for each
607  // original map output. Only for dynamic dimesion of `newMemRefType`.
608  unsigned newDimIdx = 0;
609  ArrayRef<int64_t> newMemRefShape = newMemRefType.getShape();
610  SmallVector<std::tuple<AffineExpr, unsigned, unsigned>> tileSizePos;
611  (void)getTileSizePos(map, tileSizePos);
612  for (AffineExpr expr : map.getResults()) {
613  if (newMemRefShape[newDimIdx] < 0) {
614  // Create new maps to calculate each dimension size of new memref.
615  enum TileExprPattern pat = TileExprPattern::TileNone;
616  for (auto pos : tileSizePos) {
617  if (newDimIdx == std::get<1>(pos))
618  pat = TileExprPattern::TileFloorDiv;
619  else if (newDimIdx == std::get<2>(pos))
620  pat = TileExprPattern::TileMod;
621  }
622  AffineExpr newMapOutput = createDimSizeExprForTiledLayout(expr, pat);
623  AffineMap newMap =
624  AffineMap::get(map.getNumInputs(), map.getNumSymbols(), newMapOutput);
625  Value affineApp =
626  b.create<AffineApplyOp>(allocOp->getLoc(), newMap, inAffineApply);
627  newDynamicSizes.emplace_back(affineApp);
628  }
629  newDimIdx++;
630  }
631 }

633 // TODO: Currently works for static memrefs with a single layout map.
634 LogicalResult mlir::normalizeMemRef(memref::AllocOp *allocOp) {
635  MemRefType memrefType = allocOp->getType();
636  OpBuilder b(*allocOp);
637 
638  // Fetch a new memref type after normalizing the old memref to have an
639  // identity map layout.
640  MemRefType newMemRefType =
641  normalizeMemRefType(memrefType, b, allocOp->symbolOperands().size());
642  if (newMemRefType == memrefType)
643  // Either memrefType already had an identity map or the map couldn't be
644  // transformed to an identity map.
645  return failure();
646 
647  Value oldMemRef = allocOp->getResult();
648 
649  SmallVector<Value, 4> symbolOperands(allocOp->symbolOperands());
650  AffineMap layoutMap = memrefType.getLayout().getAffineMap();
651  memref::AllocOp newAlloc;
652  // Check if `layoutMap` is a tiled layout. Only single layout map is
653  // supported for normalizing dynamic memrefs.
654  SmallVector<std::tuple<AffineExpr, unsigned, unsigned>> tileSizePos;
655  (void)getTileSizePos(layoutMap, tileSizePos);
656  if (newMemRefType.getNumDynamicDims() > 0 && !tileSizePos.empty()) {
657  MemRefType oldMemRefType = oldMemRef.getType().cast<MemRefType>();
658  SmallVector<Value, 4> newDynamicSizes;
659  createNewDynamicSizes(oldMemRefType, newMemRefType, layoutMap, allocOp, b,
660  newDynamicSizes);
661  // Add the new dynamic sizes in new AllocOp.
662  newAlloc =
663  b.create<memref::AllocOp>(allocOp->getLoc(), newMemRefType,
664  newDynamicSizes, allocOp->alignmentAttr());
665  } else {
666  newAlloc = b.create<memref::AllocOp>(allocOp->getLoc(), newMemRefType,
667  allocOp->alignmentAttr());
668  }
669  // Replace all uses of the old memref.
670  if (failed(replaceAllMemRefUsesWith(oldMemRef, /*newMemRef=*/newAlloc,
671  /*extraIndices=*/{},
672  /*indexRemap=*/layoutMap,
673  /*extraOperands=*/{},
674  /*symbolOperands=*/symbolOperands,
675  /*domOpFilter=*/nullptr,
676  /*postDomOpFilter=*/nullptr,
677  /*allowNonDereferencingOps=*/true))) {
678  // If it failed (due to escapes for example), bail out.
679  newAlloc.erase();
680  return failure();
681  }
682  // Replace any uses of the original alloc op and erase it. All remaining uses
683  // have to be dealloc's; RAMUW above would've failed otherwise.
684  assert(llvm::all_of(oldMemRef.getUsers(), [](Operation *op) {
685  return isa<memref::DeallocOp>(op);
686  }));
687  oldMemRef.replaceAllUsesWith(newAlloc);
688  allocOp->erase();
689  return success();
690 }

692 MemRefType mlir::normalizeMemRefType(MemRefType memrefType, OpBuilder b,
693  unsigned numSymbolicOperands) {
694  unsigned rank = memrefType.getRank();
695  if (rank == 0)
696  return memrefType;
697 
698  if (memrefType.getLayout().isIdentity()) {
699  // Either no maps is associated with this memref or this memref has
700  // a trivial (identity) map.
701  return memrefType;
702  }
703  AffineMap layoutMap = memrefType.getLayout().getAffineMap();
704 
705  // We don't do any checks for one-to-one'ness; we assume that it is
706  // one-to-one.
707 
708  // Normalize only static memrefs and dynamic memrefs with a tiled-layout map
709  // for now.
710  // TODO: Normalize the other types of dynamic memrefs.
711  SmallVector<std::tuple<AffineExpr, unsigned, unsigned>> tileSizePos;
712  (void)getTileSizePos(layoutMap, tileSizePos);
713  if (memrefType.getNumDynamicDims() > 0 && tileSizePos.empty())
714  return memrefType;
715 
716  // We have a single map that is not an identity map. Create a new memref
717  // with the right shape and an identity layout map.
718  ArrayRef<int64_t> shape = memrefType.getShape();
719  // FlatAffineConstraint may later on use symbolicOperands.
720  FlatAffineConstraints fac(rank, numSymbolicOperands);
721  SmallVector<unsigned, 4> memrefTypeDynDims;
722  for (unsigned d = 0; d < rank; ++d) {
723  // Use constraint system only in static dimensions.
724  if (shape[d] > 0) {
725  fac.addBound(FlatAffineConstraints::LB, d, 0);
726  fac.addBound(FlatAffineConstraints::UB, d, shape[d] - 1);
727  } else {
728  memrefTypeDynDims.emplace_back(d);
729  }
730  }
731  // We compose this map with the original index (logical) space to derive
732  // the upper bounds for the new index space.
733  unsigned newRank = layoutMap.getNumResults();
734  if (failed(fac.composeMatchingMap(layoutMap)))
735  return memrefType;
736  // TODO: Handle semi-affine maps.
737  // Project out the old data dimensions.
738  fac.projectOut(newRank, fac.getNumIds() - newRank - fac.getNumLocalIds());
739  SmallVector<int64_t, 4> newShape(newRank);
740  for (unsigned d = 0; d < newRank; ++d) {
741  // Check if each dimension of normalized memrefType is dynamic.
742  bool isDynDim = isNormalizedMemRefDynamicDim(
743  d, layoutMap, memrefTypeDynDims, b.getContext());
744  if (isDynDim) {
745  newShape[d] = -1;
746  } else {
747  // The lower bound for the shape is always zero.
748  auto ubConst = fac.getConstantBound(FlatAffineConstraints::UB, d);
749  // For a static memref and an affine map with no symbols, this is
750  // always bounded.
751  assert(ubConst.hasValue() && "should always have an upper bound");
752  if (ubConst.getValue() < 0)
753  // This is due to an invalid map that maps to a negative space.
754  return memrefType;
755  // If dimension of new memrefType is dynamic, the value is -1.
756  newShape[d] = ubConst.getValue() + 1;
757  }
758  }
759 
760  // Create the new memref type after trivializing the old layout map.
761  MemRefType newMemRefType =
762  MemRefType::Builder(memrefType)
763  .setShape(newShape)
764  .setLayout(AffineMapAttr::get(b.getMultiDimIdentityMap(newRank)));
765 
766  return newMemRefType;
767 }
