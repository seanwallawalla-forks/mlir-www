70  /// Handle the given inlined terminator by replacing it with a new operation
71  /// as necessary.
72  void handleTerminator(Operation *op, Block *newDest) const final {
73  // Only "std.return" needs to be handled here.
74  auto returnOp = dyn_cast<ReturnOp>(op);
75  if (!returnOp)
76  return;
77 
78  // Replace the return with a branch to the dest.
79  OpBuilder builder(op);
80  builder.create<BranchOp>(op->getLoc(), newDest, returnOp.getOperands());
81  op->erase();

89  auto returnOp = cast<ReturnOp>(op);

337 /// Given a successor, try to collapse it to a new destination if it only
338 /// contains a passthrough unconditional branch. If the successor is
339 /// collapsable, `successor` and `successorOperands` are updated to reference
340 /// the new destination and values. `argStorage` is used as storage if operands
341 /// to the collapsed successor need to be remapped. It must outlive uses of
342 /// successorOperands.
343 static LogicalResult collapseBranch(Block *&successor,
344  ValueRange &successorOperands,
345  SmallVectorImpl<Value> &argStorage) {
346  // Check that the successor only contains a unconditional branch.
347  if (std::next(successor->begin()) != successor->end())
348  return failure();
349  // Check that the terminator is an unconditional branch.
350  BranchOp successorBranch = dyn_cast<BranchOp>(successor->getTerminator());
351  if (!successorBranch)
352  return failure();
353  // Check that the arguments are only used within the terminator.
354  for (BlockArgument arg : successor->getArguments()) {
355  for (Operation *user : arg.getUsers())
356  if (user != successorBranch)
357  return failure();
358  }
359  // Don't try to collapse branches to infinite loops.
360  Block *successorDest = successorBranch.getDest();
361  if (successorDest == successor)
362  return failure();
363 
364  // Update the operands to the successor. If the branch parent has no
365  // arguments, we can use the branch operands directly.
366  OperandRange operands = successorBranch.getOperands();
367  if (successor->args_empty()) {
368  successor = successorDest;
369  successorOperands = operands;
370  return success();
371  }
372 
373  // Otherwise, we need to remap any argument operands.
374  for (Value operand : operands) {
375  BlockArgument argOperand = operand.dyn_cast<BlockArgument>();
376  if (argOperand && argOperand.getOwner() == successor)
377  argStorage.push_back(successorOperands[argOperand.getArgNumber()]);
378  else
379  argStorage.push_back(operand);
380  }
381  successor = successorDest;
382  successorOperands = argStorage;
383  return success();
384 }

386 /// Simplify a branch to a block that has a single predecessor. This effectively
387 /// merges the two blocks.
388 static LogicalResult
389 simplifyBrToBlockWithSinglePred(BranchOp op, PatternRewriter &rewriter) {
390  // Check that the successor block has a single predecessor.
391  Block *succ = op.getDest();
392  Block *opParent = op->getBlock();
393  if (succ == opParent || !llvm::hasSingleElement(succ->getPredecessors()))
394  return failure();
395 
396  // Merge the successor into the current block and erase the branch.
397  rewriter.mergeBlocks(succ, opParent, op.getOperands());
398  rewriter.eraseOp(op);
399  return success();
400 }

402 /// br ^bb1
403 /// ^bb1
404 /// br ^bbN(...)
405 ///
406 /// -> br ^bbN(...)
407 ///
408 static LogicalResult simplifyPassThroughBr(BranchOp op,
409  PatternRewriter &rewriter) {
410  Block *dest = op.getDest();
411  ValueRange destOperands = op.getOperands();
412  SmallVector<Value, 4> destOperandStorage;
413 
414  // Try to collapse the successor if it points somewhere other than this
415  // block.
416  if (dest == op->getBlock() ||
417  failed(collapseBranch(dest, destOperands, destOperandStorage)))
418  return failure();
419 
420  // Create a new branch with the collapsed successor.
421  rewriter.replaceOpWithNewOp<BranchOp>(op, dest, destOperands);
422  return success();
423 }

448 LogicalResult CallOp::verifySymbolUses(SymbolTableCollection &symbolTable) {
449  // Check that the callee attribute was specified.
450  auto fnAttr = (*this)->getAttrOfType<FlatSymbolRefAttr>("callee");
451  if (!fnAttr)
452  return emitOpError("requires a 'callee' symbol reference attribute");
453  FuncOp fn = symbolTable.lookupNearestSymbolFrom<FuncOp>(*this, fnAttr);
454  if (!fn)
455  return emitOpError() << "'" << fnAttr.getValue()
456  << "' does not reference a valid function";
457 
458  // Verify that the operand and result types match the callee.
459  auto fnType = fn.getType();
460  if (fnType.getNumInputs() != getNumOperands())
461  return emitOpError("incorrect number of operands for callee");
462 
463  for (unsigned i = 0, e = fnType.getNumInputs(); i != e; ++i)
464  if (getOperand(i).getType() != fnType.getInput(i))
465  return emitOpError("operand type mismatch: expected operand type ")
466  << fnType.getInput(i) << ", but provided "
467  << getOperand(i).getType() << " for operand number " << i;
468 
469  if (fnType.getNumResults() != getNumResults())
470  return emitOpError("incorrect number of results for callee");
471 
472  for (unsigned i = 0, e = fnType.getNumResults(); i != e; ++i)
473  if (getResult(i).getType() != fnType.getResult(i)) {
474  auto diag = emitOpError("result type mismatch at index ") << i;
475  diag.attachNote() << " op result types: " << getResultTypes();
476  diag.attachNote() << "function result types: " << fnType.getResults();

586 /// cond_br %cond, ^bb1(A, ..., N), ^bb1(A, ..., N)
587 /// -> br ^bb1(A, ..., N)
588 ///
589 /// cond_br %cond, ^bb1(A), ^bb1(B)
590 /// -> %select = select %cond, A, B
591 /// br ^bb1(%select)
592 ///
593 struct SimplifyCondBranchIdenticalSuccessors
594  : public OpRewritePattern<CondBranchOp> {
595  using OpRewritePattern<CondBranchOp>::OpRewritePattern;
596 
597  LogicalResult matchAndRewrite(CondBranchOp condbr,
598  PatternRewriter &rewriter) const override {
599  // Check that the true and false destinations are the same and have the same
600  // operands.
601  Block *trueDest = condbr.getTrueDest();
602  if (trueDest != condbr.getFalseDest())
603  return failure();
604 
605  // If all of the operands match, no selects need to be generated.
606  OperandRange trueOperands = condbr.getTrueOperands();
607  OperandRange falseOperands = condbr.getFalseOperands();
608  if (trueOperands == falseOperands) {
609  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, trueOperands);
610  return success();
611  }
612 
613  // Otherwise, if the current block is the only predecessor insert selects
614  // for any mismatched branch operands.
615  if (trueDest->getUniquePredecessor() != condbr->getBlock())
616  return failure();
617 
618  // Generate a select for any operands that differ between the two.
619  SmallVector<Value, 8> mergedOperands;
620  mergedOperands.reserve(trueOperands.size());
621  Value condition = condbr.getCondition();
622  for (auto it : llvm::zip(trueOperands, falseOperands)) {
623  if (std::get<0>(it) == std::get<1>(it))
624  mergedOperands.push_back(std::get<0>(it));
625  else
626  mergedOperands.push_back(rewriter.create<SelectOp>(
627  condbr.getLoc(), condition, std::get<0>(it), std::get<1>(it)));

711  Value constantTrue = nullptr;
712  Value constantFalse = nullptr;

738  replaced = true;

742  condbr.getLoc(), ty, rewriter.getBoolAttr(false));
743 
744  rewriter.updateRootInPlace(use.getOwner(),

977 OpFoldResult SelectOp::fold(ArrayRef<Attribute> operands) {
978  auto trueVal = getTrueValue();
979  auto falseVal = getFalseValue();
980  if (trueVal == falseVal)
981  return trueVal;

984

997  auto cmpRhs = cmp.getRhs();

1000  // %1 = select %0, %arg0, %arg1 => %arg1
1001 
1002  // %0 = arith.cmpi ne, %arg0, %arg1
1003  // %1 = select %0, %arg0, %arg1 => %arg0
1004 
1005  if ((cmpLhs == trueVal && cmpRhs == falseVal) ||
1006  (cmpRhs == trueVal && cmpLhs == falseVal))
1007  return pred == arith::CmpIPredicate::ne ? trueVal : falseVal;
