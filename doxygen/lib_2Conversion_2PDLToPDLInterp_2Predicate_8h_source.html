9 // This file contains definitions for "predicates" used when converting PDL into
10 // a matcher tree. Predicates are composed of three different parts:
11 //
12 // * Positions
13 // - A position refers to a specific location on the input DAG, i.e. an
14 // existing MLIR entity being matched. These can be attributes, operands,
15 // operations, results, and types. Each position also defines a relation to
16 // its parent. For example, the operand `[0] -> 1` has a parent operation
17 // position `[0]`. The attribute `[0, 1] -> "myAttr"` has parent operation
18 // position of `[0, 1]`. The operation `[0, 1]` has a parent operand edge
19 // `[0] -> 1` (i.e. it is the defining op of operand 1). The only position
20 // without a parent is `[0]`, which refers to the root operation.
21 // * Questions
22 // - A question refers to a query on a specific positional value. For
23 // example, an operation name question checks the name of an operation
24 // position.
25 // * Answers
26 // - An answer is the expected result of a question. For example, when
27 // matching an operation with the name "foo.op". The question would be an
28 // operation name question, with an expected answer of "foo.op".
29 //
30 //===----------------------------------------------------------------------===//

76 /// Base class for all predicates, used to allow efficient pointer comparison.
77 template <typename ConcreteT, typename BaseT, typename Key,
78  Predicates::Kind Kind>
79 class PredicateBase : public BaseT {
80 public:
81  using KeyTy = Key;
82  using Base = PredicateBase<ConcreteT, BaseT, Key, Kind>;
83 
84  template <typename KeyT>
85  explicit PredicateBase(KeyT &&key)
86  : BaseT(Kind), key(std::forward<KeyT>(key)) {}
87 
88  /// Get an instance of this position.
89  template <typename... Args>
90  static ConcreteT *get(StorageUniquer &uniquer, Args &&...args) {
91  return uniquer.get<ConcreteT>(/*initFn=*/{}, std::forward<Args>(args)...);
92  }
93 
94  /// Construct an instance with the given storage allocator.
95  template <typename KeyT>
96  static ConcreteT *construct(StorageUniquer::StorageAllocator &alloc,
97  KeyT &&key) {
98  return new (alloc.allocate<ConcreteT>()) ConcreteT(std::forward<KeyT>(key));
99  }
100 
101  /// Utility methods required by the storage allocator.
102  bool operator==(const KeyTy &key) const { return this->key == key; }
103  static bool classof(const BaseT *pred) { return pred->getKind() == Kind; }
104 
105  /// Return the key value of this predicate.
106  const KeyTy &getValue() const { return key; }
107 
108 protected:
109  KeyTy key;
110 };

130 struct OperationPosition;
131 
132 /// A position describes a value on the input IR on which a predicate may be
133 /// applied, such as an operation or attribute. This enables re-use between
134 /// predicates, and assists generating bytecode and memory management.
135 ///
136 /// Operation positions form the base of other positions, which are formed
137 /// relative to a parent operation. Operations are anchored at Operand nodes,
138 /// except for the root operation which is parentless.
139 class Position : public StorageUniquer::BaseStorage {
140 public:
141  explicit Position(Predicates::Kind kind) : kind(kind) {}
142  virtual ~Position();
143 
144  /// Returns the depth of the first ancestor operation position.
145  unsigned getOperationDepth() const;
146 
147  /// Returns the parent position. The root operation position has no parent.
148  Position *getParent() const { return parent; }
149 
150  /// Returns the kind of this position.
151  Predicates::Kind getKind() const { return kind; }
152 
153 protected:
154  /// Link to the parent position.
155  Position *parent = nullptr;
156 
157 private:
158  /// The kind of this position.
159  Predicates::Kind kind;
160 };

218 /// An operation position describes an operation node in the IR. Other position
219 /// kinds are formed with respect to an operation position.
220 struct OperationPosition
221  : public PredicateBase<OperationPosition, Position,
222  std::tuple<Position *, Optional<unsigned>, unsigned>,
223  Predicates::OperationPos> {
224  static constexpr unsigned kDown = std::numeric_limits<unsigned>::max();
225 
226  explicit OperationPosition(const KeyTy &key) : Base(key) {
227  parent = std::get<0>(key);
228  }
229 
230  /// Returns a hash suitable for the given keytype.
231  static llvm::hash_code hashKey(const KeyTy &key) {
232  return llvm::hash_value(key);
233  }
234 
235  /// Gets the root position.
236  static OperationPosition *getRoot(StorageUniquer &uniquer) {
237  return Base::get(uniquer, nullptr, kDown, 0);
238  }
239 
240  /// Gets an downward operation position with the given parent.
241  static OperationPosition *get(StorageUniquer &uniquer, Position *parent) {
242  return Base::get(uniquer, parent, kDown, parent->getOperationDepth() + 1);
243  }
244 
245  /// Gets an upward operation position with the given parent and operand.
246  static OperationPosition *get(StorageUniquer &uniquer, Position *parent,
247  Optional<unsigned> operand) {
248  return Base::get(uniquer, parent, operand, parent->getOperationDepth() + 1);
249  }
250 
251  /// Returns the operand index for an upward operation position.
252  Optional<unsigned> getIndex() const { return std::get<1>(key); }
253 
254  /// Returns if this operation position is upward, accepting an input.
255  bool isUpward() const { return getIndex().getValueOr(0) != kDown; }
256 
257  /// Returns the depth of this position.
258  unsigned getDepth() const { return std::get<2>(key); }
259 
260  /// Returns if this operation position corresponds to the root.
261  bool isRoot() const { return getDepth() == 0; }

324 /// An ordinal predicate consists of a "Question" and a set of acceptable
325 /// "Answers" (later converted to ordinal values). A predicate will query some
326 /// property of a positional value and decide what to do based on the result.
327 ///
328 /// This makes top-level predicate representations ordinal (SwitchOp). Later,
329 /// predicates that end up with only one acceptable answer (including all
330 /// boolean kinds) will be converted to boolean predicates (PredicateOp) in the
331 /// matcher.
332 ///
333 /// For simplicity, both are represented as "qualifiers", with a base kind and
334 /// perhaps additional properties. For example, all OperationName predicates ask
335 /// the same question, but GenericConstraint predicates may ask different ones.
336 class Qualifier : public StorageUniquer::BaseStorage {
337 public:
338  explicit Qualifier(Predicates::Kind kind) : kind(kind) {}
339 
340  /// Returns the kind of this qualifier.
341  Predicates::Kind getKind() const { return kind; }
342 
343 private:
344  /// The kind of this position.
345  Predicates::Kind kind;
346 };
