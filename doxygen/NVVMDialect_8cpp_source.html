46 // <operation> ::= `llvm.nvvm.vote.ballot.sync %mask, %pred` : result_type
47 static ParseResult parseNVVMVoteBallotOp(OpAsmParser &parser,
48  OperationState &result) {
49  MLIRContext *context = parser.getContext();
50  auto int32Ty = IntegerType::get(context, 32);
51  auto int1Ty = IntegerType::get(context, 1);
52 
53  SmallVector<OpAsmParser::OperandType, 8> ops;
54  Type type;
55  return failure(parser.parseOperandList(ops) ||
56  parser.parseOptionalAttrDict(result.attributes) ||
57  parser.parseColonType(type) ||
58  parser.addTypeToList(type, result.types) ||
59  parser.resolveOperands(ops, {int32Ty, int1Ty},
60  parser.getNameLoc(), result.operands));
61 }

63 static LogicalResult verify(MmaOp op) {
64  MLIRContext *context = op.getContext();
65  auto f16Ty = Float16Type::get(context);
66  auto f16x2Ty = LLVM::getFixedVectorType(f16Ty, 2);
67  auto f32Ty = Float32Type::get(context);
68  auto f16x2x4StructTy = LLVM::LLVMStructType::getLiteral(
69  context, {f16x2Ty, f16x2Ty, f16x2Ty, f16x2Ty});
70  auto f32x8StructTy = LLVM::LLVMStructType::getLiteral(
71  context, {f32Ty, f32Ty, f32Ty, f32Ty, f32Ty, f32Ty, f32Ty, f32Ty});
72 
73  SmallVector<Type, 12> operandTypes(op.getOperandTypes().begin(),
74  op.getOperandTypes().end());
75  if (operandTypes != SmallVector<Type, 8>(8, f16x2Ty) &&
76  operandTypes != SmallVector<Type, 12>{f16x2Ty, f16x2Ty, f16x2Ty, f16x2Ty,
77  f32Ty, f32Ty, f32Ty, f32Ty, f32Ty,
78  f32Ty, f32Ty, f32Ty}) {
79  return op.emitOpError(
80  "expected operands to be 4 <halfx2>s followed by either "
81  "4 <halfx2>s or 8 floats");
82  }
83  if (op.getType() != f32x8StructTy && op.getType() != f16x2x4StructTy) {
84  return op.emitOpError("expected result type to be a struct of either 4 "
85  "<halfx2>s or 8 floats");
86  }
87 
88  auto alayout = op->getAttrOfType<StringAttr>("alayout");
89  auto blayout = op->getAttrOfType<StringAttr>("blayout");
90 
91  if (!(alayout && blayout) ||
92  !(alayout.getValue() == "row" || alayout.getValue() == "col") ||
93  !(blayout.getValue() == "row" || blayout.getValue() == "col")) {
94  return op.emitOpError(
95  "alayout and blayout attributes must be set to either "
96  "\"row\" or \"col\"");
97  }
98 
99  if (operandTypes == SmallVector<Type, 12>{f16x2Ty, f16x2Ty, f16x2Ty, f16x2Ty,
100  f32Ty, f32Ty, f32Ty, f32Ty, f32Ty,
101  f32Ty, f32Ty, f32Ty} &&
102  op.getType() == f32x8StructTy && alayout.getValue() == "row" &&
103  blayout.getValue() == "col") {
104  return success();
105  }
106  return op.emitOpError("unimplemented mma.sync variant");
107 }

109 std::pair<mlir::Type, unsigned>
110 inferMMAType(NVVM::MMATypes type, NVVM::MMAFrag frag, MLIRContext *context) {
111  unsigned numberElements = 0;
112  Type elementType;
113  OpBuilder builder(context);
114  Type f16x2 = VectorType::get(2, builder.getF16Type());
115  if (type == NVVM::MMATypes::f16) {
116  elementType = f16x2;
117  if (frag == NVVM::MMAFrag::a || frag == NVVM::MMAFrag::b)
118  numberElements = 8;
119  else
120  numberElements = 4;
121  } else if (type == NVVM::MMATypes::f32) {
122  elementType = builder.getF32Type();
123  numberElements = 8;
124  } else if (type == NVVM::MMATypes::tf32) {
125  elementType = builder.getI32Type();
126  numberElements = 4;
127  }
128  assert(numberElements != 0 && elementType != nullptr);
129  return std::make_pair(elementType, numberElements);
130 }

132 static LogicalResult verify(NVVM::WMMALoadOp op) {
133  unsigned addressSpace =
134  op.ptr().getType().cast<LLVM::LLVMPointerType>().getAddressSpace();
135  if (addressSpace != 0 && addressSpace != 1 && addressSpace != 3)
136  return op.emitOpError("expected source pointer in memory "
137  "space 0, 1, 3");
138 
139  if (NVVM::WMMALoadOp::getIntrinsicID(op.m(), op.n(), op.k(), op.layout(),
140  op.eltype(), op.frag()) == 0)
141  return op.emitOpError() << "invalid attribute combination";
142  std::pair<Type, unsigned> typeInfo =
143  inferMMAType(op.eltype(), op.frag(), op.getContext());
144  Type dstType = LLVM::LLVMStructType::getLiteral(
145  op.getContext(), SmallVector<Type, 8>(typeInfo.second, typeInfo.first));
146  if (op.getType() != dstType)
147  return op.emitOpError("expected destination type is a structure of ")
148  << typeInfo.second << " elements of type " << typeInfo.first;
149  return success();
150 }

152 static LogicalResult verify(NVVM::WMMAStoreOp op) {
153  unsigned addressSpace =
154  op.ptr().getType().cast<LLVM::LLVMPointerType>().getAddressSpace();
155  if (addressSpace != 0 && addressSpace != 1 && addressSpace != 3)
156  return op.emitOpError("expected operands to be a source pointer in memory "
157  "space 0, 1, 3");
158 
159  if (NVVM::WMMAStoreOp::getIntrinsicID(op.m(), op.n(), op.k(), op.layout(),
160  op.eltype()) == 0)
161  return op.emitOpError() << "invalid attribute combination";
162  std::pair<Type, unsigned> typeInfo =
163  inferMMAType(op.eltype(), NVVM::MMAFrag::c, op.getContext());
164  if (op.args().size() != typeInfo.second)
165  return op.emitOpError()
166  << "expected " << typeInfo.second << " data operands";
167  if (llvm::any_of(op.args(), [&typeInfo](Value operands) {
168  return operands.getType() != typeInfo.first;
169  }))
170  return op.emitOpError()
171  << "expected data operands of type " << typeInfo.first;
172  return success();
173 }

175 static LogicalResult verify(NVVM::WMMAMmaOp op) {
176  if (NVVM::WMMAMmaOp::getIntrinsicID(op.m(), op.n(), op.k(), op.layoutA(),
177  op.layoutB(), op.eltypeA(),
178  op.eltypeB()) == 0)
179  return op.emitOpError() << "invalid attribute combination";
180  std::pair<Type, unsigned> typeInfoA =
181  inferMMAType(op.eltypeA(), NVVM::MMAFrag::a, op.getContext());
182  std::pair<Type, unsigned> typeInfoB =
183  inferMMAType(op.eltypeA(), NVVM::MMAFrag::b, op.getContext());
184  std::pair<Type, unsigned> typeInfoC =
185  inferMMAType(op.eltypeB(), NVVM::MMAFrag::c, op.getContext());
186  SmallVector<Type, 32> arguments;
187  arguments.append(typeInfoA.second, typeInfoA.first);
188  arguments.append(typeInfoB.second, typeInfoB.first);
189  arguments.append(typeInfoC.second, typeInfoC.first);
190  unsigned numArgs = arguments.size();
191  if (op.args().size() != numArgs)
192  return op.emitOpError() << "expected " << numArgs << " arguments";
193  for (unsigned i = 0; i < numArgs; i++) {
194  if (op.args()[i].getType() != arguments[i])
195  return op.emitOpError()
196  << "expected argument " << i << " to be of type " << arguments[i];
197  }
198  Type dstType = LLVM::LLVMStructType::getLiteral(
199  op.getContext(), SmallVector<Type, 8>(typeInfoC.second, typeInfoC.first));
200  if (op.getType() != dstType)
201  return op.emitOpError("expected destination type is a structure of ")
202  << typeInfoC.second << " elements of type " << typeInfoC.first;
203  return success();
204 }
