170 /// Fuse a sequence of linalg operations (`ops`) using tile-and-fuse. This
171 /// proceeds as follows:
172 /// - Find outer parallel loops in these ops that can be fused.
173 /// - Tile fusable outer parallel loops of the last operation in the sequence.
174 /// - Fuse the remaining operations with the tiled operation
175 ///
176 /// For example, consider the sequence of matmul below
177 ///
178 /// linalg.matmul ins(%arg0, %arg1 : memref<256x32xf32>, memref<32x32xf32>)
179 /// outs(%arg2 : memref<256x32xf32>)
180 /// linalg.matmul ins(%arg2, %arg3 : memref<256x32xf32>, memref<32x32xf32>)
181 /// outs(%arg4 : memref<256x32xf32>)
182 ///
183 /// It is legal to fuse the RAW dependence (through %arg2) by only fusing the
184 /// matmuls row-wise. For example, the fused computation for the above is shown
185 /// below. The outer `scf.parallel` loop is the "fused" loop obtained by tiling
186 /// along the rows of the matrix. The entire rows of the first matmul operation
187 /// need to be computed before they can be used for the second matmul. The
188 /// second matmul is further tiled (similar to normal tiling).
189 ///
190 /// #map0 = affine_map<(d0, d1)[s0] -> (d0 * 32 + s0 + d1)>
191 /// #map1 = affine_map<(d0, d1) -> (d0 * 32 + d1)>
192 /// scf.parallel (%arg5) = (%c0) to (%c256) step (%c16) {
193 /// %0 = subview %arg2[%arg5, 0] [16, 32] [1, 1]
194 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
195 /// %1 = subview %arg4[%arg5, 0] [16, 32] [1, 1]
196 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
197 /// %2 = subview %arg0[%arg5, 0] [16, 32] [1, 1]
198 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
199 /// %3 = subview %arg1[0, 0] [32, 32] [1, 1]
200 /// : memref<32x32xf32> to memref<32x32xf32, #map1>
201 /// %4 = subview %arg3[0, 0] [32, 32] [1, 1]
202 /// : memref<32x32xf32> to memref<32x32xf32, #map1>
203 /// linalg.matmul
204 /// ins(%2, %3 : memref<16x32xf32, #map0>, memref<32x32xf32, #map1>)
205 /// outs(%0 : memref<16x32xf32, #map0>)
206 /// linalg.matmul
207 /// ins(%0, %4 : memref<16x4xf32, #map0>, memref<4x8xf32, #map0>)
208 /// outs(%1 : memref<16x8xf32, #map0>)
209 /// }
210 ///
211 /// `tilingOptions` are used to tile the corresponding operation in `ops` (the
212 /// size of the former should be same as size of the latter. Based on how
213 /// tile+fuse is implemented, the fused loops are generated based on the last
214 /// operation in the sequence. For example, the tile sizes for the fused loops
215 /// is obtained from `tilingOptions.back()`. The following tiling options are
216 /// handled differently in tile+fuse (compared to tile only)
217 /// - Interchange of the tiling loops is not supported right now.
218 /// - Only the fused loops are distributed.
219 struct TiledAndFusedLinalgOps {
220  /// Operation obtained by tiling the last operation in sequence of `ops`
221  /// passed to `tileAndFuseLinalgOps`.
222  LinalgOp op;
223  /// The dimension of the loops that are fused.
224  std::set<unsigned> fusedLoopDims;
225  /// The generated fused operations (created within the fused loops).
226  SmallVector<LinalgOp, 1> fusedProducers;
227  /// The fused loop generated.
228  SmallVector<Operation *, 4> fusedLoops;
229 };

1305 /// Converts Convolution op into vector contraction.
1306 ///
1307 /// Conversion expects ConvOp to have dimensions marked in the *mask* as
1308 /// false of size 1. This ensures that the ConvOp can be lowered to vector
1309 /// contraction of dimensions marked in the *mask* as true.
1310 ///

1312 /// with channels as the last dimension. Let's vectorize last 3 dimensions.
1313 /// The initial op definition looks like this:
1314 /// ```
1315 /// linalg.conv_2d_nhwc %arg0, %arg1, %arg2 :
1316 /// (memref<1x3x3x3xf32>, memref<1x3x3x3xf32>, memref<?x?x?x?xf32>)

1319 /// %arg1 (kernel) which is written into first entry of %arg2 (output). This is
1320 /// the ConvOp this pass expects and converts into:
1321 /// ```
1322 /// #map0 = affine_map<(d0, d1, d2) -> (d0, d1, d2)>
1323 /// #map1 = affine_map<(d0, d1, d2) -> ()>
1324 /// .....
1325 /// %0 = vector.transfer_read %arg0[%c0, %c0, %c0, %c0], %c0_f32
1326 /// : memref<1x3x3x3xf32>, vector<3x3x3xf32>

1328 /// : memref<1x3x3x3xf32>, vector<3x3x3xf32>
1329 /// %2 = vector.contract {indexing_maps = [#map0, #map0, #map1],
1330 /// iterator_types = ["reduction", "reduction", "reduction"]} %0, %1,
1331 /// %c0_f32 : vector<3x3x3xf32>, vector<3x3x3xf32> into f32
1332 /// store %2, %arg2[%c0, %c0, %c0, %c0] : memref<?x?x?x?xf32>
