28 /// This class provides an efficient unique identifier for a specific C++ type.
29 /// This allows for a C++ type to be compared, hashed, and stored in an opaque
30 /// context. This class is similar in some ways to std::type_index, but can be
31 /// used for any type. For example, this class could be used to implement LLVM
32 /// style isa/dyn_cast functionality for a type hierarchy:
33 ///
34 /// struct Base {
35 /// Base(TypeID typeID) : typeID(typeID) {}
36 /// TypeID typeID;
37 /// };
38 ///
39 /// struct DerivedA : public Base {
40 /// DerivedA() : Base(TypeID::get<DerivedA>()) {}
41 ///
42 /// static bool classof(const Base *base) {
43 /// return base->typeID == TypeID::get<DerivedA>();
44 /// }
45 /// };
46 ///
47 /// void foo(Base *base) {
48 /// if (DerivedA *a = llvm::dyn_cast<DerivedA>(base))
49 /// ...
50 /// }
51 ///
52 class TypeID {
53  /// This class represents the storage of a type info object.
54  /// Note: We specify an explicit alignment here to allow use with
55  /// PointerIntPair and other utilities/data structures that require a known
56  /// pointer alignment.
57  struct alignas(8) Storage {};
58 
59 public:
60  TypeID() : TypeID(get<void>()) {}
61 
62  /// Comparison operations.
63  inline bool operator==(const TypeID &other) const {
64  return storage == other.storage;
65  }
66  inline bool operator!=(const TypeID &other) const {
67  return !(*this == other);
68  }
69 
70  /// Construct a type info object for the given type T.
71  template <typename T>
72  static TypeID get();
73  template <template <typename> class Trait>
74  static TypeID get();
75 
76  /// Methods for supporting PointerLikeTypeTraits.
77  const void *getAsOpaquePointer() const {
78  return static_cast<const void *>(storage);
79  }
80  static TypeID getFromOpaquePointer(const void *pointer) {
81  return TypeID(reinterpret_cast<const Storage *>(pointer));
82  }
83 
84  /// Enable hashing TypeID.
85  friend ::llvm::hash_code hash_value(TypeID id);
86 
87 private:
88  TypeID(const Storage *storage) : storage(storage) {}
89 
90  /// The storage of this type info object.
91  const Storage *storage;
92 
93  // See TypeIDExported below for an explanation of the trampoline behavior.
94  friend struct detail::TypeIDExported;
95 };
96 
97 /// Enable hashing TypeID.
98 inline ::llvm::hash_code hash_value(TypeID id) {
99  return DenseMapInfo<const TypeID::Storage *>::getHashValue(id.storage);
100 }

104 /// The static local instance of each get method must be emitted with
105 /// "default" (public) visibility across all shared libraries, regardless of
106 /// whether they are compiled with hidden visibility or not. The only reliable
107 /// way to make this happen is to set the visibility attribute at the
108 /// containing namespace/struct scope. We don't do this on the TypeID (internal
109 /// API) class in order to reduce the scope of what gets exported with
110 /// public visibility. Instead, the get() methods on TypeID trampoline
111 /// through those on this detail class with specific visibility controls
112 /// applied, making visibility declarations on the internal TypeID class not
113 /// required (all visibility relevant pieces are here).
114 /// TODO: This currently won't work when using DLLs as it requires properly
115 /// attaching dllimport and dllexport. Fix this when that information is
116 /// available within LLVM.
117 struct LLVM_EXTERNAL_VISIBILITY TypeIDExported {
118  template <typename T>
119  static TypeID get() {
120  static TypeID::Storage instance;
121  return TypeID(&instance);
122  }
123  template <template <typename> class Trait>
124  static TypeID get() {
125  static TypeID::Storage instance;
126  return TypeID(&instance);
127  }
128 };

143 // Declare/define an explicit specialization for TypeID: this forces the
144 // compiler to emit a strong definition for a class and controls which
145 // translation unit and shared object will actually have it.
146 // This can be useful to turn to a link-time failure what would be in other
147 // circumstances a hard-to-catch runtime bug when a TypeID is hidden in two
148 // different shared libraries and instances of the same class only gets the same
149 // TypeID inside a given DSO.
150 #define DECLARE_EXPLICIT_TYPE_ID(CLASS_NAME) \
151  namespace mlir { \
152  namespace detail { \
153  template <> \
154  LLVM_EXTERNAL_VISIBILITY TypeID TypeIDExported::get<CLASS_NAME>(); \
155  } \
156  }

159  namespace mlir { \
160  namespace detail { \
161  template <> \
162  LLVM_EXTERNAL_VISIBILITY TypeID TypeIDExported::get<CLASS_NAME>() { \
163  static TypeID::Storage instance; \
