25 /// This is a class to represent a resizable matrix.
26 ///
27 /// More columns and rows can be reserved than are currently used. The data is
28 /// stored as a single 1D array, viewed as a 2D matrix with nRows rows and
29 /// nReservedColumns columns, stored in row major form. Thus the element at
30 /// (i, j) is stored at data[i*nReservedColumns + j]. The reserved but unused
31 /// columns always have all zero values. The reserved rows are just reserved
32 /// space in the underlying SmallVector's capacity.
33 class Matrix {
34 public:
35  Matrix() = delete;
36 
37  /// Construct a matrix with the specified number of rows and columns.
38  /// The number of reserved rows and columns will be at least the number
39  /// specified, and will always be sufficient to accomodate the number of rows
40  /// and columns specified.
41  ///
42  /// Initially, the entries are initialized to ero.
43  Matrix(unsigned rows, unsigned columns, unsigned reservedRows = 0,
44  unsigned reservedColumns = 0);
45 
46  /// Return the identity matrix of the specified dimension.
47  static Matrix identity(unsigned dimension);

53  int64_t operator()(unsigned row, unsigned column) const;
54

57

59  void swapRows(unsigned row, unsigned otherRow);

77  /// Insert columns having positions pos, pos + 1, ... pos + count - 1.
78  /// Columns that were at positions 0 to pos - 1 will stay where they are;
79  /// columns that were at positions pos to nColumns - 1 will be pushed to the
80  /// right. pos should be at most nColumns.
81  void insertColumns(unsigned pos, unsigned count);
82  void insertColumn(unsigned pos);
83 
84  /// Insert rows having positions pos, pos + 1, ... pos + count - 1.
85  /// Rows that were at positions 0 to pos - 1 will stay where they are;
86  /// rows that were at positions pos to nColumns - 1 will be pushed to the
87  /// right. pos should be at most nRows.
88  void insertRows(unsigned pos, unsigned count);
89  void insertRow(unsigned pos);
90 
91  /// Remove the columns having positions pos, pos + 1, ... pos + count - 1.
92  /// Rows that were at positions 0 to pos - 1 will stay where they are;
93  /// columns that were at positions pos + count - 1 or later will be pushed to
94  /// the right. The columns to be deleted must be valid rows: pos + count - 1
95  /// must be at most nColumns - 1.
96  void removeColumns(unsigned pos, unsigned count);
97  void removeColumn(unsigned pos);
98 
99  /// Remove the rows having positions pos, pos + 1, ... pos + count - 1.
100  /// Rows that were at positions 0 to pos - 1 will stay where they are;
101  /// rows that were at positions pos + count - 1 or later will be pushed to the
102  /// right. The rows to be deleted must be valid rows: pos + count - 1 must be
103  /// at most nRows - 1.
104  void removeRows(unsigned pos, unsigned count);
105  void removeRow(unsigned pos);
106 
107  void copyRow(unsigned sourceRow, unsigned targetRow);
108

111 
112  /// Add `scale` multiples of the source column to the target column.

114

117 
118  /// Resize the matrix to the specified dimensions. If a dimension is smaller,
119  /// the values are truncated; if it is bigger, the new values are initialized
120  /// to zero.
121  ///
122  /// Due to the representation of the matrix, resizing vertically (adding rows)

124  /// nReservedColumns.

126  void resizeHorizontally(unsigned newNColumns);
127  void resizeVertically(unsigned newNRows);

135

137  /// invariants satisfied.
138  bool hasConsistentState() const;
139 
140 private:
141  /// The current number of rows, columns, and reserved columns. The underlying
142  /// data vector is viewed as an nRows x nReservedColumns matrix, of which the
143  /// first nColumns columns are currently in use, and the remaining are
144  /// reserved columns filled with zeros.
145  unsigned nRows, nColumns, nReservedColumns;
146 
147  /// Stores the data. data.size() is equal to nRows * nReservedColumns.
148  /// data.capacity() / nReservedColumns is the number of reserved rows.
149  SmallVector<int64_t, 64> data;
