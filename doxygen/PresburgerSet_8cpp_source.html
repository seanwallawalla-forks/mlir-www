105 // Return the intersection of this set with the given set.
106 //
107 // We directly compute (S_1 or S_2 ...) and (T_1 or T_2 ...)
108 // as (S_1 and T_1) or (S_1 and T_2) or ...
109 //
110 // If S_i or T_j have local variables, then S_i and T_j contains the local
111 // variables of both.
112 PresburgerSet PresburgerSet::intersect(const PresburgerSet &set) const {
113  assertDimensionsCompatible(set, *this);
114 
115  PresburgerSet result(nDim, nSym);
116  for (const FlatAffineConstraints &csA : flatAffineConstraints) {
117  for (const FlatAffineConstraints &csB : set.flatAffineConstraints) {
118  FlatAffineConstraints csACopy = csA, csBCopy = csB;
119  csACopy.mergeLocalIds(csBCopy);
120  csACopy.append(std::move(csBCopy));
121  if (!csACopy.isEmpty())
122  result.unionFACInPlace(std::move(csACopy));
123  }
124  }
125  return result;
126 }

137 /// Return the complement of the given inequality.
138 ///
139 /// The complement of a_1 x_1 + ... + a_n x_ + c >= 0 is
140 /// a_1 x_1 + ... + a_n x_ + c < 0, i.e., -a_1 x_1 - ... - a_n x_ - c - 1 >= 0,
141 /// since all the variables are constrained to be integers.
142 static SmallVector<int64_t, 8> getComplementIneq(ArrayRef<int64_t> ineq) {
143  SmallVector<int64_t, 8> coeffs;
144  coeffs.reserve(ineq.size());
145  for (int64_t coeff : ineq)
146  coeffs.emplace_back(-coeff);
147  --coeffs.back();
148  return coeffs;
149 }

151 /// Return the set difference b \ s and accumulate the result into `result`.
152 /// `simplex` must correspond to b.
153 ///
154 /// In the following, U denotes union, ^ denotes intersection, \ denotes set
155 /// difference and ~ denotes complement.
156 /// Let b be the FlatAffineConstraints and s = (U_i s_i) be the set. We want
157 /// b \ (U_i s_i).
158 ///
159 /// Let s_i = ^_j s_ij, where each s_ij is a single inequality. To compute
160 /// b \ s_i = b ^ ~s_i, we partition s_i based on the first violated inequality:
161 /// ~s_i = (~s_i1) U (s_i1 ^ ~s_i2) U (s_i1 ^ s_i2 ^ ~s_i3) U ...
162 /// And the required result is (b ^ ~s_i1) U (b ^ s_i1 ^ ~s_i2) U ...
163 /// We recurse by subtracting U_{j > i} S_j from each of these parts and
164 /// returning the union of the results. Each equality is handled as a
165 /// conjunction of two inequalities.
166 ///
167 /// Note that the same approach works even if an inequality involves a floor
168 /// division. For example, the complement of x <= 7*floor(x/7) is still
169 /// x > 7*floor(x/7). Since b \ s_i contains the inequalities of both b and s_i
170 /// (or the complements of those inequalities), b \ s_i may contain the
171 /// divisions present in both b and s_i. Therefore, we need to add the local
172 /// division variables of both b and s_i to each part in the result. This means
173 /// adding the local variables of both b and s_i, as well as the corresponding
174 /// division inequalities to each part. Since the division inequalities are
175 /// added to each part, we can skip the parts where the complement of any
176 /// division inequality is added, as these parts will become empty anyway.
177 ///
178 /// As a heuristic, we try adding all the constraints and check if simplex
179 /// says that the intersection is empty. If it is, then subtracting this FAC is
180 /// a no-op and we just skip it. Also, in the process we find out that some
181 /// constraints are redundant. These redundant constraints are ignored.
182 ///
183 /// b and simplex are callee saved, i.e., their values on return are
184 /// semantically equivalent to their values when the function is called.
185 static void subtractRecursively(FlatAffineConstraints &b, Simplex &simplex,
186  const PresburgerSet &s, unsigned i,
187  PresburgerSet &result) {
188  if (i == s.getNumFACs()) {
189  result.unionFACInPlace(b);
190  return;
191  }
192  FlatAffineConstraints sI = s.getFlatAffineConstraints(i);
193  unsigned bInitNumLocals = b.getNumLocalIds();
194 
195  // Find out which inequalities of sI correspond to division inequalities for
196  // the local variables of sI.
197  std::vector<llvm::Optional<std::pair<unsigned, unsigned>>> repr(
198  sI.getNumLocalIds());
199  sI.getLocalReprs(repr);
200 
201  // Add sI's locals to b, after b's locals. Also add b's locals to sI, before
202  // sI's locals.
203  b.mergeLocalIds(sI);
204 
205  // Mark which inequalities of sI are division inequalities and add all such
206  // inequalities to b.
207  llvm::SmallBitVector isDivInequality(sI.getNumInequalities());
208  for (Optional<std::pair<unsigned, unsigned>> &maybePair : repr) {
209  assert(maybePair &&
210  "Subtraction is not supported when a representation of the local "
211  "variables of the subtrahend cannot be found!");
212 
213  b.addInequality(sI.getInequality(maybePair->first));
214  b.addInequality(sI.getInequality(maybePair->second));
215 
216  assert(maybePair->first != maybePair->second &&
217  "Upper and lower bounds must be different inequalities!");
218  isDivInequality[maybePair->first] = true;
219  isDivInequality[maybePair->second] = true;
220  }
221 
222  unsigned initialSnapshot = simplex.getSnapshot();
223  unsigned offset = simplex.getNumConstraints();
224  unsigned numLocalsAdded = b.getNumLocalIds() - bInitNumLocals;
225  simplex.appendVariable(numLocalsAdded);
226 
227  unsigned snapshotBeforeIntersect = simplex.getSnapshot();
228  simplex.intersectFlatAffineConstraints(sI);
229 
230  if (simplex.isEmpty()) {
231  /// b ^ s_i is empty, so b \ s_i = b. We move directly to i + 1.
232  simplex.rollback(initialSnapshot);
233  b.removeIdRange(FlatAffineConstraints::IdKind::Local, bInitNumLocals,
234  b.getNumLocalIds());
235  subtractRecursively(b, simplex, s, i + 1, result);
236  return;
237  }
238 
239  simplex.detectRedundant();
240 
241  // Equalities are added to simplex as a pair of inequalities.
242  unsigned totalNewSimplexInequalities =
243  2 * sI.getNumEqualities() + sI.getNumInequalities();
244  llvm::SmallBitVector isMarkedRedundant(totalNewSimplexInequalities);
245  for (unsigned j = 0; j < totalNewSimplexInequalities; j++)
246  isMarkedRedundant[j] = simplex.isMarkedRedundant(offset + j);
247 
248  simplex.rollback(snapshotBeforeIntersect);
249 
250  // Recurse with the part b ^ ~ineq. Note that b is modified throughout
251  // subtractRecursively. At the time this function is called, the current b is
252  // actually equal to b ^ s_i1 ^ s_i2 ^ ... ^ s_ij, and ineq is the next
253  // inequality, s_{i,j+1}. This function recurses into the next level i + 1
254  // with the part b ^ s_i1 ^ s_i2 ^ ... ^ s_ij ^ ~s_{i,j+1}.
255  auto recurseWithInequality = [&, i](ArrayRef<int64_t> ineq) {
256  size_t snapshot = simplex.getSnapshot();
257  b.addInequality(ineq);
258  simplex.addInequality(ineq);
259  subtractRecursively(b, simplex, s, i + 1, result);
260  b.removeInequality(b.getNumInequalities() - 1);
261  simplex.rollback(snapshot);
262  };
263 
264  // For each inequality ineq, we first recurse with the part where ineq
265  // is not satisfied, and then add the ineq to b and simplex because
266  // ineq must be satisfied by all later parts.
267  auto processInequality = [&](ArrayRef<int64_t> ineq) {
268  recurseWithInequality(getComplementIneq(ineq));
269  b.addInequality(ineq);
270  simplex.addInequality(ineq);
271  };
272 
273  // processInequality appends some additional constraints to b. We want to
274  // rollback b to its initial state before returning, which we will do by
275  // removing all constraints beyond the original number of inequalities
276  // and equalities, so we store these counts first.
277  unsigned bInitNumIneqs = b.getNumInequalities();
278  unsigned bInitNumEqs = b.getNumEqualities();
279 
280  // Process all the inequalities, ignoring redundant inequalities and division
281  // inequalities. The result is correct whether or not we ignore these, but
282  // ignoring them makes the result simpler.
283  for (unsigned j = 0, e = sI.getNumInequalities(); j < e; j++) {
284  if (isMarkedRedundant[j])
285  continue;
286  if (isDivInequality[j])
287  continue;
288  processInequality(sI.getInequality(j));
289  }
290 
291  offset = sI.getNumInequalities();
292  for (unsigned j = 0, e = sI.getNumEqualities(); j < e; ++j) {
293  ArrayRef<int64_t> coeffs = sI.getEquality(j);
294  // For each equality, process the positive and negative inequalities that
295  // make up this equality. If Simplex found an inequality to be redundant, we
296  // skip it as above to make the result simpler. Divisions are always
297  // represented in terms of inequalities and not equalities, so we do not
298  // check for division inequalities here.
299  if (!isMarkedRedundant[offset + 2 * j])
300  processInequality(coeffs);
301  if (!isMarkedRedundant[offset + 2 * j + 1])
302  processInequality(getNegatedCoeffs(coeffs));
303  }
304 
305  // Rollback b and simplex to their initial states.
306  b.removeIdRange(FlatAffineConstraints::IdKind::Local, bInitNumLocals,
307  b.getNumLocalIds());
308  b.removeInequalityRange(bInitNumIneqs, b.getNumInequalities());
309  b.removeEqualityRange(bInitNumEqs, b.getNumEqualities());
310 
311  simplex.rollback(initialSnapshot);
312 }

349 /// Two sets S and T are equal iff S contains T and T contains S.
350 /// By "S contains T", we mean that S is a superset of or equal to T.
351 ///
352 /// S contains T iff T \ S is empty, since if T \ S contains a
353 /// point then this is a point that is contained in T but not S.
354 ///
355 /// Therefore, S is equal to T iff S \ T and T \ S are both empty.
356 bool PresburgerSet::isEqual(const PresburgerSet &set) const {
357  assertDimensionsCompatible(set, *this);
358  return this->subtract(set).isIntegerEmpty() &&
359  set.subtract(*this).isIntegerEmpty();
360 }
