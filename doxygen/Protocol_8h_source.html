509 /// Represents programming constructs like variables, classes, interfaces etc.
510 /// that appear in a document. Document symbols can be hierarchical and they
511 /// have two ranges: one that encloses its definition and one that points to its
512 /// most interesting range, e.g. the range of an identifier.
513 struct DocumentSymbol {
514  DocumentSymbol() = default;
515  DocumentSymbol(DocumentSymbol &&) = default;
516  DocumentSymbol(const Twine &name, SymbolKind kind, Range range,
517  Range selectionRange)
518  : name(name.str()), kind(kind), range(range),
519  selectionRange(selectionRange) {}
520 
521  /// The name of this symbol.
522  std::string name;
523 
524  /// More detail for this symbol, e.g the signature of a function.
525  std::string detail;
526 
527  /// The kind of this symbol.
528  SymbolKind kind;
529 
530  /// The range enclosing this symbol not including leading/trailing whitespace
531  /// but everything else like comments. This information is typically used to
532  /// determine if the clients cursor is inside the symbol to reveal in the
533  /// symbol in the UI.
534  Range range;
535 
536  /// The range that should be selected and revealed when this symbol is being
537  /// picked, e.g the name of a function. Must be contained by the `range`.
538  Range selectionRange;
539 
540  /// Children of this symbol, e.g. properties of a class.
541  std::vector<DocumentSymbol> children;
542 };
