49 /// Holds parameters to perform n-D vectorization on a single loop nest.
50 /// For example, for the following loop nest:
51 ///
52 /// func @vec2d(%in: memref<64x128x512xf32>, %out: memref<64x128x512xf32>) {
53 /// affine.for %i0 = 0 to 64 {
54 /// affine.for %i1 = 0 to 128 {
55 /// affine.for %i2 = 0 to 512 {
56 /// %ld = affine.load %in[%i0, %i1, %i2] : memref<64x128x512xf32>
57 /// affine.store %ld, %out[%i0, %i1, %i2] : memref<64x128x512xf32>
58 /// }
59 /// }
60 /// }
61 /// return
62 /// }
63 ///
64 /// and VectorizationStrategy = 'vectorSizes = {8, 4}', 'loopToVectorDim =
65 /// {{i1->0}, {i2->1}}', SuperVectorizer will generate:
66 ///
67 /// func @vec2d(%arg0: memref<64x128x512xf32>, %arg1: memref<64x128x512xf32>) {
68 /// affine.for %arg2 = 0 to 64 {
69 /// affine.for %arg3 = 0 to 128 step 8 {
70 /// affine.for %arg4 = 0 to 512 step 4 {
71 /// %cst = arith.constant 0.000000e+00 : f32
72 /// %0 = vector.transfer_read %arg0[%arg2, %arg3, %arg4], %cst : ...
73 /// vector.transfer_write %0, %arg1[%arg2, %arg3, %arg4] : ...
74 /// }
75 /// }
76 /// }
77 /// return
78 /// }
79 // TODO: Hoist to a VectorizationStrategy.cpp when appropriate.
80 struct VectorizationStrategy {
81  // Vectorization factors to apply to each target vector dimension.
82  // Each factor will be applied to a different loop.
83  SmallVector<int64_t, 8> vectorSizes;
84  // Maps each AffineForOp vectorization candidate with its vector dimension.
85  // The candidate will be vectorized using the vectorization factor in
86  // 'vectorSizes' for that dimension.
87  DenseMap<Operation *, unsigned> loopToVectorDim;
88  // Maps loops that implement vectorizable reductions to the corresponding
89  // reduction descriptors.
90  ReductionLoopMap reductionLoops;
91 };

106 /// External utility to vectorize affine loops from a single loop nest using an
107 /// n-D vectorization strategy (see doc in VectorizationStrategy definition).
108 /// Loops are provided in a 2D vector container. The first dimension represents
109 /// the nesting level relative to the loops to be vectorized. The second
110 /// dimension contains the loops. This means that:
111 /// a) every loop in 'loops[i]' must have a parent loop in 'loops[i-1]',
112 /// b) a loop in 'loops[i]' may or may not have a child loop in 'loops[i+1]'.
113 ///
114 /// For example, for the following loop nest:
115 ///
116 /// func @vec2d(%in0: memref<64x128x512xf32>, %in1: memref<64x128x128xf32>,
117 /// %out0: memref<64x128x512xf32>,
118 /// %out1: memref<64x128x128xf32>) {
119 /// affine.for %i0 = 0 to 64 {
120 /// affine.for %i1 = 0 to 128 {
121 /// affine.for %i2 = 0 to 512 {
122 /// %ld = affine.load %in0[%i0, %i1, %i2] : memref<64x128x512xf32>
123 /// affine.store %ld, %out0[%i0, %i1, %i2] : memref<64x128x512xf32>
124 /// }
125 /// affine.for %i3 = 0 to 128 {
126 /// %ld = affine.load %in1[%i0, %i1, %i3] : memref<64x128x128xf32>
127 /// affine.store %ld, %out1[%i0, %i1, %i3] : memref<64x128x128xf32>
128 /// }
129 /// }
130 /// }
131 /// return
132 /// }
133 ///
134 /// loops = {{%i0}, {%i2, %i3}}, to vectorize the outermost and the two
135 /// innermost loops;
136 /// loops = {{%i1}, {%i2, %i3}}, to vectorize the middle and the two innermost
137 /// loops;
138 /// loops = {{%i2}}, to vectorize only the first innermost loop;
139 /// loops = {{%i3}}, to vectorize only the second innermost loop;
140 /// loops = {{%i1}}, to vectorize only the middle loop.
141 LogicalResult
142 vectorizeAffineLoopNest(std::vector<SmallVector<AffineForOp, 2>> &loops,
143  const VectorizationStrategy &strategy);
