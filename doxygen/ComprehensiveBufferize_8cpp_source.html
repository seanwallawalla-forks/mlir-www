9 // Perform inplace bufferization within function boundaries.
10 // This is a specialized pass that supports inplace analysis for a fixed subset
11 // of ops that have well-defined inplace semantics.
12 // This pass caters to high-performance codegen where buffer reuse is deemed
13 // critical: the pass should fail if the bufferized form of the function needs
14 // to return any buffer.
15 // Generic control-flow and branching are unsupported.
16 // Composability with extensible set of ops is not a first-class concern.
17 //
18 // Bufferization occurs by:
19 // a. performing an inPlace analysis `inPlaceAnalysis` which marks each
20 // operation within the op with the `kInPlaceResultsAttrName` attribute.
21 // b. traversing each operation in the op and rewriting it in
22 // buffer form and keeping a BlockAndValueMapping mapping of the
23 // rewrites. New allocations are introduced during this step.
24 // TODO: Allocation + depending op hoisting to outermost enclosing
25 // sequential scope.
26 // c. at the end of this bufferization, 3 cases may occur:
27 // i. inplaceable function arguments may be reused in place after the
28 // function itself has been bufferized. This is encoded by IR resembling:
29 //
30 // ```
31 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
32 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true})
33 // -> tensor<?xf32> {
34 // %0 = bufferization.to_memref %A : memref<?xf32, #map>
35 // // ... uses of %0
36 // %res = bufferization.to_tensor %0 : memref<?xf32, #map>
37 // return %res : tensor<?xf32>
38 // }
39 // ```
40 //
41 // this is the cue for the bufferization of the function foo (and calls
42 // to it) may bufferize to `func @foo(%A: memref<?xf32, some_layout>)`.
43 // To fully achieve bufferization, an additional analysis is needed to
44 // determine whether function argument/operand pairs bufferize to a
45 // single inplace buffer argument (i.e. functions may return tensors in
46 // arbitrary order that may not match argument numbers).
47 //
48 // ii. results that don't map to an inplaceable function argument are
49 // generally allocated. Since memref semantics wrt ownership of the
50 // underlying memory region are not well-defined, comprehensive
51 // bufferization chooses to perform allocations in a scoped fashion:
52 // returning memrefs is always considered illegal.
53 // Such scenarios are encoded by IR resembling:
54 //
55 // ```
56 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
57 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true})
58 // -> tensor<?xf32> {
59 // %0 = bufferization.to_memref %A : memref<?xf32, #map>
60 // %1 = memref.dim %0, %c0 : memref<?xf32, #map>
61 // %2 = memref.alloc(%1) : memref<?xf32>
62 // %3 = memref.cast %2 : memref<?xf32> to memref<?xf32, #map>
63 // // ... uses of %3
64 // memref.dealloc %2 : memref<?xf32, #map>
65 // %res = bufferization.to_tensor %3 : memref<?xf32, #map>
66 // return %res : tensor<?xf32>
67 // }
68 // ```
69 //
70 // this is the cue for the bufferization of the function foo (and calls
71 // to it) that it must bufferize to `func @foo(%A: memref<?xf32,
72 // some_layout>,
73 // %B: memref<?xf32, some_layout>)` (i.e. make a cloned
74 // allocation of the result tensor)
75 // To fully achieve bufferization, the alloc/dealloc pair must be lifted
76 // out of the function at each call site.
77 //
78 // iii. as an optimization over ii., it may be possible to reuse an argument
79 // and only want to return a slice.
80 // This may forego allocation by letting *all* callers decide whether to
81 // pass a new *aliasing* memref function argument (i.e. a subview).
82 // Without loss of generality, callers may agree to allocate a new buffer
83 // to avoid this aliasing. Such scenarios are encoded by IR resembling:
84 //
85 // ```
86 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
87 // func @foo(%arg0: tensor<?xf32> {linalg.inplaceable = true})
88 // -> tensor<4xf32> {
89 // %0 = bufferization.to_memref %arg0 : memref<?xf32, #map>
90 // %1 = memref.subview %0[0] [4] [1] : memref<?xf32, #map> to
91 // memref<4xf32, #map>
92 // // ... inplace computes into %1
93 // %3 = bufferization.to_tensor %1 : memref<4xf32, #map>
94 // return %3 : tensor<4xf32>
95 // }
96 // ```
97 //
98 // Note: In the future, it may be worthwhile to design special bufferization
99 // ops to encode the desired semantics at function boundaries for i., ii. and
100 // iii.
101 //
102 // Lastly, note that layout map chosen to bufferize is the most dynamic
103 // canonical strided layout of the proper rank. This ensures compatibility with
104 // expected layouts after transformations. Combinations of memref.cast +
105 // canonicalization are responsible for clean ups.

149 /// Attribute marker to specify op results that can be bufferized inPlace.
150 constexpr StringLiteral kInPlaceResultsAttrName = "__inplace_results_attr__";
151 
152 /// Mark whether OpResult can actually be bufferized inplace.
153 /// If `inPlace` is `true`, the use-def chain analysis has guaranteed that no
154 /// subsequent write would occur to the bufferized tensor value (i.e. the result
155 /// can be bufferized inplace).
156 static void setInPlaceOpResult(OpResult opResult, bool inPlace) {
157  if (!opResult)
158  return;
159 
160  Operation *op = opResult.getOwner();
161  auto attr =
162  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
163  SmallVector<StringRef> inPlaceVector =
164  attr ? SmallVector<StringRef>(
165  llvm::to_vector<4>(attr.getAsValueRange<StringAttr>()))
166  : SmallVector<StringRef>(op->getNumResults(), "false");
167  LDBG("->set inPlace=" << inPlace << " <- #" << opResult.getResultNumber()
168  << ": " << printOperationInfo(op) << "\n");
169  inPlaceVector[opResult.getResultNumber()] = inPlace ? "true" : "false";
170  op->setAttr(kInPlaceResultsAttrName,
171  OpBuilder(op).getStrArrayAttr(inPlaceVector));
172 }

229 /// Return true if opOperand has been decided to bufferize in-place.
230 static bool isInplaceMemoryWrite(OpOperand &opOperand,
231  const BufferizationAliasInfo &aliasInfo) {
232  // The analysis does not know what happens to the result of a ToMemrefOp, so
233  // we assume that it is written to.
234  // TODO: This is a conservative implementation. This rule will have to be
235  // relaxed for partial bufferization.
236  if (isa<bufferization::ToMemrefOp>(opOperand.getOwner()))
237  return true;
238  // OpOperands without an aliasing OpResult do not write.
239  OpResult opResult = getAliasingOpResult(opOperand);
240  if (!opResult)
241  return false;
242  // OpOperands that do not bufferize to a memory write do not write in-place.
243  if (!bufferizesToMemoryWrite(opOperand))
244  return false;
245  // Check current bufferization decisions.
246  return aliasInfo.isInPlace(opResult);
247 }

249 /// Return true if, under current bufferization decisions, the buffer of `value`
250 /// is not writable.
251 static bool aliasesNonWritableBuffer(Value value,
252  const BufferizationAliasInfo &aliasInfo,
253  BufferizationState &state) {
254  LDBG("WRITABILITY ANALYSIS FOR " << printValueInfo(value) << "\n");
255  bool foundNonWritableBuffer = false;
256  aliasInfo.applyOnAliases(value, [&](Value v) {
257  // Query BufferizableOpInterface to see if the OpResult is writable.
258  // TODO: Out-of-place bufferized OpResult could be considered writable.
259  if (auto bufferizableOp = v.getDefiningOp<BufferizableOpInterface>())
260  if (bufferizableOp && bufferizableOp.isWritable(v, state))
261  return;
262 
263  // Query BufferizableOpInterface to see if the BlockArgument is writable.
264  if (auto bbArg = v.dyn_cast<BlockArgument>())
265  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(
266  bbArg.getOwner()->getParentOp()))
267  if (bufferizableOp.isWritable(bbArg, state))
268  return;
269 
270  foundNonWritableBuffer = true;
271  });
272 
273  if (foundNonWritableBuffer)
274  LDBG("--> NON WRITABLE\n");
275  else
276  LDBG("--> WRITABLE\n");
277 
278  return foundNonWritableBuffer;
279 }

281 /// Return true if the buffer to which `operand` would bufferize is equivalent
282 /// to some buffer write.
283 static bool aliasesInPlaceWrite(Value value,
284  const BufferizationAliasInfo &aliasInfo) {
285  LDBG("----Start aliasesInPlaceWrite\n");
286  LDBG("-------for : " << printValueInfo(value) << '\n');
287  bool foundInplaceWrite = false;
288  aliasInfo.applyOnAliases(value, [&](Value v) {
289  for (auto &use : v.getUses()) {
290  if (isInplaceMemoryWrite(use, aliasInfo)) {
291  LDBG("-----------wants to bufferize to inPlace write: "
292  << printOperationInfo(use.getOwner()) << '\n');
293  foundInplaceWrite = true;
294  return;
295  }
296  }
297  });
298 
299  if (!foundInplaceWrite)
300  LDBG("----------->does not alias an inplace write\n");
301 
302  return foundInplaceWrite;
303 }

305 /// Return true if `a` happens before `b`, i.e., `a` or one of its ancestors
306 /// properly dominates `b` and `b` is not inside `a`.
307 static bool happensBefore(Operation *a, Operation *b,
308  const DominanceInfo &domInfo) {
309  do {
310  // TODO: Instead of isProperAncestor + properlyDominates, we should use
311  // properlyDominatesImpl(a, b, /*enclosingOpOk=*/false)
312  if (a->isProperAncestor(b))
313  return false;
314  if (domInfo.properlyDominates(a, b))
315  return true;
316  } while ((a = a->getParentOp()));
317  return false;
318 }

320 /// Given sets of uses and writes, return true if there is a RaW conflict under
321 /// the assumption that all given reads/writes alias the same buffer and that
322 /// all given writes bufferize inplace.
323 ///
324 /// A conflict is: According to SSA use-def chains, a read R is supposed to read
325 /// the result of a write W1. But because of bufferization decisions, R actually
326 /// reads another write W2.
327 static bool
328 hasReadAfterWriteInterference(const DenseSet<OpOperand *> &usesRead,
329  const DenseSet<OpOperand *> &usesWrite,
330  const DominanceInfo &domInfo,
331  const BufferizationAliasInfo &aliasInfo) {
332  for (OpOperand *uRead : usesRead) {
333  Operation *readingOp = uRead->getOwner();
334 
335  // Find most recent write of uRead by following the SSA use-def chain. E.g.:
336  //
337  // %0 = "writing_op"(%t) : tensor<?x32> -> tensor<?xf32>
338  // %1 = "aliasing_op"(%0) : tensor<?x32> -> tensor<?xf32>
339  // %2 = "reading_op"(%1) : : tensor<?x32> -> not_a_tensor_type
340  //
341  // In the above example, if uRead is the OpOperand of reading_op, lastWrite
342  // is %0. Note that operations that create an alias but do not write (such
343  // as ExtractSliceOp) are skipped.
344  Value lastWrite = findLastPrecedingWrite(uRead->get());
345 
346  // Look for conflicting memory writes. Potential conflicts are writes to an
347  // alias that have been decided to bufferize inplace.
348  for (OpOperand *uConflictingWrite : usesWrite) {
349  // Throughout this loop, check for multiple requirements that have to be
350  // met for uConflictingWrite to be an actual conflict.
351  Operation *conflictingWritingOp = uConflictingWrite->getOwner();
352 
353  // Print some debug info.
354  LDBG("Found potential conflict:\n");
355  LDBG("READ = #" << uRead->getOperandNumber() << " of "
356  << printOperationInfo(readingOp) << "\n");
357  LDBG("CONFLICTING WRITE = #"
358  << uConflictingWrite->getOperandNumber() << " of "
359  << printOperationInfo(conflictingWritingOp) << "\n");
360 
361  // No conflict if the readingOp dominates conflictingWritingOp, i.e., the
362  // write is not visible when reading.
363  if (happensBefore(readingOp, conflictingWritingOp, domInfo))
364  continue;
365 
366  // No conflict if the reading use equals the use of the conflicting write.
367  // A use cannot conflict with itself. Note: Just being the same op is not
368  // enough. It has to be the same use.
369  if (uConflictingWrite == uRead)
370  continue;
371 
372  // No conflict if the op interface says so.
373  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(readingOp))
374  if (bufferizableOp.isNotConflicting(uRead, uConflictingWrite,
375  aliasInfo))
376  continue;
377 
378  if (conflictingWritingOp != readingOp)
379  if (auto bufferizableOp =
380  dyn_cast<BufferizableOpInterface>(conflictingWritingOp))
381  if (bufferizableOp.isNotConflicting(uRead, uConflictingWrite,
382  aliasInfo))
383  continue;
384 
385  // Ops are not conflicting if they are in mutually exclusive regions.
386  if (insideMutuallyExclusiveRegions(readingOp, conflictingWritingOp))
387  continue;
388 
389  LDBG("WRITE = #" << printValueInfo(lastWrite) << "\n");
390 
391  // No conflict if the conflicting write happens before the last
392  // write.
393  if (Operation *writingOp = lastWrite.getDefiningOp()) {
394  if (happensBefore(conflictingWritingOp, writingOp, domInfo))
395  // conflictingWritingOp happens before writingOp. No conflict.
396  continue;
397  // No conflict if conflictingWritingOp is contained in writingOp.
398  if (writingOp->isProperAncestor(conflictingWritingOp))
399  continue;
400  } else {
401  auto bbArg = lastWrite.cast<BlockArgument>();
402  Block *block = bbArg.getOwner();
403  if (!block->findAncestorOpInBlock(*conflictingWritingOp))
404  // conflictingWritingOp happens outside of the block. No
405  // conflict.
406  continue;
407  }
408 
409  // No conflict if the conflicting write and the last write are the same
410  // use.
411  if (getAliasingOpResult(*uConflictingWrite) == lastWrite)
412  continue;
413 
414  // All requirements are met. Conflict found!
415  LDBG("CONFLICT CONFIRMED!\n\n");
416  return true;
417  }
418  }
419 
420  LDBG("NOT A CONFLICT!\n\n");
421  return false;
422 }

424 /// Return true if bufferizing result inplace would create a conflict. A read R
425 /// and a write W of the same alias set is a conflict if inplace bufferization
426 /// of W changes the value read by R to a value different from the one that
427 /// would be expected by tracing back R's origin through SSA use-def chains.
428 /// A conflict can only be introduced by a new alias and/or an inplace
429 /// bufferization decision.
430 ///
431 /// Example:
432 /// %0 = tensor.extract_slice %t[...][...][1, 1] {inplace?}
433 /// %1 = vector.transfer_write %v1, %t {inplace} : vector<5xf32>, tensor<?xf32>
434 /// %e = tensor.extract_slice %1
435 /// %2 = vector.transfer_write %v2, %0 {inplace} : vector<6xf32>, tensor<?xf32>
436 /// %3 = vector.transfer_read %e, %cst : tensor<?xf32>, vector<7xf32>
437 ///
438 /// In the above example, the two TransferWriteOps have already been decided to
439 /// bufferize inplace. Bufferizing the ExtractSliceOp inplace would create a
440 /// conflict because:
441 /// * According to SSA use-def chains, we expect to read the result of %1.
442 /// * However, adding an alias {%0, %t} would mean that the second
443 /// TransferWriteOp overwrites the first one. Therefore, the TransferReadOp
444 /// would no longer be reading the result of %1.
445 ///
446 /// If `checkConsistencyOnly` is true, this function checks if there is a
447 /// read-after-write conflict without bufferizing `operand` inplace. This would
448 /// indicate a problem with the current inplace bufferization decisions.
449 ///
450 /// Note: If `checkConsistencyOnly`, this function may be called with a null
451 /// OpResult. In that case, only the consistency of bufferization decisions
452 /// involving aliases of the given OpOperand are checked.
453 bool wouldCreateReadAfterWriteInterference(
454  OpOperand &operand, OpResult result, const DominanceInfo &domInfo,
455  const BufferizationAliasInfo &aliasInfo,
456  bool checkConsistencyOnly = false) {
457 #ifndef NDEBUG
458  if (result) {
459  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(result);
460  assert(llvm::find(opOperands, &operand) != opOperands.end() &&
461  "operand and result do not match");
462  } else {
463  assert(checkConsistencyOnly &&
464  "result not provided, can only check consistency");
465  }
466 #endif // NDEBUG
467 
468  // Helper function to iterate on aliases of `root` and capture the reads.
469  auto getAliasingReads = [&](DenseSet<OpOperand *> &res, Value root) {
470  aliasInfo.applyOnAliases(root, [&](Value alias) {
471  for (auto &use : alias.getUses())
472  // Read to a value that aliases root.
473  if (bufferizesToMemoryRead(use))
474  res.insert(&use);
475  });
476  };
477 
478  // Helper function to iterate on aliases of `root` and capture the writes.
479  auto getAliasingInplaceWrites = [&](DenseSet<OpOperand *> &res, Value root) {
480  aliasInfo.applyOnAliases(root, [&](Value alias) {
481  for (auto &use : alias.getUses())
482  // Inplace write to a value that aliases root.
483  if (isInplaceMemoryWrite(use, aliasInfo))
484  res.insert(&use);
485  });
486  };
487 
488  // Collect reads and writes of all aliases of OpOperand and OpResult.
489  DenseSet<OpOperand *> usesRead, usesWrite;
490  getAliasingReads(usesRead, operand.get());
491  if (result)
492  getAliasingReads(usesRead, result);
493  getAliasingInplaceWrites(usesWrite, operand.get());
494  if (result)
495  getAliasingInplaceWrites(usesWrite, result);
496  if (!checkConsistencyOnly && bufferizesToMemoryWrite(operand))
497  usesWrite.insert(&operand);
498 
499  return hasReadAfterWriteInterference(usesRead, usesWrite, domInfo, aliasInfo);
500 }

502 /// Return true if bufferizing `opOperand` inplace with `opResult` would create
503 /// a write to a non-writable buffer.
504 static bool
505 wouldCreateWriteToNonWritableBuffer(OpOperand &opOperand, OpResult opResult,
506  const BufferizationAliasInfo &aliasInfo,
507  BufferizationState &state) {
508 #ifndef NDEBUG
509  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(opResult);
510  assert(llvm::find(opOperands, &opOperand) != opOperands.end() &&
511  "operand and result do not match");
512 #endif // NDEBUG
513 
514  // Certain buffers are not writeable:
515  // 1. A function bbArg that is not inplaceable or
516  // 2. A constant op.
517  assert(!aliasesNonWritableBuffer(opResult, aliasInfo, state) &&
518  "expected that opResult does not alias non-writable buffer");
519  bool nonWritable =
520  aliasesNonWritableBuffer(opOperand.get(), aliasInfo, state);
521  if (!nonWritable)
522  return false;
523 
524  // This is a problem only if the buffer is written to via some alias.
525  bool hasWrite = aliasesInPlaceWrite(opResult, aliasInfo) ||
526  aliasesInPlaceWrite(opOperand.get(), aliasInfo) ||
527  bufferizesToMemoryWrite(opOperand);
528  if (!hasWrite)
529  return false;
530 
531  LDBG("->the corresponding buffer is not writeable\n");
532  return true;
533 }

539 /// Determine if `operand` can be bufferized in-place with `result`.
540 static LogicalResult bufferizableInPlaceAnalysisImpl(
541  OpOperand &operand, OpResult result, BufferizationAliasInfo &aliasInfo,
542  BufferizationState &state, const DominanceInfo &domInfo) {
543 #ifndef NDEBUG
544  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(result);
545  assert(llvm::find(opOperands, &operand) != opOperands.end() &&
546  "operand and result do not match");
547 #endif // NDEBUG
548 
549  int64_t resultNumber = result.getResultNumber();
550  (void)resultNumber;
551  LDBG('\n');
552  LDBG("Inplace analysis for <- #" << resultNumber << " -> #"
553  << operand.getOperandNumber() << " in "
554  << printValueInfo(result) << '\n');
555 
556  bool foundInterference =
557  wouldCreateWriteToNonWritableBuffer(operand, result, aliasInfo, state) ||
558  wouldCreateReadAfterWriteInterference(operand, result, domInfo,
559  aliasInfo);
560 
561  if (foundInterference)
562  aliasInfo.bufferizeOutOfPlace(result);
563  else
564  aliasInfo.bufferizeInPlace(result, operand);
565 
566  LDBG("Done inplace analysis for result #" << resultNumber << '\n');
567 
568  return success();
569 }

571 /// Analyze the `ops` to determine which OpResults are inplaceable. Walk ops in
572 /// reverse and bufferize ops greedily. This is a good starter heuristic.
573 ///
574 /// Even if an op does not read or write, it may still create an alias when
575 /// bufferized in-place. An example of such ops is tensor.extract_slice.
576 ///
577 /// Rationale for bufferizing `%1 = tensor.extract_slice %0[...]` inplace:
578 ///
579 /// When bufferized out of place, an ExtractSliceOp lowers to alloc + copy. This
580 /// cannot change the flow of information for either the source or the
581 /// result buffers.
582 ///
583 /// When bufferized inplace, an ExtractSliceOp does not by itself create any
584 /// read or write from memory. Instead, it has the effect of merging the alias
585 /// sets of the source and the result buffers.
586 ///
587 /// An analysis is required to ensure inplace bufferization would not result in
588 /// RaW dependence violations.
589 static LogicalResult inPlaceAnalysis(SmallVector<Operation *> &ops,
590  BufferizationAliasInfo &aliasInfo,
591  BufferizationState &state,
592  const DominanceInfo &domInfo,
593  unsigned analysisFuzzerSeed = 0) {
594  if (analysisFuzzerSeed) {
595  // This is a fuzzer. For testing purposes only. Randomize the order in which
596  // operations are analyzed. The bufferization quality is likely worse, but
597  // we want to make sure that no assertions are triggered anywhere.
598  std::mt19937 g(analysisFuzzerSeed);
599  llvm::shuffle(ops.begin(), ops.end(), g);
600  }
601 
602  // Walk ops in reverse for better interference analysis.
603  for (Operation *op : reverse(ops))
604  for (OpOperand &opOperand : op->getOpOperands())
605  if (opOperand.get().getType().isa<TensorType>())
606  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op))
607  if (OpResult opResult = bufferizableOp.getAliasingOpResult(opOperand))
608  if (failed(bufferizableInPlaceAnalysisImpl(
609  opOperand, opResult, aliasInfo, state, domInfo)))
610  return failure();
611 
612  return success();
613 }
614 
615 /// Analyze all ops that are contained in `op`.
616 static LogicalResult inPlaceAnalysis(Operation *op,
617  BufferizationAliasInfo &aliasInfo,
618  BufferizationState &state,
619  const DominanceInfo &domInfo,
620  unsigned analysisFuzzerSeed = 0) {
621  // Collect ops so we can build our own reverse traversal.
622  SmallVector<Operation *> ops;
623  op->walk([&](Operation *op) {
624  // No tensors => no buffers.
625  if (none_of(op->getOperandTypes(), isaTensor) &&
626  none_of(op->getResultTypes(), isaTensor))
627  return;
628  ops.push_back(op);
629  });
630 
631  return inPlaceAnalysis(ops, aliasInfo, state, domInfo, analysisFuzzerSeed);
632 }

634 /// Analyze equivalence of tied OpResult/OpOperand pairs of the given ops.
635 static void equivalenceAnalysis(SmallVector<Operation *> &ops,
636  BufferizationAliasInfo &aliasInfo) {
637  for (Operation *op : ops)
638  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op))
639  for (OpResult opResult : op->getOpResults())
640  if (opResult.getType().isa<TensorType>())
641  if (aliasInfo.isInPlace(opResult)) {
642  SmallVector<OpOperand *> opOperands =
643  bufferizableOp.getAliasingOpOperand(opResult);
644  if (!opOperands.empty())
645  if (bufferizableOp.bufferRelation(opResult, aliasInfo) ==
646  BufferRelation::Equivalent)
647  for (OpOperand *opOperand : opOperands)
648  aliasInfo.unionEquivalenceClasses(opResult, opOperand->get());
649  }
650 }
651 
652 /// Analyze equivalence of tied OpResult/OpOperand pairs of all ops contained
653 /// in `op`.
654 static void equivalenceAnalysis(Operation *op,
655  BufferizationAliasInfo &aliasInfo) {
656  // Traverse ops in PostOrder: Nested ops first, then enclosing ops.
657  SmallVector<Operation *> ops;
658  op->walk<WalkOrder::PostOrder>([&](Operation *op) {
659  // No tensors => no buffers.
660  if (none_of(op->getResultTypes(), isaTensor))
661  return;
662  ops.push_back(op);
663  });
664 
665  equivalenceAnalysis(ops, aliasInfo);
666 }

668 /// Assert that the current bufferization decisions are consistent.
669 static LogicalResult
670 checkAliasInfoConsistency(FuncOp funcOp, const DominanceInfo &domInfo,
671  const BufferizationAliasInfo &aliasInfo) {
672  Operation *inconsistentOp = nullptr;
673  WalkResult walkResult = funcOp.walk([&](Operation *op) {
674  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op))
675  for (OpOperand &opOperand : op->getOpOperands())
676  if (opOperand.get().getType().isa<TensorType>()) {
677  OpResult opResult = bufferizableOp.getAliasingOpResult(opOperand);
678  if (wouldCreateReadAfterWriteInterference(
679  opOperand, opResult, domInfo, aliasInfo,
680  /*checkConsistencyOnly=*/true)) {
681  // This error can happen for two reasons. Either the input IR
682  // already has a read-after-write conflict. Or certain
683  // "mustBufferizeInPlace" interface methods are implemented
684  // incorrectly.
685  inconsistentOp = op;
686  return WalkResult::interrupt();
687  }
688  }
689  return WalkResult::advance();
690  });
691 
692  if (walkResult.wasInterrupted())
693  // This can currently happen in one situation: When a tensor is passed into
694  // a ToMemrefOp and read by another op consecutively. ToMemrefOps are
695  // currently handled conservatively. Once a tensor is passed into a
696  // ToMemrefOp, it may longer be read.
697  return inconsistentOp->emitError("input IR has RaW conflict");
698  return success();
699 }

712 LogicalResult mlir::linalg::comprehensive_bufferize::runComprehensiveBufferize(
713  FuncOp funcOp, const BufferizationOptions &options,
714  BufferizationState &state, const PostAnalysisStepList &extraSteps) {
715 
716  DominanceInfo domInfo(funcOp);
717  BufferizationAliasInfo &aliasInfo = state.aliasInfo;
718 
719  if (funcOp.body().empty())
720  return success();
721 
722  if (failed(checkAliasInfoConsistency(funcOp, domInfo, aliasInfo)))
723  return failure();
724 
725  // If the analysis fails, just return.
726  Operation *op = funcOp.getOperation();
727  if (failed(inPlaceAnalysis(op, aliasInfo, state, domInfo,
728  options.analysisFuzzerSeed)))
729  return failure();
730  equivalenceAnalysis(op, aliasInfo);
731 
732  auto runPostAnalysisSteps = [&](const PostAnalysisStepList &steps) {
733  for (const std::unique_ptr<PostAnalysisStep> &step : steps) {
734  SmallVector<Operation *> newOps;
735  if (failed(step->run(funcOp, state, newOps)))
736  return failure();
737  // Analyze ops that were created by the PostAnalysisStep.
738  if (failed(inPlaceAnalysis(newOps, aliasInfo, state, domInfo)))
739  return failure();
740  equivalenceAnalysis(newOps, aliasInfo);
741  }
742  return success();
743  };
744 
745  if (failed(runPostAnalysisSteps(extraSteps)))
746  return failure();
747  if (failed(runPostAnalysisSteps(options.postAnalysisSteps)))
748  return failure();
749 
750  // Annotate operations if we only want to report the analysis.
751  if (options.testAnalysisOnly) {
752  annotateOpsWithBufferizationMarkers(funcOp, aliasInfo);
753  return success();
754  }
755 
756  // Bufferize all ops in funcOp.
757  OpBuilder b(funcOp.getContext());
758  auto bufferizableOp =
759  dyn_cast<BufferizableOpInterface>(funcOp.getOperation());
760  assert(bufferizableOp && "must use ModuleBufferization");
761  if (failed(bufferizableOp.bufferize(b, state)))
762  return failure();
763 
764  // Erase all obsolete ops.
765  state.eraseObsoleteOps();
766 
767  return success();
768 }
