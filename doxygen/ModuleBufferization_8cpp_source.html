52 /// Return the unique ReturnOp that terminates `funcOp`.
53 /// Return nullptr if there is no such unique ReturnOp.
54 static ReturnOp getAssumedUniqueReturnOp(FuncOp funcOp) {
55  ReturnOp returnOp;
56  for (Block &b : funcOp.body()) {
57  if (auto candidateOp = dyn_cast<ReturnOp>(b.getTerminator())) {
58  if (returnOp)
59  return nullptr;
60  returnOp = candidateOp;
61  }
62  }
63  return returnOp;
64 }

67 /// Store function BlockArguments that are equivalent to a returned value in
68 /// ModuleBufferizationState.
69 struct EquivalentFuncOpBBArgsAnalysis : public PostAnalysisStep {
70  /// Annotate IR with the results of the analysis. For testing purposes only.
71  static void annotateReturnOp(OpOperand &returnVal, BlockArgument bbArg) {
72  const char *kEquivalentArgsAttr = "__equivalent_func_args__";
73  Operation *op = returnVal.getOwner();
74 
75  SmallVector<int64_t> equivBbArgs;
76  if (op->hasAttr(kEquivalentArgsAttr)) {
77  auto attr = op->getAttr(kEquivalentArgsAttr).cast<ArrayAttr>();
78  equivBbArgs = llvm::to_vector<4>(llvm::map_range(attr, [](Attribute a) {
79  return a.cast<IntegerAttr>().getValue().getSExtValue();
80  }));
81  } else {
82  equivBbArgs.append(op->getNumOperands(), -1);
83  }
84  equivBbArgs[returnVal.getOperandNumber()] = bbArg.getArgNumber();
85 
86  OpBuilder b(op->getContext());
87  op->setAttr(kEquivalentArgsAttr, b.getI64ArrayAttr(equivBbArgs));
88  }

90  LogicalResult run(FuncOp funcOp, BufferizationState &state,
91  SmallVector<Operation *> &newOps) override {
92  ModuleBufferizationState &moduleState = getModuleBufferizationState(state);
93 
94  // Support only single return-terminated block in the function.
95  ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
96  assert(returnOp && "expected func with single return op");
97 
98  for (OpOperand &returnVal : returnOp->getOpOperands())
99  if (returnVal.get().getType().isa<RankedTensorType>())
100  for (BlockArgument bbArg : funcOp.getArguments())
101  if (bbArg.getType().isa<RankedTensorType>())
102  if (state.aliasInfo.areEquivalentBufferizedValues(returnVal.get(),
103  bbArg)) {
104  moduleState
105  .equivalentFuncArgs[funcOp][returnVal.getOperandNumber()] =
106  bbArg.getArgNumber();
107  if (state.options.testAnalysisOnly)
108  annotateReturnOp(returnVal, bbArg);
109  }
110 
111  return success();
112  }

145 /// Return the FunctionType with `argumentTypes` and `resultTypes` where each
146 /// tensor is replaced by the corresponding buffer type.
147 /// In order for all the callers to agree, this *must* bufferize to the most
148 /// dynamic buffer type supported.
149 /// A later pass across all CallOps in the module can decide whether to simplify
150 /// the types of to version according to some cost model.
151 static FunctionType getBufferizedFunctionType(MLIRContext *ctx,
152  TypeRange argumentTypes,
153  TypeRange resultTypes) {
154  auto rewrite = [](Type t) -> Type {
155  // TODO: non-zero address space.
156  // TODO: layout information if relevant.
157  if (auto rankedTensorType = t.dyn_cast<RankedTensorType>())
158  return getDynamicMemRefType(rankedTensorType);
159  if (auto tensorType = t.dyn_cast<TensorType>())
160  return getContiguousOrUnrankedMemRefType(tensorType);
161  return t;
162  };
163  auto argTypes = llvm::to_vector<4>(llvm::map_range(argumentTypes, rewrite));
164  auto retTypes = llvm::to_vector<4>(llvm::map_range(resultTypes, rewrite));
165  return FunctionType::get(ctx, argTypes, retTypes);
166 }

168 /// If an entry for `funcOp` is available in `bufferizedFunctionTypes`, return
169 /// it. Otherwise, construct a new entry based on `argumentTypes` and
170 /// `resultTypes`.
171 // TODO: improve the layering.
172 static FunctionType getOrCreateBufferizedFunctionType(
173  FuncOp funcOp, TypeRange argumentTypes, TypeRange resultTypes,
174  DenseMap<FuncOp, FunctionType> &bufferizedFunctionTypes) {
175  auto it = bufferizedFunctionTypes.find(funcOp);
176  if (it != bufferizedFunctionTypes.end())
177  return it->second;
178 
179  auto it2 = bufferizedFunctionTypes.try_emplace(
180  funcOp, getBufferizedFunctionType(funcOp.getContext(), argumentTypes,
181  resultTypes));
182  LDBG("FT: " << funcOp.getType() << " -> " << it2.first->second << "\n");
183  return it2.first->second;

186 /// Gather equivalence info of CallOps.
187 /// Note: This only adds new equivalence info if `funcOp` was already analyzed.
188 // TODO: This does not handle cyclic function call graphs etc.
189 static void equivalenceAnalysis(FuncOp funcOp,
190  BufferizationAliasInfo &aliasInfo,
191  ModuleBufferizationState &moduleState) {
192  funcOp->walk([&](CallOp callOp) {
193  FuncOp calledFunction = getCalledFunction(callOp);
194  assert(calledFunction && "could not retrieved called FuncOp");
195 
196  // No equivalence info available for the called function.
197  if (!moduleState.equivalentFuncArgs.count(calledFunction))
198  return WalkResult::skip();
199 
200  for (auto it : moduleState.equivalentFuncArgs[calledFunction]) {
201  int64_t returnIdx = it.first;
202  int64_t bbargIdx = it.second;
203  Value returnVal = callOp.getResult(returnIdx);
204  Value argVal = callOp->getOperand(bbargIdx);
205  aliasInfo.unionEquivalenceClasses(returnVal, argVal);
206  }
207 
208  return WalkResult::advance();
209  });
210 }

212 /// Rewrite the `funcOp` arguments analysis return values and terminator into
213 /// buffer form (using the canonical memref layout for now), according to the
214 /// inPlace-bufferizable information of the function arguments.
215 ///
216 /// This relies on a buffer equivalence analysis of each return operand. When a
217 /// result buffer is equivalent to a BlockArgument of `funcOp`, it can be
218 /// dropped from the return values and becomes inplaceable at all callers. This
219 /// assumes all CallOp perform the necessary work to clone operands so as to
220 /// make them inplaceable. Reliance on this logic will need to be relaxed in the
221 /// future.
222 ///
223 /// Note: Returning a memref currently fails bufferization. If such memrefs
224 /// originate from an op with an Alloc effect, they could be hoisted in the
225 /// future.
226 static LogicalResult bufferizeFuncOpBoundary(FuncOp funcOp,
227  BufferizationState &state) {
228  LLVM_DEBUG(DBGS() << "Begin bufferizeFuncOpBoundary:\n" << funcOp << "\n");
229  ModuleBufferizationState &moduleState = getModuleBufferizationState(state);
230 
231  // If nothing to do then we are done.
232  if (!llvm::any_of(funcOp.getType().getInputs(), isaTensor) &&
233  !llvm::any_of(funcOp.getType().getResults(), isaTensor))
234  return success();
235 
236  // Get the bufferized FunctionType for funcOp or construct it if not yet
237  // available.
238  // TODO: Atm we have 3 cases:
239  // 1. if a function is called from within the Module, it must have bufferized
240  // to inplaceable tensor results.
241  // 2. if it is bodiless, it must have bufferized and is not allowed to have
242  // result tensors.
243  // 3. if it is not called internally, it still must bufferize to inplaceable
244  // tensor results and we construct it now (e.g. top-level function called
245  // externally).
246  // -> Figure out a better layering.
247  TypeRange resultTypes;
248 
249  // Corner case: Bodiless FuncOp
250  // ============================
251  // The body of such functions is assumed opaque and we can't know the
252  // bufferization contract they want to enforce atm.
253  // As a consequence, only support functions that don't return any tensor atm.
254  if (funcOp.getBody().empty()) {
255  if (llvm::any_of(funcOp.getType().getResults(), isaTensor))
256  return funcOp->emitError() << "cannot bufferize bodiless function that "
257  << "returns a tensor";
258  FunctionType bufferizedFuncType = getOrCreateBufferizedFunctionType(
259  funcOp, funcOp.getType().getInputs(), TypeRange{},
260  moduleState.bufferizedFunctionTypes);
261  funcOp.setType(bufferizedFuncType);
262  LLVM_DEBUG(DBGS() << "End bufferizeFuncOpBoundary no fun body: " << funcOp);
263  return success();
264  }
265 
266  // Support only single return-terminated block in the function.
267  ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
268  assert(returnOp && "expected func with single return op");
269 
270  // 1. For each FuncOp result, keep track of which inplace argument it reuses.
271  SmallVector<Value> returnValues;
272  for (OpOperand &returnOperand : returnOp->getOpOperands()) {
273  Value returnVal = returnOperand.get();
274 
275  // If not a renturn tensor type just forward it.
276  if (!returnVal.getType().isa<RankedTensorType>()) {
277  returnValues.push_back(returnVal);
278  continue;
279  }
280 
281  // If return operand is equivalent to some bbArg, no need to return it.
282  if (moduleState.equivalentFuncArgs[funcOp].count(
283  returnOperand.getOperandNumber()))
284  continue;
285 
286  // Cast values at the call site if necessary.
287  returnValues.push_back(getNonCastedValue(state.lookupBuffer(returnVal)));
288  }
289 
290  // 2. Rewrite the terminator without the inPlace bufferizable values.
291  ValueRange retValues{returnValues};
292  FunctionType bufferizedFuncType = getOrCreateBufferizedFunctionType(
293  funcOp, funcOp.getType().getInputs(), retValues.getTypes(),
294  moduleState.bufferizedFunctionTypes);
295  OpBuilder b(returnOp);
296  b.create<ReturnOp>(returnOp.getLoc(), returnValues);
297  returnOp->erase();
298 
299  // 3. Rewrite the bbArgs.
300  // Iterate on the original `numArgs` and replace them in order.
301  // This guarantees the argument order still matches after the rewrite.
302  Block &frontBlock = funcOp.body().front();
303  unsigned numArgs = frontBlock.getNumArguments();
304  for (unsigned idx = 0; idx < numArgs; ++idx) {
305  auto bbArg = frontBlock.getArgument(0);
306  auto tensorType = bbArg.getType().dyn_cast<TensorType>();
307  // Non-tensor types are just forwarded.
308  if (!tensorType) {
309  frontBlock.addArgument(bbArg.getType());
310  bbArg.replaceAllUsesWith(frontBlock.getArguments().back());
311  frontBlock.eraseArgument(0);
312  continue;
313  }
314 
315  // Get the buffer type from the bufferized function type.
316  Type memrefType = bufferizedFuncType.getInput(idx);
317  Value memref = frontBlock.addArgument(memrefType);
318  OpBuilder b(funcOp->getContext());
319  b.setInsertionPointToStart(&frontBlock);
320  // Replace all uses of bbArg through a ToMemRefOp by a memref::CastOp.
321  for (auto &use : llvm::make_early_inc_range(bbArg.getUses())) {
322  if (auto toMemrefOp =
323  dyn_cast<bufferization::ToMemrefOp>(use.getOwner())) {
324  auto castOp = b.create<memref::CastOp>(
325  funcOp.getLoc(), toMemrefOp.memref().getType(), memref);
326  toMemrefOp.memref().replaceAllUsesWith(castOp);
327  }
328  }
329  // Replace all remaining uses by a to_tensor.
330  if (!bbArg.use_empty()) {
331  auto toTensorOp =
332  b.create<bufferization::ToTensorOp>(funcOp.getLoc(), memref);
333  bbArg.replaceAllUsesWith(toTensorOp);
334  }
335  frontBlock.eraseArgument(0);
336  // TODO: add support to erase aliasInfo entries if deemed necessary.
337  }
338 
339  // 4. Rewrite the FuncOp type to buffer form.
340  funcOp.setType(bufferizedFuncType);
341 
342  LLVM_DEBUG(DBGS() << "End bufferizeFuncOpBoundary:\n" << funcOp);
343 
344  return success();
345 }

347 /// Store all functions of the `moduleOp` in `orderedFuncOps`, sorted by
348 /// callee-caller order (i.e. callees without callers first).
349 /// Store the map of FuncOp to all its callers in `callerMap`.
350 /// Return `failure()` if a cycle of calls is detected or if we are unable to
351 /// retrieve the called FuncOp from any CallOpInterface.
352 static LogicalResult
353 getFuncOpsOrderedByCalls(ModuleOp moduleOp,
354  SmallVectorImpl<FuncOp> &orderedFuncOps,
355  DenseMap<FuncOp, DenseSet<Operation *>> &callerMap) {
356  // For each FuncOp, the set of functions called by it (i.e. the union of
357  // symbols of all nested CallOpInterfaceOp).
358  DenseMap<FuncOp, DenseSet<FuncOp>> calledBy;
359  // For each FuncOp, the number of CallOpInterface it contains.
360  DenseMap<FuncOp, unsigned> numberCallOpsContainedInFuncOp;
361  WalkResult res = moduleOp.walk([&](FuncOp funcOp) -> WalkResult {
362  if (!funcOp.body().empty()) {
363  ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
364  if (!returnOp)
365  return funcOp->emitError()
366  << "cannot bufferize a FuncOp with tensors and "
367  "without a unique ReturnOp";
368  }
369 
370  numberCallOpsContainedInFuncOp[funcOp] = 0;
371  return funcOp.walk([&](CallOpInterface callOp) -> WalkResult {
372  // Only support CallOp for now.
373  if (!isa<CallOp>(callOp.getOperation()))
374  return callOp->emitError() << "expected a CallOp";
375  FuncOp calledFunction = getCalledFunction(callOp);
376  assert(calledFunction && "could not retrieved called FuncOp");
377  auto it = callerMap.try_emplace(calledFunction, DenseSet<Operation *>{});
378  it.first->getSecond().insert(callOp);
379  if (calledBy[calledFunction].count(funcOp) == 0) {
380  calledBy[calledFunction].insert(funcOp);
381  numberCallOpsContainedInFuncOp[funcOp]++;
382  }
383  return WalkResult::advance();
384  });
385  });
386  if (res.wasInterrupted())
387  return failure();
388  // Iteratively remove function operation that do not call any of the
389  // functions remaining in the callCounter map and add them to the worklist.
390  while (!numberCallOpsContainedInFuncOp.empty()) {
391  auto it = llvm::find_if(numberCallOpsContainedInFuncOp,
392  [](auto entry) { return entry.getSecond() == 0; });
393  if (it == numberCallOpsContainedInFuncOp.end())
394  return moduleOp.emitOpError(
395  "expected callgraph to be free of circular dependencies.");
396  orderedFuncOps.push_back(it->getFirst());
397  for (auto callee : calledBy[it->getFirst()])
398  numberCallOpsContainedInFuncOp[callee]--;
399  numberCallOpsContainedInFuncOp.erase(it);
400  }
401  return success();
402 }

404 static void
405 foreachCaller(const DenseMap<FuncOp, DenseSet<Operation *>> &callerMap,
406  FuncOp callee, llvm::function_ref<void(Operation *)> doit) {
407  auto itCallers = callerMap.find(callee);
408  if (itCallers == callerMap.end())
409  return;
410  for (Operation *caller : itCallers->second)
411  doit(caller);
412 }

414 /// Postprocess the linalg.buffer_layout annotation across function boundaries.
415 /// This is a purely mechanical process that may later become part of a
416 /// separate pass with its own layout assignment heuristic.
417 static void layoutPostProcessing(ModuleOp moduleOp) {
418  SmallVector<FuncOp> orderedFuncOps;
419  DenseMap<FuncOp, DenseSet<Operation *>> callerMap;
420  auto res = getFuncOpsOrderedByCalls(moduleOp, orderedFuncOps, callerMap);
421  (void)res;
422  assert(succeeded(res) && "unexpected getFuncOpsOrderedByCalls failure");
423 
424  for (FuncOp funcOp : orderedFuncOps) {
425  DenseMap<Operation *, SmallVector<Value>> operandsPerCaller;
426  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
427  operandsPerCaller.try_emplace(caller, SmallVector<Value>());
428  });
429 
430  SmallVector<Type> argumentTypes;
431  // Iterate on each function argument and check it it was marked with a
432  // desired layout.
433  for (const auto &it : llvm::enumerate(funcOp.getType().getInputs())) {
434  int argNumber = it.index();
435  Type inputType = it.value();
436  auto memrefType = inputType.dyn_cast<MemRefType>();
437  auto layoutAttr = funcOp.getArgAttrOfType<AffineMapAttr>(
438  argNumber, BufferizableOpInterface::kBufferLayoutAttrName);
439  AffineMap desiredLayoutMap =
440  layoutAttr ? layoutAttr.getValue() : AffineMap();
441  AffineMap currentLayoutMap =
442  memrefType ? getStridedLinearLayoutMap(memrefType) : AffineMap();
443  if (!memrefType || !layoutAttr || desiredLayoutMap == currentLayoutMap) {
444  argumentTypes.push_back(inputType);
445  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
446  operandsPerCaller.find(caller)->getSecond().push_back(
447  caller->getOperand(argNumber));
448  });
449  continue;
450  }
451 
452  // Compute the buffer type with desired layout and add to input argument
453  // types.
454  MemRefType desiredMemrefType = MemRefType::get(
455  memrefType.getShape(), memrefType.getElementType(), desiredLayoutMap);
456  argumentTypes.push_back(desiredMemrefType);
457 
458  // If funcOp's body is not empty, change the bbArg type and propagate.
459  if (!funcOp.body().empty()) {
460  BlockArgument bbArg = funcOp.getArgument(argNumber);
461  bbArg.setType(desiredMemrefType);
462  OpBuilder b(bbArg.getContext());
463  b.setInsertionPointToStart(bbArg.getOwner());
464  // Cast back to the original memrefType and let it canonicalize.
465  Value cast =
466  b.create<memref::CastOp>(funcOp.getLoc(), memrefType, bbArg);
467  bbArg.replaceAllUsesExcept(cast, cast.getDefiningOp());
468  }
469 
470  // Cast to desired buffer type on all callers to `funcOp`.
471  // TODO: on the callee side, this may even have to trigger a copy to
472  // change the layout. For now let the memref::CastOp fail to verify in
473  // such cases.
474  auto castArg = [&](Operation *caller) {
475  OpBuilder b(caller);
476  Value newOperand = b.create<memref::CastOp>(
477  funcOp.getLoc(), desiredMemrefType, caller->getOperand(argNumber));
478  operandsPerCaller.find(caller)->getSecond().push_back(newOperand);
479  };
480  foreachCaller(callerMap, funcOp, castArg);
481  }
482 
483  // Set operands with cast buffer on all callers to `funcOp`.
484  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
485  caller->setOperands(operandsPerCaller.lookup(caller));
486  });
487 
488  // Finally set the funcOp type to update the arguments.
489  auto newFuncType = FunctionType::get(moduleOp.getContext(), argumentTypes,
490  funcOp.getType().getResults());
491  funcOp.setType(newFuncType);
492  }

517  /// In a first approximation, all the function arguments of a FuncOp are
518  /// marked inplaceable. For now, it is the responsibility of the `callOp`
519  /// bufferization to allow FuncOp that are inplaceable to write inPlace.
520  LogicalResult bufferize(Operation *op, OpBuilder &b,
521  BufferizationState &state) const {
522  CallOp callOp = cast<CallOp>(op);
523  FuncOp funcOp = getCalledFunction(callOp);
524  assert(isa<CallOp>(callOp.getOperation()) && funcOp &&
525  "expected Callop to a FuncOp");
526  ModuleBufferizationState &moduleState = getModuleBufferizationState(state);
527 
528  // 1. Filter return types:
529  // - if the callee is bodiless / external, we cannot inspect it and we
530  // cannot assume anything. We can just assert that it does not return a
531  // tensor as this would have to bufferize to "return a memref", whose
532  // semantics is ill-defined.
533  // - if the callee has a body, we perform inter-procedural equivalence
534  // analysis. When successful, a result folds onto an operand. When
535  // unsuccessful, additional work is needed (TODO) to either:
536  // * hoist a result into an inplaceable operand or
537  // * devise a better representation to truly return a buffer.
538  SmallVector<Type> resultTypes;
539  if (funcOp.body().empty()) {
540  if (llvm::any_of(funcOp.getType().getResults(), isaTensor))
541  return callOp->emitError()
542  << "cannot bufferize bodiless function that returns a tensor";
543  } else {
544  ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
545  assert(returnOp && "expected func with single return op");
546 
547  // For each FuncOp result, keep track of which inplace argument it reuses.
548  for (OpOperand &returnOperand : returnOp->getOpOperands()) {
549  Type returnType = returnOperand.get().getType();
550  if (!isaTensor(returnType)) {
551  resultTypes.push_back(returnType);
552  continue;
553  }
554 
555  // If return operand is equivalent to some bbArg, no need to return it.
556  if (moduleState.equivalentFuncArgs[funcOp].count(
557  returnOperand.getOperandNumber())) {
558  int64_t idx =
559  moduleState
560  .equivalentFuncArgs[funcOp][returnOperand.getOperandNumber()];
561  Value oldRes = callOp->getResult(returnOperand.getOperandNumber());
562  Value buffer = state.lookupBuffer(callOp->getOperand(idx));
563  // Add CallOp operand/result equivalence: this is interprocedural
564  // info.
565  state.mapBuffer(oldRes, buffer);
566  // Add a ToTensorOp to kill all uses of the CallOp return.
567  // Replace all uses of the CallOp results so we can erase the CallOp.
568  // This ToTensorOp must fold/DCE away or bufferization should be
569  // considered failed.
570  Value toTensorOp =
571  b.create<bufferization::ToTensorOp>(callOp.getLoc(), buffer);
572  oldRes.replaceAllUsesWith(toTensorOp);
573  // Add new op equivalence info.
574  state.mapBuffer(toTensorOp, buffer);
575  continue;
576  }
577 
578  resultTypes.push_back(returnType);
579  }
580  }
581 
582  // 2. Compute bufferized FunctionType.
583  SmallVector<Type> argumentTypes{callOp->getOperandTypes()};
584  // Get the bufferized FunctionType for funcOp or construct it if not yet
585  // available.
586  FunctionType bufferizedFuncType =
587  getOrCreateBufferizedFunctionType(funcOp, argumentTypes, resultTypes,
588  moduleState.bufferizedFunctionTypes);
589 
590  // 3. Rewrite tensor operands as memrefs based on `bufferizedFuncType`.
591  SmallVector<Value> newOperands;
592  newOperands.reserve(callOp->getNumOperands());
593  for (OpOperand &opOperand : callOp->getOpOperands()) {
594  Value tensorOperand = opOperand.get();
595  // Non-tensor operands are just copied.
596  if (!tensorOperand.getType().isa<TensorType>()) {
597  newOperands.push_back(tensorOperand);
598  continue;
599  }
600 
601  // Tensor operands are guaranteed to have been buferized.
602  int64_t idx = opOperand.getOperandNumber();
603  Value buffer = state.lookupBuffer(tensorOperand);
604 
605  // Caller / callee type mistmatch is handled with a CastOp.
606  auto memRefType = bufferizedFuncType.getInput(idx);
607  // Since we don't yet have a clear layout story, buffer_cast may
608  // conservatively turn tensors into more dynamic memref than necessary.
609  // If the memref type of the callee fails, introduce an extra memref.cast
610  // that will either canonicalize away or fail compilation until we can do
611  // something better.
612  if (buffer.getType() != memRefType) {
613  Value castBuffer =
614  b.create<memref::CastOp>(callOp.getLoc(), memRefType, buffer);
615  // Add new op equivalence info.
616  state.mapBuffer(tensorOperand, castBuffer);
617  buffer = castBuffer;
618  }
619  newOperands.push_back(buffer);
620  }
621 
622  // 4. Create the new CallOp.
623  Operation *newCallOp = b.create<CallOp>(callOp.getLoc(), funcOp.sym_name(),
624  resultTypes, newOperands);
625  newCallOp->setAttrs(callOp->getAttrs());
626 
627  // 5. Delete the op at the end of bufferization.
628  state.markOpObsolete(callOp);
629 
630  return success();
631  }
632 };

670  : public BufferizableOpInterface::ExternalModel<FuncOpInterface, FuncOp> {
671  LogicalResult bufferize(Operation *op, OpBuilder &b,
672  BufferizationState &state) const {
673  auto funcOp = cast<FuncOp>(op);
674  b.setInsertionPointToStart(&funcOp.body().front());
675 
676  // Create BufferCastOps for function args.
677  for (auto bbArg : funcOp.getArguments()) {
678  auto tensorType = bbArg.getType().dyn_cast<TensorType>();
679  if (!tensorType)
680  continue;
681  auto rankedTensorType = tensorType.dyn_cast<RankedTensorType>();
682  // Cast the tensor to the most dynamic buffer possible. Further
683  // canonicalizations will clean up.
684  Type memRefType = rankedTensorType
685  ? getDynamicMemRefType(rankedTensorType)
686  : getContiguousOrUnrankedMemRefType(tensorType);
687  Value bufferCast = b.create<bufferization::ToMemrefOp>(funcOp.getLoc(),
688  memRefType, bbArg);
689  state.mapBuffer(bbArg, bufferCast);
690  }
691 
692  // Bufferize function body.
693  return comprehensive_bufferize::bufferize(&funcOp.body(), state);
694  }

698  auto funcOp = cast<FuncOp>(op);

702

704  // =========================
705  // If the function is called, we can allocate on the caller side which lets
706  // us force inplace arguments at function boundaries.
707  // TODO: do not rely on this behavior.
708  if (moduleState.callerMap.find(funcOp) != moduleState.callerMap.end())

715  if (inplaceAttr && inplaceAttr.getValue())
716  return true;

719  return false;
720  }

755 LogicalResult mlir::linalg::comprehensive_bufferize::runComprehensiveBufferize(
756  ModuleOp moduleOp, const BufferizationOptions &options) {
757  BufferizationState state(moduleOp, options);
758  ModuleBufferizationState &moduleState = getModuleBufferizationState(state);
759  BufferizationAliasInfo &aliasInfo = state.aliasInfo;
760 
761  if (failed(getFuncOpsOrderedByCalls(moduleOp, moduleState.orderedFuncOps,
762  moduleState.callerMap)))
763  return failure();
764 
765  // Interestingly, all function args that are not visible outside of a module
766  // can be fully bufferized inplace by guaranteeing the CallOp is bufferized
767  // inplace. Therefore, we just bufferize funcOp as if none of its results were
768  // inplaceable, detect which operands are cloned internally and decide what to
769  // do at call sites.
770  for (FuncOp funcOp : moduleState.orderedFuncOps) {
771  // No body => no analysis.
772  if (funcOp.body().empty())
773  continue;
774 
775  // Register extra post analysis steps. These cannot be stored in `options`
776  // because `options` is immutable.
777  PostAnalysisStepList extraSteps;
778  extraSteps.emplace_back(std::make_unique<EquivalentFuncOpBBArgsAnalysis>());
779 
780  // Gather equivalence info for CallOps.
781  equivalenceAnalysis(funcOp, aliasInfo, moduleState);
782 
783  // Analyze and bufferize funcOp.
784  if (failed(runComprehensiveBufferize(funcOp, options, state, extraSteps)))
785  return failure();
786 
787  // Add annotations to function arguments.
788  if (options.testAnalysisOnly)
789  annotateOpsWithBufferizationMarkers(funcOp, state);
790  }
791 
792  if (options.testAnalysisOnly)
793  return success();
794 
795  for (FuncOp funcOp : moduleState.orderedFuncOps) {
796  // Note: It would be good to apply cleanups here but we cannot as aliasInfo
797  // would be invalidated.
798  if (failed(bufferizeFuncOpBoundary(funcOp, state)))
799  return failure();
800 
801  if (!options.allowReturnMemref &&
802  llvm::any_of(funcOp.getType().getResults(), [](Type t) {
803  return t.isa<MemRefType, UnrankedMemRefType>();
804  })) {
805  funcOp->emitError("memref return type is unsupported");
806  return failure();
807  }
808  }
809 
810  // Perform a post-processing pass of layout modification at function boundary
811  // according to the kBufferLayoutAttrName.
812  layoutPostProcessing(moduleOp);
813 
814  // Post-pass cleanup of inplaceable and buffer_layout attributes.
815  moduleOp.walk([&](FuncOp op) {
816  for (BlockArgument bbArg : op.getArguments())
817  removeBufferizationFuncArguments(bbArg);
818  });
819 
820  return success();
821 }
