55 // This converts the TOSA ApplyScale operator to a set of StandardOps ops,
56 // using 64-bit operations to perform the necessary multiply, bias, and shift.
57 // Multiple types are used to use minimal bit width operations.
58 class ApplyScaleOpConverter : public OpRewritePattern<tosa::ApplyScaleOp> {
59 public:
60  using OpRewritePattern<tosa::ApplyScaleOp>::OpRewritePattern;
61 
62  LogicalResult matchAndRewrite(tosa::ApplyScaleOp op,
63  PatternRewriter &rewriter) const final {
64  Location loc = op.getLoc();
65  Value value32 = op.value();
66  Value multiplier32 = op.multiplier();
67  Value shift8 = op.shift();
68  bool doubleRound = op.double_round();
69  Type inType = op.value().getType();
70 
71  Value one8 = rewriter.create<arith::ConstantOp>(
72  loc, rewriter.getIntegerAttr(rewriter.getIntegerType(8), 1));
73  Value one64 = rewriter.create<arith::ConstantOp>(
74  loc, rewriter.getIntegerAttr(rewriter.getI64Type(), 1));
75 
76  Value shiftSubOne8 = rewriter.create<arith::SubIOp>(loc, shift8, one8);
77 
78  // The rounding value semantics below equate to the following code:
79  // int64_t round = 1 << (shift - 1);
80  // if (double_round) {
81  // if (shift > 31 && value >= 0) round += 1<<30;
82  // if (shift > 31 && value < 0) round -= 1<<30;
83  // }
84  //
85  // Note that minimal bitwidth operators are used throughout the block.
86 
87  Value round64 = rewriter.create<arith::ShLIOp>(
88  loc, one64,
89  rewriter.create<arith::ExtSIOp>(loc, rewriter.getI64Type(),
90  shiftSubOne8));
91 
92  // Double rounding is performing a round operation before the shift
93  if (doubleRound) {
94  Value one32 = rewriter.create<arith::ConstantOp>(
95  loc, rewriter.getIntegerAttr(rewriter.getI32Type(), 1));
96  Value shift32 =
97  rewriter.create<arith::ExtSIOp>(loc, rewriter.getI32Type(), shift8);
98  Value thirty32 = rewriter.create<arith::ConstantOp>(
99  loc, rewriter.getIntegerAttr(rewriter.getI32Type(), 30));
100 
101  Value shiftThirty32 =
102  rewriter.create<arith::ShLIOp>(loc, one32, thirty32);
103  Value shiftThirty64 = rewriter.create<arith::ExtSIOp>(
104  loc, rewriter.getI64Type(), shiftThirty32);
105 
106  // Round value needs to with be added or subtracted depending on the sign
107  // of the input value.
108  Value roundAdd64 =
109  rewriter.create<arith::AddIOp>(loc, round64, shiftThirty64);
110  Value roundSub64 =
111  rewriter.create<arith::SubIOp>(loc, round64, shiftThirty64);
112 
113  Value zero32 =
114  rewriter.create<arith::ConstantOp>(loc, rewriter.getZeroAttr(inType));
115  Value valueGreaterThanZero = rewriter.create<arith::CmpIOp>(
116  loc, arith::CmpIPredicate::sge, value32, zero32);
117 
118  Value doubleRound64 = rewriter.create<mlir::SelectOp>(
119  loc, valueGreaterThanZero, roundAdd64, roundSub64);
120 
121  // We only perform double rounding if the shift value is greater than 32.
122  Value thirtyTwo32 = rewriter.create<arith::ConstantOp>(
123  loc, rewriter.getIntegerAttr(rewriter.getI32Type(), 32));
124  Value shiftGreaterThanThirtyTwo = rewriter.create<arith::CmpIOp>(
125  loc, arith::CmpIPredicate::sge, shift32, thirtyTwo32);
126  round64 = rewriter.create<mlir::SelectOp>(loc, shiftGreaterThanThirtyTwo,
127  doubleRound64, round64);
128  }
129 
130  // The computation below equates to the following pseudocode:
131  // int64_t result = (int64_t)value * multiplier + round;
132  // result = result >> shift;
133  //
134  // Note that multiply and shift need to be perform in i64 to preserve bits.
135 
136  Value value64 =
137  rewriter.create<arith::ExtSIOp>(loc, rewriter.getI64Type(), value32);
138  Value multiplier64 = rewriter.create<arith::ExtSIOp>(
139  loc, rewriter.getI64Type(), multiplier32);
140  Value shift64 =
141  rewriter.create<arith::ExtSIOp>(loc, rewriter.getI64Type(), shift8);
142 
143  // Multiply as a pair of i64 values to guarantee the end value fits.
144  Value result64 = rewriter.create<arith::MulIOp>(loc, value64, multiplier64);
145  result64 = rewriter.create<arith::AddIOp>(loc, result64, round64);
146  result64 = rewriter.create<arith::ShRSIOp>(loc, result64, shift64);
147 
148  Value result32 =
149  rewriter.create<arith::TruncIOp>(loc, rewriter.getI32Type(), result64);
150 
151  rewriter.replaceOp(op, result32);
152  return success();
153  }
154 };
