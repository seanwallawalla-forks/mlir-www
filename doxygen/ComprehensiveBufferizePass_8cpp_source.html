75 void LinalgComprehensiveModuleBufferize::runOnOperation() {
76  BufferizationOptions options;
77  if (useAlloca) {
78  options.allocationFns->allocationFn = allocationFnUsingAlloca;
79  options.allocationFns->deallocationFn = [](OpBuilder &b, Location loc,
80  Value v) {};
81  }
82  // TODO: Change to memref::CopyOp (default memCpyFn).
83  options.allocationFns->memCpyFn = [](OpBuilder &b, Location loc, Value from,
84  Value to) {
85  b.create<linalg::CopyOp>(loc, from, to);
86  };
87 
88  options.allowReturnMemref = allowReturnMemref;
89  options.allowUnknownOps = allowUnknownOps;
90  options.analysisFuzzerSeed = analysisFuzzerSeed;
91  options.testAnalysisOnly = testAnalysisOnly;
92 
93  // Enable InitTensorOp elimination.
94  options.addPostAnalysisStep<
95  linalg_ext::InsertSliceAnchoredInitTensorEliminationStep>();
96  // TODO: Find a way to enable this step automatically when bufferizing tensor
97  // dialect ops.
98  options.addPostAnalysisStep<tensor_ext::InplaceInsertSliceOpAnalysis>();
99  options.addPostAnalysisStep<scf_ext::AssertDestinationPassingStyle>();
100 
101  ModuleOp moduleOp = getOperation();
102  applyEnablingTransformations(moduleOp);
103 
104  if (failed(runComprehensiveBufferize(moduleOp, options))) {
105  signalPassFailure();
106  return;
107  }
108 
109  if (options.testAnalysisOnly)
110  return;
111 
112  OpPassManager cleanupPipeline("builtin.module");
113  cleanupPipeline.addPass(createCanonicalizerPass());
114  cleanupPipeline.addPass(createCSEPass());
115  cleanupPipeline.addPass(createLoopInvariantCodeMotionPass());
116  (void)runPipeline(cleanupPipeline, moduleOp);
117 }
