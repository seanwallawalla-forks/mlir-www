88 /// This is the concrete base class that holds the operation pointer and has
89 /// non-generic methods that only depend on State (to avoid having them
90 /// instantiated on template types that don't affect them.
91 ///
92 /// This also has the fallback implementations of customization hooks for when
93 /// they aren't customized.
94 class OpState {
95 public:
96  /// Ops are pointer-like, so we allow conversion to bool.
97  explicit operator bool() { return getOperation() != nullptr; }
98 
99  /// This implicitly converts to Operation*.
100  operator Operation *() const { return state; }
101 
102  /// Shortcut of `->` to access a member of Operation.
103  Operation *operator->() const { return state; }
104 
105  /// Return the operation that this refers to.
106  Operation *getOperation() { return state; }
107 
108  /// Return the context this operation belongs to.
109  MLIRContext *getContext() { return getOperation()->getContext(); }
110 
111  /// Print the operation to the given stream.
112  void print(raw_ostream &os, OpPrintingFlags flags = llvm::None) {
113  state->print(os, flags);
114  }
115  void print(raw_ostream &os, AsmState &asmState,
116  OpPrintingFlags flags = llvm::None) {
117  state->print(os, asmState, flags);
118  }
119 
120  /// Dump this operation.
121  void dump() { state->dump(); }
122 
123  /// The source location the operation was defined or derived from.
124  Location getLoc() { return state->getLoc(); }
125 
126  /// Return true if there are no users of any results of this operation.
127  bool use_empty() { return state->use_empty(); }
128 
129  /// Remove this operation from its parent block and delete it.
130  void erase() { state->erase(); }
131 
132  /// Emit an error with the op name prefixed, like "'dim' op " which is
133  /// convenient for verifiers.
134  InFlightDiagnostic emitOpError(const Twine &message = {});
135 
136  /// Emit an error about fatal conditions with this operation, reporting up to
137  /// any diagnostic handlers that may be listening.
138  InFlightDiagnostic emitError(const Twine &message = {});
139 
140  /// Emit a warning about this operation, reporting up to any diagnostic
141  /// handlers that may be listening.
142  InFlightDiagnostic emitWarning(const Twine &message = {});
143 
144  /// Emit a remark about this operation, reporting up to any diagnostic
145  /// handlers that may be listening.
146  InFlightDiagnostic emitRemark(const Twine &message = {});
147 
148  /// Walk the operation by calling the callback for each nested operation
149  /// (including this one), block or region, depending on the callback provided.
150  /// Regions, blocks and operations at the same nesting level are visited in
151  /// lexicographical order. The walk order for enclosing regions, blocks and
152  /// operations with respect to their nested ones is specified by 'Order'
153  /// (post-order by default). A callback on a block or operation is allowed to
154  /// erase that block or operation if either:
155  /// * the walk is in post-order, or
156  /// * the walk is in pre-order and the walk is skipped after the erasure.
157  /// See Operation::walk for more details.
158  template <WalkOrder Order = WalkOrder::PostOrder, typename FnT,
159  typename RetT = detail::walkResultType<FnT>>
160  RetT walk(FnT &&callback) {
161  return state->walk<Order>(std::forward<FnT>(callback));
162  }
163 
164  // These are default implementations of customization hooks.
165 public:
166  /// This hook returns any canonicalization pattern rewrites that the operation
167  /// supports, for use by the canonicalization pass.
168  static void getCanonicalizationPatterns(RewritePatternSet &results,
169  MLIRContext *context) {}
170 
171 protected:
172  /// If the concrete type didn't implement a custom verifier hook, just fall
173  /// back to this one which accepts everything.
174  LogicalResult verify() { return success(); }
175 
176  /// Unless overridden, the custom assembly form of an op is always rejected.
177  /// Op implementations should implement this to return failure.
178  /// On success, they should fill in result with the fields to use.
179  static ParseResult parse(OpAsmParser &parser, OperationState &result);
180 
181  // The fallback for the printer is to print it the generic assembly form.
182  static void print(Operation *op, OpAsmPrinter &p);
183  static void printOpName(Operation *op, OpAsmPrinter &p,
184  StringRef defaultDialect);
185 
186  /// Mutability management is handled by the OpWrapper/OpConstWrapper classes,
187  /// so we can cast it away here.
188  explicit OpState(Operation *state) : state(state) {}
189 
190 private:
191  Operation *state;
192 
193  /// Allow access to internal hook implementation methods.
194  friend RegisteredOperationName;
195 };

282 /// Helper class for implementing traits. Clients are not expected to interact
283 /// with this directly, so its members are all protected.
284 template <typename ConcreteType, template <typename> class TraitType>
285 class TraitBase {
286 protected:
287  /// Return the ultimate Operation being worked on.
288  Operation *getOperation() {
289  // We have to cast up to the trait type, then to the concrete type, then to
290  // the BaseState class in explicit hops because the concrete type will
291  // multiply derive from the (content free) TraitBase class, and we need to
292  // be able to disambiguate the path for the C++ compiler.
293  auto *trait = static_cast<TraitType<ConcreteType> *>(this);
294  auto *concrete = static_cast<ConcreteType *>(trait);
295  auto *base = static_cast<OpState *>(concrete);
296  return base->getOperation();
297  }
298 };

861  private:
862  using Base = SingleBlock<ConcreteType>;
863  /// Builds a terminator operation without relying on OpBuilder APIs to avoid
864  /// cyclic header inclusion.
865  static Operation *buildTerminator(OpBuilder &builder, Location loc) {
866  OperationState state(loc, TerminatorOpType::getOperationName());
867  TerminatorOpType::build(builder, state);
868  return Operation::create(state);
869  }

871  public:
872  /// The type of the operation used as the implicit terminator type.
873  using ImplicitTerminatorOpT = TerminatorOpType;

876  if (failed(Base::verifyTrait(op)))
877  return failure();

879  Region &region = op->getRegion(i);
880  // Empty regions are fine.
881  if (region.empty())
882  continue;

1284 /// This trait tags element-wise ops on vectors or tensors.
1285 ///
1286 /// NOTE: Not all ops that are "elementwise" in some abstract sense satisfy this
1287 /// trait. In particular, broadcasting behavior is not allowed.
1288 ///
1289 /// An `Elementwise` op must satisfy the following properties:
1290 ///
1291 /// 1. If any result is a vector/tensor then at least one operand must also be a
1292 /// vector/tensor.
1293 /// 2. If any operand is a vector/tensor then there must be at least one result
1294 /// and all results must be vectors/tensors.
1295 /// 3. All operand and result vector/tensor types must be of the same shape. The
1296 /// shape may be dynamic in which case the op's behaviour is undefined for
1297 /// non-matching shapes.
1298 /// 4. The operation must be elementwise on its vector/tensor operands and
1299 /// results. When applied to single-element vectors/tensors, the result must
1300 /// be the same per elememnt.
1301 ///
1302 /// TODO: Avoid hardcoding vector/tensor, and generalize this trait to a new
1303 /// interface `ElementwiseTypeInterface` that describes the container types for
1304 /// which the operation is elementwise.
1305 ///
1306 /// Rationale:
1307 /// - 1. and 2. guarantee a well-defined iteration space and exclude the cases
1308 /// of 0 non-scalar operands or 0 non-scalar results, which complicate a
1309 /// generic definition of the iteration space.
1310 /// - 3. guarantees that folding can be done across scalars/vectors/tensors with
1311 /// the same pattern, as otherwise lots of special handling for type
1312 /// mismatches would be needed.
1313 /// - 4. guarantees that no error handling is needed. Higher-level dialects
1314 /// should reify any needed guards or error handling code before lowering to
1315 /// an `Elementwise` op.
1316 template <typename ConcreteType>
1317 struct Elementwise : public TraitBase<ConcreteType, Elementwise> {
1318  static LogicalResult verifyTrait(Operation *op) {
1319  return ::mlir::OpTrait::impl::verifyElementwise(op);
1320  }
1321 };

1323 /// This trait tags `Elementwise` operatons that can be systematically
1324 /// scalarized. All vector/tensor operands and results are then replaced by
1325 /// scalars of the respective element type. Semantically, this is the operation
1326 /// on a single element of the vector/tensor.
1327 ///
1328 /// Rationale:
1329 /// Allow to define the vector/tensor semantics of elementwise operations based
1330 /// on the same op's behavior on scalars. This provides a constructive procedure
1331 /// for IR transformations to, e.g., create scalar loop bodies from tensor ops.
1332 ///
1333 /// Example:
1334 /// ```
1335 /// %tensor_select = "std.select"(%pred_tensor, %true_val, %false_val)
1336 /// : (tensor<?xi1>, tensor<?xf32>, tensor<?xf32>)
1337 /// -> tensor<?xf32>
1338 /// ```
1339 /// can be scalarized to
1340 ///
1341 /// ```
1342 /// %scalar_select = "std.select"(%pred, %true_val_scalar, %false_val_scalar)
1343 /// : (i1, f32, f32) -> f32
1344 /// ```

1348  static_assert(

1469 /// Returns the result of folding a trait that implements a `foldTrait` function
1470 /// that is specialized for operations that have a single result.
1471 template <typename Trait>
1472 static std::enable_if_t<detect_has_single_result_fold_trait<Trait>::value,
1473  LogicalResult>
1474 foldTrait(Operation *op, ArrayRef<Attribute> operands,
1475  SmallVectorImpl<OpFoldResult> &results) {
1476  assert(op->hasTrait<OpTrait::OneResult>() &&
1477  "expected trait on non single-result operation to implement the "
1478  "general `foldTrait` method");
1479  // If a previous trait has already been folded and replaced this operation, we
1480  // fail to fold this trait.
1481  if (!results.empty())
1482  return failure();
1483 
1484  if (OpFoldResult result = Trait::foldTrait(op, operands)) {
1485  if (result.template dyn_cast<Value>() != op->getResult(0))
1486  results.push_back(result);
1487  return success();
1488  }
1489  return failure();
1490 }

1498  // fail to fold this trait.
