52 /// This class represents a successor of a region. A region successor can either
53 /// be another region, or the parent operation. If the successor is a region,
54 /// this class represents the destination region, as well as a set of arguments
55 /// from that region that will be populated when control flows into the region.
56 /// If the successor is the parent operation, this class represents an optional
57 /// set of results that will be populated when control returns to the parent
58 /// operation.
59 ///
60 /// This interface assumes that the values from the current region that are used
61 /// to populate the successor inputs are the operands of the return-like
62 /// terminator operations in the blocks within this region.
63 class RegionSuccessor {
64 public:
65  /// Initialize a successor that branches to another region of the parent
66  /// operation.
67  RegionSuccessor(Region *region, Block::BlockArgListType regionInputs = {})
68  : region(region), inputs(regionInputs) {}
69  /// Initialize a successor that branches back to/out of the parent operation.
70  RegionSuccessor(Optional<Operation::result_range> results = {})
71  : region(nullptr), inputs(results ? ValueRange(*results) : ValueRange()) {
72  }
73 
74  /// Return the given region successor. Returns nullptr if the successor is the
75  /// parent operation.
76  Region *getSuccessor() const { return region; }
77 
78  /// Return true if the successor is the parent operation.
79  bool isParent() const { return region == nullptr; }
80 
81  /// Return the inputs to the successor that are remapped by the exit values of
82  /// the current region.
83  ValueRange getSuccessorInputs() const { return inputs; }
84 
85 private:
86  Region *region;
87  ValueRange inputs;
88 };

124 namespace OpTrait {
125 /// This trait indicates that a terminator operation is "return-like". This
126 /// means that it exits its current region and forwards its operands as "exit"
127 /// values to the parent region. Operations with this trait are not permitted to
128 /// contain successors or produce results.
129 template <typename ConcreteType>
130 struct ReturnLike : public TraitBase<ConcreteType, ReturnLike> {
131  static LogicalResult verifyTrait(Operation *op) {
132  static_assert(ConcreteType::template hasTrait<IsTerminator>(),
133  "expected operation to be a terminator");
134  static_assert(ConcreteType::template hasTrait<ZeroResult>(),
135  "expected operation to have zero results");
136  static_assert(ConcreteType::template hasTrait<ZeroSuccessor>(),
137  "expected operation to have zero successors");
138  return success();
139  }
140 };
141 } // namespace OpTrait
