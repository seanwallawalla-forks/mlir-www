31 /// ToMemrefOp casts a tensor into a memref. The resulting memref is the memory
32 /// location of the incoming tensor once it will be bufferized. In the anlysis,
33 /// the incoming tensor is assumed to bufferize to a memory read and to an
34 /// inplace memory write, since it is unknown what will happen to the resulting
35 /// memref.
36 struct ToMemrefOpInterface
37  : public BufferizableOpInterface::ExternalModel<ToMemrefOpInterface,
38  bufferization::ToMemrefOp> {
39  bool bufferizesToMemoryRead(Operation *op, OpOperand &opOperand) const {
40  // It is unknown whether the resulting MemRef will be read or not.
41  return true;
42  }
43 
44  OpResult getAliasingOpResult(Operation *op, OpOperand &opOperand) const {
45  return OpResult();
46  }
47 
48  LogicalResult bufferize(Operation *op, OpBuilder &b,
49  BufferizationState &state) const {
50  return success();
51  }
52 };

54 /// ToTensorOp conceptually loads a tensor from a memory location. Such ops do
55 /// not lower any further, and they should have disappeared by the time the
56 /// input is fully bufferized.
57 ///
58 /// The analysis has no information about the memref that is loaded from by the
59 /// ToTensorOp. We have to assume that the loaded tensor may after bufferization
60 /// potentially alias with any other bufferized tensor. Since ToTensorOp and
61 /// ToMemrefOp have no aliasing OpOperand/OpResult pairs, this cannot be encoded
62 /// directly in the analysis. However, declaring ToTensorOp results as not
63 /// writable also enforces a buffer copy and has the same effect.
64 struct ToTensorOpInterface
65  : public BufferizableOpInterface::ExternalModel<ToTensorOpInterface,
66  bufferization::ToTensorOp> {
67  LogicalResult bufferize(Operation *op, OpBuilder &b,
68  BufferizationState &state) const {
69  auto tensorLoadOp = cast<bufferization::ToTensorOp>(op);
70  state.mapBuffer(tensorLoadOp.result(), tensorLoadOp.memref());
71  return success();
72  }
73 
74  bool isWritable(Operation *op, Value value, BufferizationState &state) const {
75  // It is unknown whether the MemRef operand is writable or not.
76  return false;
77  }
78 };
