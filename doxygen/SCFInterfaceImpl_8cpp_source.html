20 struct ExecuteRegionOpInterface
21  : public BufferizableOpInterface::ExternalModel<ExecuteRegionOpInterface,
22  scf::ExecuteRegionOp> {
23  SmallVector<OpOperand *> getAliasingOpOperand(Operation *op,
24  OpResult opResult) const {
25  // ExecuteRegionOps do not have tensor OpOperands. The yielded value can be
26  // any SSA value that is in scope. To allow for use-def chain traversal
27  // through ExecuteRegionOps in the analysis, the corresponding yield value
28  // is considered to be aliasing with the result.
29  auto executeRegionOp = cast<scf::ExecuteRegionOp>(op);
30  size_t resultNum = std::distance(op->getOpResults().begin(),
31  llvm::find(op->getOpResults(), opResult));
32  assert(executeRegionOp.region().getBlocks().size() == 1 &&
33  "expected exactly 1 block");
34  auto yieldOp = dyn_cast<scf::YieldOp>(
35  executeRegionOp.region().front().getTerminator());
36  assert(yieldOp && "expected scf.yield terminator in scf.execute_region");
37  return {&yieldOp->getOpOperand(resultNum)};
38  }
39 
40  bool mustBufferizeInPlace(Operation *op, OpResult opResult) const {
41  // ExecuteRegionOp results always bufferize in-place. Since they have no
42  // OpOperands, they are mostly ignored by the analysis once alias sets are
43  // set up.
44  return true;
45  }
46 
47  // TODO: For better bufferization results, this could return `true` only if
48  // there is a memory write in the region.
49  bool isMemoryWrite(Operation *op, OpResult opResult) const {
50  // Similar to scf.if, results of this op are always considered memory writes
51  // in the analysis. This is a useful pattern for all ops that have tensor
52  // OpResults but no tensor OpOperands. By default, `isMemoryWrite` is
53  // implemented in terms of `bufferizesToMemoryWrite`, which does not work on
54  // ops without OpOperands.
55  return true;
56  }

59  BufferizationState &state) const {
60  // TODO: Add bufferization support when needed. scf.execute_region should be
61  // bufferized similar to scf.if.

63  bool hasTensorReturnType = any_of(

65  if (hasTensorReturnType)

77 struct IfOpInterface
78  : public BufferizableOpInterface::ExternalModel<IfOpInterface, scf::IfOp> {
79  SmallVector<OpOperand *> getAliasingOpOperand(Operation *op,
80  OpResult opResult) const {
81  // IfOps do not have tensor OpOperands. The yielded value can be any SSA
82  // value that is in scope. To allow for use-def chain traversal through
83  // IfOps in the analysis, both corresponding yield values from the then/else
84  // branches are considered to be aliasing with the result.
85  auto ifOp = cast<scf::IfOp>(op);
86  size_t resultNum = std::distance(op->getOpResults().begin(),
87  llvm::find(op->getOpResults(), opResult));
88  return {&ifOp.thenYield()->getOpOperand(resultNum),
89  &ifOp.elseYield()->getOpOperand(resultNum)};
90  }
91 
92  // TODO: For better bufferization results, this could return `true` only if
93  // there is a memory write in one (or both) of the branches. Since this is not
94  // allowed at the moment, we should never encounter scf.ifs that yield
95  // unmodified tensors. Such scf.yield ops could just fold away.
96  bool isMemoryWrite(Operation *op, OpResult opResult) const {
97  // IfOp results are always considered memory writes in the analysis. This
98  // design decision simplifies the analysis considerably. E.g., consider the
99  // following test case:
100  //
101  // %0 = "some_writing_op" : tensor<?xf32>
102  // %r = scf.if %c -> (tensor<?xf32>) {
103  // scf.yield %0
104  // } else {
105  // %1 = "another_writing_op"(%0) : tensor<?xf32>
106  // }
107  // "some_reading_op"(%r)
108  //
109  // "another_writing_op" in the above example should be able to bufferize
110  // inplace in the absence of another read of %0. However, if the scf.if op
111  // would not be considered a "write", the analysis would detect the
112  // following conflict:
113  //
114  // * read = some_reading_op
115  // * lastWrite = %0 (Note: The last write of %r would be a set: {%0, %1}.)
116  // * conflictingWrite = %1
117  //
118  // For more details, check the "scf.IfOp" section of the design document.
119  return true;
120  }
121 
122  bool mustBufferizeInPlace(Operation *op, OpResult opResult) const {
123  // IfOp results always bufferize in-place. Since they have no OpOperands,
124  // they are mostly ignored by the analysis once alias sets are set up.
125  return true;
126  }
127 
128  LogicalResult bufferize(Operation *op, OpBuilder &b,
129  BufferizationState &state) const {
130  auto ifOp = cast<scf::IfOp>(op);
131 
132  // Bufferize then/else blocks.
133  if (failed(comprehensive_bufferize::bufferize(ifOp.thenBlock(), state)))
134  return failure();
135  if (failed(comprehensive_bufferize::bufferize(ifOp.elseBlock(), state)))
136  return failure();
137 
138  for (OpResult opResult : ifOp->getResults()) {
139  if (!opResult.getType().isa<TensorType>())
140  continue;
141  // TODO: Atm we bail on unranked TensorType because we don't know how to
142  // alloc an UnrankedMemRefType + its underlying ranked MemRefType.
143  assert(opResult.getType().isa<RankedTensorType>() &&
144  "unsupported unranked tensor");
145 
146  Value resultBuffer = state.getResultBuffer(opResult);
147  if (!resultBuffer)
148  return failure();
149 
150  state.mapBuffer(opResult, resultBuffer);
151  }
152 
153  return success();
154  }
155 
156  BufferRelation bufferRelation(Operation *op, OpResult opResult,
157  const BufferizationAliasInfo &aliasInfo) const {
158  // IfOp results are equivalent to their corresponding yield values if both
159  // yield values are equivalent to each other.
160  auto bufferizableOp = cast<BufferizableOpInterface>(op);
161  SmallVector<OpOperand *> yieldValues =
162  bufferizableOp.getAliasingOpOperand(opResult);
163  assert(yieldValues.size() == 2 && "expected 2 yield values");
164  bool equivalentYields = aliasInfo.areEquivalentBufferizedValues(
165  yieldValues[0]->get(), yieldValues[1]->get());
166  return equivalentYields ? BufferRelation::Equivalent : BufferRelation::None;
167  }

170 struct ForOpInterface
171  : public BufferizableOpInterface::ExternalModel<ForOpInterface,
172  scf::ForOp> {
173  bool bufferizesToMemoryRead(Operation *op, OpOperand &opOperand) const {
174  // scf::ForOp alone doesn't bufferize to a memory read, one of the uses of
175  // its matching bbArg may.
176  auto forOp = cast<scf::ForOp>(op);
177  return isValueRead(forOp.getRegionIterArgForOpOperand(opOperand));
178  }
179 
180  bool bufferizesToMemoryWrite(Operation *op, OpOperand &opOperand) const {
181  // Tensor iter_args of scf::ForOps are always considered as a write. This is
182  // to simplify the analysis.
183  // TODO: Consider doing sth. like isValueWritten.
184  return true;
185  }

187  OpResult getAliasingOpResult(Operation *op, OpOperand &opOperand) const {
188  auto forOp = cast<scf::ForOp>(op);
189  if (!opOperand.get().getType().isa<RankedTensorType>())
190  return OpResult();
191  return forOp.getResultForOpOperand(opOperand);
192  }
193 
194  BufferRelation bufferRelation(Operation *op, OpResult opResult,
195  const BufferizationAliasInfo &aliasInfo) const {
196  // ForOp results are equivalent to their corresponding init_args if the
197  // corresponding iter_args and yield values are equivalent.
198  auto forOp = cast<scf::ForOp>(op);
199  OpOperand &forOperand = forOp.getOpOperandForResult(opResult);
200  auto bbArg = forOp.getRegionIterArgForOpOperand(forOperand);
201  auto yieldOp = cast<scf::YieldOp>(&forOp.getLoopBody().front().back());
202  bool equivalentYield = aliasInfo.areEquivalentBufferizedValues(
203  bbArg, yieldOp->getOperand(opResult.getResultNumber()));
204  return equivalentYield ? BufferRelation::Equivalent : BufferRelation::None;
205  }
206 
207  bool isWritable(Operation *op, Value value, BufferizationState &state) const {
208  // Interestingly, scf::ForOp's bbArg can **always** be viewed
209  // inplace from the perspective of ops nested under:
210  // 1. Either the matching iter operand is not bufferized inplace and an
211  // alloc + optional copy makes the bbArg itself inplaceable.
212  // 2. Or the matching iter operand is bufferized inplace and bbArg just
213  // bufferizes to that too.
214  return true;
215  }
216 
217  LogicalResult bufferize(Operation *op, OpBuilder &b,
218  BufferizationState &state) const {
219  auto forOp = cast<scf::ForOp>(op);
220 
221  // Take a guard before anything else.
222  OpBuilder::InsertionGuard g(b);
223 
224  for (OpResult opResult : forOp->getResults()) {
225  if (!opResult.getType().isa<TensorType>())
226  continue;
227  // TODO: Atm we bail on unranked TensorType because we don't know how to
228  // alloc an UnrankedMemRefType + its underlying ranked MemRefType.
229  assert(opResult.getType().isa<RankedTensorType>() &&
230  "unsupported unranked tensor");
231 
232  // TODO: More general: Matching bbArg does not bufferize to a read.
233  Value resultBuffer = state.getResultBuffer(opResult);
234  if (!resultBuffer)
235  return failure();
236 
237  OpOperand &opOperand = forOp.getOpOperandForResult(opResult);
238  BlockArgument bbArg = forOp.getRegionIterArgForOpOperand(opOperand);
239  state.mapBuffer(bbArg, resultBuffer);
240  state.mapBuffer(opResult, resultBuffer);
241  }
242 
243  // Bufferize loop body.
244  if (failed(comprehensive_bufferize::bufferize(&forOp.region(), state)))
245  return failure();
246 
247  // Finish bufferizing scf::ForOp.
248  auto yieldOp = cast<scf::YieldOp>(&forOp.region().front().back());
249  for (OpOperand &operand : yieldOp->getOpOperands()) {
250  auto tensorType = operand.get().getType().dyn_cast<TensorType>();
251  if (!tensorType)
252  continue;
253 
254  OpOperand &forOperand = forOp.getOpOperandForResult(
255  forOp->getResult(operand.getOperandNumber()));
256  auto bbArg = forOp.getRegionIterArgForOpOperand(forOperand);
257 
258  // Buffers are equivalent so the work is already done and we just yield
259  // the bbArg so that it later canonicalizes away.
260  operand.set(bbArg);
261  }
262  return success();
263  }

266 LogicalResult mlir::linalg::comprehensive_bufferize::scf_ext::
267  AssertDestinationPassingStyle::run(FuncOp funcOp, BufferizationState &state,
268  SmallVector<Operation *> &newOps) {
269  LogicalResult status = success();
270  funcOp->walk([&](scf::YieldOp yieldOp) {
271  auto forOp = dyn_cast<scf::ForOp>(yieldOp->getParentOp());
272  if (!forOp)
273  return WalkResult::advance();
274 
275  for (OpOperand &operand : yieldOp->getOpOperands()) {
276  auto tensorType = operand.get().getType().dyn_cast<TensorType>();
277  if (!tensorType)
278  continue;
279 
280  OpOperand &forOperand = forOp.getOpOperandForResult(
281  forOp->getResult(operand.getOperandNumber()));
282  auto bbArg = forOp.getRegionIterArgForOpOperand(forOperand);
283  if (!state.aliasInfo.areEquivalentBufferizedValues(operand.get(),
284  bbArg)) {
285  // TODO: this could get resolved with copies but it can also turn into
286  // swaps so we need to be careful about order of copies.
287  status =
288  yieldOp->emitError()
289  << "Yield operand #" << operand.getOperandNumber()
290  << " does not bufferize to an equivalent buffer to the matching"
291  << " enclosing scf::for operand";
292  return WalkResult::interrupt();
293  }
294  }
295 
296  return WalkResult::advance();
297  });
298  return status;
299 }
