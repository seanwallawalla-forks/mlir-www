128 Type mlir::linalg::LinalgDialect::parseType(DialectAsmParser &parser) const {
129  // Parse the main keyword for the type.
130  StringRef keyword;
131  if (parser.parseKeyword(&keyword))
132  return Type();
133  MLIRContext *context = getContext();
134 
135  // Handle 'range' types.
136  if (keyword == "range")
137  return RangeType::get(context);
138 
139  parser.emitError(parser.getNameLoc(), "unknown Linalg type: " + keyword);
140  return Type();
141 }

151 LogicalResult LinalgDialect::verifyOperationAttribute(Operation *op,
152  NamedAttribute attr) {
153  using comprehensive_bufferize::BufferizableOpInterface;
154 
155  if (attr.getName() == BufferizableOpInterface::kInplaceableAttrName) {
156  if (!attr.getValue().isa<BoolAttr>()) {
157  return op->emitError()
158  << "'" << BufferizableOpInterface::kInplaceableAttrName
159  << "' is expected to be a boolean attribute";
160  }
161  if (!op->hasTrait<OpTrait::FunctionLike>())
162  return op->emitError() << "expected " << attr.getName()
163  << " to be used on function-like operations";
164  return success();
165  }
166  if (attr.getName() == BufferizableOpInterface::kBufferLayoutAttrName) {
167  if (!attr.getValue().isa<AffineMapAttr>()) {
168  return op->emitError()
169  << "'" << BufferizableOpInterface::kBufferLayoutAttrName
170  << "' is expected to be a affine map attribute";
171  }
172  if (!op->hasTrait<OpTrait::FunctionLike>())
173  return op->emitError() << "expected " << attr.getName()
174  << " to be used on function-like operations";
175  return success();
176  }
177  if (attr.getName() == LinalgDialect::kMemoizedIndexingMapsAttrName)
178  return success();
179  return op->emitError() << "attribute '" << attr.getName()
180  << "' not supported by the linalg dialect";
181 }
