24 /// Helper function for LinalgOp bufferization.
25 /// When allocating a new buffer, analyze whether `op` wants to read form that
26 /// buffer. Only in that case, a copy of the result buffer may be needed.
27 static LogicalResult
28 allocateBuffersForResults(OpBuilder &b, Location loc, LinalgOp op,
29  SmallVectorImpl<Value> &resultBuffers,
30  BufferizationState &state) {
31  // Take a guard before anything else.
32  OpBuilder::InsertionGuard g(b);
33  b.setInsertionPoint(op);
34 
35  // TODO: provide the proper interface to iterate on OpResults and get the
36  // matching OpOperands.
37  for (OpOperand *opOperand : op.getOutputOperands()) {
38  OpResult opResult = cast<BufferizableOpInterface>(op.getOperation())
39  .getAliasingOpResult(*opOperand);
40  assert(opResult && "could not find correspond OpResult");
41  Value resultBuffer = state.getResultBuffer(opResult);
42  if (!resultBuffer)
43  return failure();
44  resultBuffers.push_back(resultBuffer);
45  }
46 
47  if (op->getNumResults())
48  state.mapBuffer(op->getResults(), resultBuffers);
49 
50  return success();
51 }

53 /// Generic conversion for any LinalgOp on tensors.
54 static LogicalResult bufferizeLinalgOp(OpBuilder &b, LinalgOp op,
55  BufferizationState &state) {
56  // Take a guard before anything else.
57  OpBuilder::InsertionGuard g(b);
58 
59  // Ensure op has only tensors. Allow mixed tensor-buffer mode on a per-need
60  // basis.
61  if (!op.hasTensorSemantics())
62  return op->emitError() << "op does not have tensor semantics";
63 
64  Location loc = op.getLoc();
65  SmallVector<Value> newInputBuffers;
66  newInputBuffers.reserve(op.getNumInputs());
67  for (OpOperand *opOperand : op.getInputOperands()) {
68  if (op.isScalar(opOperand)) {
69  newInputBuffers.push_back(opOperand->get());
70  continue;
71  }
72  newInputBuffers.push_back(state.lookupBuffer(opOperand->get()));
73  }
74  SmallVector<Value> newOutputBuffers;
75  // Try to allocate new buffers depending on op's inplace semantics.
76  if (failed(allocateBuffersForResults(b, loc, op, newOutputBuffers, state)))
77  return failure();
78 
79  // Clone the newly bufferized op.
80  SmallVector<Value> newOperands = newInputBuffers;
81  newOperands.append(newOutputBuffers.begin(), newOutputBuffers.end());
82 
83  // Set insertion point now that potential alloc/dealloc are introduced.
84  b.setInsertionPoint(op);
85  op.clone(b, loc, /*resultTypes=*/TypeRange{}, newOperands);
86 
87  // Replace the results of the old op with the new output buffers.
88  if (op->getNumResults())
89  state.mapBuffer(op->getResults(), newOutputBuffers);
90 
91  // The original op will be DCE'd away later.
92 
93  return success();
94 }

97 struct LinalgOpInterface
98  : public BufferizableOpInterface::ExternalModel<LinalgOpInterface<OpTy>,
99  OpTy> {
100  bool bufferizesToMemoryRead(Operation *op, OpOperand &opOperand) const {
101  auto genericOp = cast<linalg::LinalgOp>(op);
102  return (genericOp.isInputTensor(&opOperand) ||
103  genericOp.isInitTensor(&opOperand)) &&
104  genericOp.payloadUsesValueFromOperand(&opOperand);
105  }
106 
107  bool bufferizesToMemoryWrite(Operation *op, OpOperand &opOperand) const {
108  auto genericOp = cast<linalg::LinalgOp>(op);
109  return genericOp.isOutputTensor(&opOperand);
110  }
111 
112  SmallVector<OpOperand *> getAliasingOpOperand(Operation *op,
113  OpResult opResult) const {
114  auto genericOp = cast<linalg::LinalgOp>(op);
115  return {genericOp.getOutputTensorOperands()[opResult.getResultNumber()]};
116  }
117 
118  OpResult getAliasingOpResult(Operation *op, OpOperand &opOperand) const {
119  auto genericOp = cast<linalg::LinalgOp>(op);
120  if (!opOperand.get().getType().isa<RankedTensorType>())
121  return OpResult();
122  // For now assume inputs are never inplaceable.
123  // TODO: refine this.
124  if (opOperand.getOperandNumber() < genericOp.getNumInputs())
125  return OpResult();
126  int64_t outputOperandIndex =
127  opOperand.getOperandNumber() - genericOp.getNumInputs();
128  int64_t numOutputBuffers = 0;
129  for (unsigned idx = 0; idx < outputOperandIndex; ++idx)
130  if (!genericOp.getOutputOperand(idx)->get().getType().isa<TensorType>())
131  ++numOutputBuffers;
132  return genericOp->getResult(outputOperandIndex - numOutputBuffers);
133  }
134 
135  BufferRelation bufferRelation(Operation *op, OpResult opResult,
136  const BufferizationAliasInfo &aliasInfo) const {
137  return BufferRelation::Equivalent;
138  }
139 
140  LogicalResult bufferize(Operation *op, OpBuilder &b,
141  BufferizationState &state) const {
142  return bufferizeLinalgOp(b, cast<LinalgOp>(op), state);
143  }

208  LogicalResult bufferize(Operation *op, OpBuilder &b,
209  BufferizationState &state) const {
210  auto tiledLoopOp = cast<linalg::TiledLoopOp>(op);
211 
212  // Take a guard before anything else.
213  OpBuilder::InsertionGuard g(b);
214 
215  // Allocate output buffers if needed, forward output tensor args to the
216  // terminator.
217  Operation *yieldOp = tiledLoopOp.getBody()->getTerminator();
218  Block *body = tiledLoopOp.getBody();
219 
220  // Take copies of the old input and output operands, so we can insert
221  // inplace easily.
222  auto oldInputs = llvm::to_vector<4>(tiledLoopOp.inputs());
223  auto oldOutputs = llvm::to_vector<4>(tiledLoopOp.outputs());
224 
225  int numLoops = tiledLoopOp.getNumLoops();
226  int numControlOperands = tiledLoopOp.getNumControlOperands();
227 
228  // Add buffers for outputs and the corresponding block arguments.
229  // Keep separate iterators to increment without further leaking impl.
230  // details. Start with outputs to avoid interference from new input buffers.
231  int numNewOutputBuffers = 0;
232  int resultIndex = 0;
233  int oldOutputBBArgIndex = numLoops + oldInputs.size();
234  int nextOutputBBArgIndex = numLoops + oldInputs.size() + oldOutputs.size();
235  int nextOutputOperandIndex =
236  numControlOperands + oldInputs.size() + oldOutputs.size();
237  for (Value oldOutputTensor : oldOutputs) {
238  if (!oldOutputTensor.getType().isa<TensorType>()) {
239  // Skip and increment the old bbarg index only.
240  ++oldOutputBBArgIndex;
241  // Do not increment resultIndex as only tensors are returned.
242  // TODO: better interface to avoid leaking such impl details.
243  continue;
244  }
245 
246  assert(oldOutputTensor.getType().isa<RankedTensorType>() &&
247  "bufferizable output must be a ranked tensor");
248 
249  const OpResult &opResult = tiledLoopOp->getResult(resultIndex);
250  OpOperand &yieldOperand = yieldOp->getOpOperand(resultIndex);
251  Value resultBuffer = state.getResultBuffer(opResult);
252  if (!resultBuffer)
253  return failure();
254 
255  // Insert mapping and aliasing info.
256  state.mapBuffer(opResult, resultBuffer);
257 
258  // Insert new operand and bbArg.
259  tiledLoopOp->insertOperands(nextOutputOperandIndex, resultBuffer);
260  BlockArgument newBufferBBArg =
261  body->insertArgument(nextOutputBBArgIndex, resultBuffer.getType());
262  BlockArgument oldTensorBBArg = body->getArgument(oldOutputBBArgIndex);
263  // Insert mapping and aliasing info.
264  state.mapBuffer(oldTensorBBArg, newBufferBBArg);
265 
266  // Set operand of `linalg.yield` to the bbArg so it just canonicalizes
267  // away later.
268  yieldOperand.set(oldTensorBBArg);
269 
270  // Increment indices.
271  ++numNewOutputBuffers;
272  ++resultIndex;
273  ++oldOutputBBArgIndex;
274  ++nextOutputBBArgIndex;
275  ++nextOutputOperandIndex;
276  }
277 
278  // Add buffers for inputs and the corresponding block arguments.
279  // Keep separate iterators to increment without further leaking impl.
280  // details.
281  int numNewInputBuffers = 0;
282  int oldInputBBArgIndex = numLoops;
283  int nextInputBBArgIndex = numLoops + oldInputs.size();
284  int nextInputOperandIndex = numControlOperands + oldInputs.size();
285  for (Value oldInputTensor : oldInputs) {
286  if (!oldInputTensor.getType().isa<TensorType>()) {
287  // Skip and increment the old bbarg index only.
288  ++oldInputBBArgIndex;
289  continue;
290  }
291 
292  Value inputBuffer = state.lookupBuffer(oldInputTensor);
293 
294  // Insert new operand and bbArg.
295  tiledLoopOp->insertOperands(nextInputOperandIndex, inputBuffer);
296  BlockArgument newBufferBBArg =
297  body->insertArgument(nextInputBBArgIndex, inputBuffer.getType());
298  BlockArgument oldTensorBBArg = body->getArgument(oldInputBBArgIndex);
299 
300  // Insert mapping and aliasing info.
301  state.mapBuffer(oldTensorBBArg, newBufferBBArg);
302 
303  // Increment indices.
304  ++numNewInputBuffers;
305  ++oldInputBBArgIndex;
306  ++nextInputBBArgIndex;
307  ++nextInputOperandIndex;
308  }
309 
310  // Update segment sizes.
311  // TODO: Helper method to avoid leaking impl details.
312  tiledLoopOp->setAttr(
313  TiledLoopOp::getOperandSegmentSizeAttr(),
314  b.getI32VectorAttr(
315  {numLoops, numLoops, numLoops,
316  static_cast<int>(oldInputs.size()) + numNewInputBuffers,
317  static_cast<int>(oldOutputs.size()) + numNewOutputBuffers}));
318 
319  // Bufferize loop body.
320  return comprehensive_bufferize::bufferize(&tiledLoopOp.region(), state);
321  }

374 /// Try to eliminate InitTensorOps inside funcOp. An InitTensorOp is replaced
375 /// with the the result of `rewriteFunc` if it is anchored on a matching
376 /// OpOperand. "Anchored" means that there is a path on the reverse SSA use-def
377 /// chain, starting from the OpOperand and always following the aliasing
378 /// OpOperand, that eventually ends at a single InitTensorOp.
379 LogicalResult mlir::linalg::comprehensive_bufferize::linalg_ext::
380  InitTensorEliminationStep::eliminateInitTensors(
381  FuncOp funcOp, BufferizationState &state,
382  std::function<bool(OpOperand &)> anchorMatchFunc,
383  std::function<Value(OpBuilder &, Location, OpOperand &)> rewriteFunc,
384  SmallVector<Operation *> &newOps) {
385  OpBuilder b(funcOp->getContext());
386  BufferizationAliasInfo &aliasInfo = state.aliasInfo;
387 
388  WalkResult status = funcOp->walk([&](Operation *op) {
389  for (OpOperand &operand : op->getOpOperands()) {
390  // Is this a matching OpOperand?
391  if (!anchorMatchFunc(operand))
392  continue;
393 
394  SetVector<Value> maybeInitTensor =
395  findValueInReverseUseDefChain(operand.get(), [&](Value val) {
396  // Continue traversal until this function returns true.
397  OpResult opResult = val.dyn_cast<OpResult>();
398  if (!opResult)
399  return true;
400  if (!aliasInfo.isInPlace(opResult))
401  return true;
402  // Only equivalent tensors are supported at the moment.
403  // TODO: Support cases such as extract_slice(init_tensor).
404  SmallVector<OpOperand *> opOperands =
405  getAliasingOpOperand(opResult);
406  if (!llvm::all_of(opOperands, [&](OpOperand *operand) {
407  return aliasInfo.areEquivalentBufferizedValues(operand->get(),
408  opResult);
409  }))
410  return true;
411  return false;
412  });
413 
414  // Replace only if the reverse use-def chain ends at exactly one
415  // InitTensorOp.
416  if (maybeInitTensor.size() != 1 ||
417  !maybeInitTensor.front().getDefiningOp<InitTensorOp>())
418  return WalkResult::skip();
419  Value initTensor = maybeInitTensor.front();
420 
421  // Create a replacement for the InitTensorOp.
422  b.setInsertionPoint(initTensor.getDefiningOp());
423  Value replacement = rewriteFunc(b, initTensor.getLoc(), operand);
424  if (!replacement)
425  continue;
426 
427  // Uses of the InitTensorOp are replaced here, but the op is not deleted.
428  // InitTensorOps without uses are ignored by the bufferization.
429  initTensor.replaceAllUsesWith(replacement);
430  aliasInfo.createAliasInfoEntry(replacement);
431  aliasInfo.unionAliasSets(initTensor, replacement);
432  aliasInfo.unionEquivalenceClasses(initTensor, replacement);
433 
434  // Register replacement ops.
435  if (Operation *newOp = replacement.getDefiningOp())
436  newOps.push_back(newOp);
437  }
438 
439  // Advance to the next operation.
440  return WalkResult::advance();
441  });
442 
443  return failure(status.wasInterrupted());
444 }

446 /// Try to eliminate InitTensorOps inside funcOp. An InitTensorOp can be
447 /// eliminated if it is eventually inserted into another tensor (and some other
448 /// conditions are met).
449 ///
450 /// E.g.:
451 /// %0 = linalg.init_tensor
452 /// %1 = linalg.fill(%cst, %0) {inplace = [true]}
453 /// %2 = tensor.insert_slice %1 into %t[10][20][1]
454 ///
455 /// InitTensorOp elimination will try to fill %t inplace instead of filling a
456 /// new allocation %0 and inserting it into %t. This is done by replacing the
457 /// InitTensorOp with:
458 ///
459 /// %0 = tensor.extract_slice %t[10][20][1]
460 ///
461 /// The analysis looks for matching ExtractSliceOp/InsertSliceOp pairs and lets
462 /// those bufferize inplace in the absence of other conflicts.
463 ///
464 /// Starting from an InsertSliceOp, an InitTensorOp at the end of the insert
465 /// source's reverse use-def chain is eliminated if:
466 /// * The InsertSliceOp was decided to bufferize inplace.
467 /// * On the reverse use-def chain path from the InsertSliceOp to the
468 /// InitTensorOp, all ops were decided to bufferize inplace and the buffer
469 /// relation is "equivalent" (TODO: can be relaxed if needed).
470 /// * The reverse use-def chain has exactly one end, which is the InitTensorOp.
471 ///
472 /// Note that the newly inserted ExtractSliceOp may have to bufferize
473 /// out-of-place due to RaW conflicts.
474 LogicalResult mlir::linalg::comprehensive_bufferize::linalg_ext::
475  InsertSliceAnchoredInitTensorEliminationStep::run(
476  FuncOp funcOp, BufferizationState &state,
477  SmallVector<Operation *> &newOps) {
478  return eliminateInitTensors(
479  funcOp, state,
480  [&](OpOperand &operand) {
481  auto insertSliceOp =
482  dyn_cast<tensor::InsertSliceOp>(operand.getOwner());
483  if (!insertSliceOp)
484  return false;
485  // Only inplace bufferized InsertSliceOps are eligible.
486  if (!state.aliasInfo.isInPlace(insertSliceOp->getOpResult(0)))
487  return false;
488  return &operand == &insertSliceOp->getOpOperand(0) /*source*/;
489  },
490  [](OpBuilder &b, Location loc, OpOperand &operand) {
491  auto insertSliceOp = cast<tensor::InsertSliceOp>(operand.getOwner());
492  auto extractOp = b.create<tensor::ExtractSliceOp>(
493  loc, insertSliceOp.dest(), insertSliceOp.getMixedOffsets(),
494  insertSliceOp.getMixedSizes(), insertSliceOp.getMixedStrides());
495  return extractOp.result();
496  },
497  newOps);
498 }
